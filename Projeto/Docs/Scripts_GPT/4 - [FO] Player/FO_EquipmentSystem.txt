#------------------------------------------------------------------------------
# Funções:
# - Equipar armas usando :weapon
# - Equipar armaduras usando :armor
# - Desequipar e devolver ao inventário correto
# - Ler $data_weapons e $data_armors do RMXP
# - Controlar slots básicos de equipamento
# - Exibir debug com nomes dos equipamentos
# - Calcular bônus de ATK, PDEF e MDEF
#------------------------------------------------------------------------------
# FO_EngineBoot:
# - NÃO precisa registrar este sistema no FO_EngineBoot por enquanto.
# - Este sistema não usa update contínuo.

module FO_EquipmentConfig

  ENABLE_EQUIPMENT = true
  ENABLE_LOG = true

  SLOT_WEAPON    = :weapon
  SLOT_SHIELD    = :shield
  SLOT_HELMET    = :helmet
  SLOT_ARMOR     = :armor
  SLOT_BOOTS     = :boots
  SLOT_ACCESSORY = :accessory

end

module FO_EquipmentSystem

  @equips = {
    FO_EquipmentConfig::SLOT_WEAPON    => nil,
    FO_EquipmentConfig::SLOT_SHIELD    => nil,
    FO_EquipmentConfig::SLOT_HELMET    => nil,
    FO_EquipmentConfig::SLOT_ARMOR     => nil,
    FO_EquipmentConfig::SLOT_BOOTS     => nil,
    FO_EquipmentConfig::SLOT_ACCESSORY => nil
  }

  #===========================================================================
  # ■ setup
  #===========================================================================
  def self.setup

    @equips = {} if @equips.nil?

    default_slots.each do |slot|

      @equips[slot] = nil unless @equips.has_key?(slot)

    end

  end

  #===========================================================================
  # ■ default_slots
  #===========================================================================
  def self.default_slots

    return [
      FO_EquipmentConfig::SLOT_WEAPON,
      FO_EquipmentConfig::SLOT_SHIELD,
      FO_EquipmentConfig::SLOT_HELMET,
      FO_EquipmentConfig::SLOT_ARMOR,
      FO_EquipmentConfig::SLOT_BOOTS,
      FO_EquipmentConfig::SLOT_ACCESSORY
    ]

  end

  #===========================================================================
  # ■ equips
  #===========================================================================
  def self.equips

    setup

    return @equips

  end

  #===========================================================================
  # ■ valid_slot?
  #===========================================================================
  def self.valid_slot?(slot)

    return false if slot.nil?

    setup

    return @equips.has_key?(slot.to_sym)

  end

  #===========================================================================
  # ■ equip_weapon
  #===========================================================================
  def self.equip_weapon(weapon_id)

    return false unless FO_EquipmentConfig::ENABLE_EQUIPMENT
    return false unless inventory_ready?
    return false unless weapon_database_ready?

    weapon_id = weapon_id.to_i

    return false if weapon_id <= 0

    weapon = $data_weapons[weapon_id]

    if weapon.nil?

      log_action("Arma ID #{weapon_id} não encontrada no Database.")

      return false

    end

    unless FO_Inventory.has_item?(:weapon, weapon_id, 1)

      log_action("Player não possui a arma #{weapon.name} [ID #{weapon_id}].")

      return false

    end

    old_weapon_id = equips[FO_EquipmentConfig::SLOT_WEAPON]

    if old_weapon_id != nil

      FO_Inventory.add_item(:weapon, old_weapon_id, 1)

      log_action(
        "Arma anterior #{weapon_name(old_weapon_id)} retornou ao inventário."
      )

    end

    FO_Inventory.remove_item(:weapon, weapon_id, 1)

    equips[FO_EquipmentConfig::SLOT_WEAPON] = weapon_id

    log_action(
      "Player equipou arma: #{weapon.name} [ID #{weapon_id}]."
    )

    refresh_stats

    return true

  end

  #===========================================================================
  # ■ equip_armor
  #===========================================================================
  def self.equip_armor(armor_id)

    return false unless FO_EquipmentConfig::ENABLE_EQUIPMENT
    return false unless inventory_ready?
    return false unless armor_database_ready?

    armor_id = armor_id.to_i

    return false if armor_id <= 0

    armor = $data_armors[armor_id]

    if armor.nil?

      log_action("Armadura ID #{armor_id} não encontrada no Database.")

      return false

    end

    unless FO_Inventory.has_item?(:armor, armor_id, 1)

      log_action("Player não possui o equipamento #{armor.name} [ID #{armor_id}].")

      return false

    end

    slot = armor_slot(armor)

    if slot.nil?

      log_action("Não foi possível identificar slot para #{armor.name}.")

      return false

    end

    old_armor_id = equips[slot]

    if old_armor_id != nil

      FO_Inventory.add_item(:armor, old_armor_id, 1)

      log_action(
        "Equipamento anterior #{armor_name(old_armor_id)} retornou ao inventário."
      )

    end

    FO_Inventory.remove_item(:armor, armor_id, 1)

    equips[slot] = armor_id

    log_action(
      "Player equipou #{armor.name} [ID #{armor_id}] no slot #{slot_label(slot)}."
    )

    refresh_stats

    return true

  end

  #===========================================================================
  # ■ unequip
  #===========================================================================
  def self.unequip(slot)

    return false unless FO_EquipmentConfig::ENABLE_EQUIPMENT
    return false unless inventory_ready?
    return false if slot.nil?

    slot = slot.to_sym

    unless valid_slot?(slot)

      log_action("Slot inválido: #{slot.to_s}.")

      return false

    end

    item_id = equips[slot]

    if item_id.nil?

      log_action("Slot #{slot_label(slot)} já está vazio.")

      return false

    end

    if slot == FO_EquipmentConfig::SLOT_WEAPON

      FO_Inventory.add_item(:weapon, item_id, 1)

      log_action(
        "Player desequipou #{weapon_name(item_id)} do slot #{slot_label(slot)}."
      )

    else

      FO_Inventory.add_item(:armor, item_id, 1)

      log_action(
        "Player desequipou #{armor_name(item_id)} do slot #{slot_label(slot)}."
      )

    end

    equips[slot] = nil

    refresh_stats

    return true

  end

  #===========================================================================
  # ■ armor_slot
  #===========================================================================
  def self.armor_slot(armor)

    return nil if armor.nil?

    # RMXP RPG::Armor kind:
    # 0 = Shield
    # 1 = Helmet
    # 2 = Body Armor
    # 3 = Accessory

    if armor.respond_to?(:kind)

      case armor.kind
      when 0
        return FO_EquipmentConfig::SLOT_SHIELD
      when 1
        return FO_EquipmentConfig::SLOT_HELMET
      when 2
        return FO_EquipmentConfig::SLOT_ARMOR
      when 3
        return FO_EquipmentConfig::SLOT_ACCESSORY
      end

    end

    return FO_EquipmentConfig::SLOT_ARMOR

  end

  #===========================================================================
  # ■ equipped?
  #===========================================================================
  def self.equipped?(slot)

    return false unless valid_slot?(slot)

    return equips[slot.to_sym] != nil

  end

  #===========================================================================
  # ■ equipped_item_id
  #===========================================================================
  def self.equipped_item_id(slot)

    return nil unless valid_slot?(slot)

    return equips[slot.to_sym]

  end

  #===========================================================================
  # ■ weapon_name
  #===========================================================================
  def self.weapon_name(weapon_id)

    return "Arma ID #{weapon_id}" unless defined?($data_weapons)

    weapon = $data_weapons[weapon_id.to_i]

    return "Arma ID #{weapon_id}" if weapon.nil?

    return weapon.name.to_s

  end

  #===========================================================================
  # ■ armor_name
  #===========================================================================
  def self.armor_name(armor_id)

    return "Armadura ID #{armor_id}" unless defined?($data_armors)

    armor = $data_armors[armor_id.to_i]

    return "Armadura ID #{armor_id}" if armor.nil?

    return armor.name.to_s

  end

  #===========================================================================
  # ■ slot_label
  #===========================================================================
  def self.slot_label(slot)

    return "Desconhecido" if slot.nil?

    case slot.to_sym
    when FO_EquipmentConfig::SLOT_WEAPON
      return "Weapon"
    when FO_EquipmentConfig::SLOT_SHIELD
      return "Shield"
    when FO_EquipmentConfig::SLOT_HELMET
      return "Helmet"
    when FO_EquipmentConfig::SLOT_ARMOR
      return "Armor"
    when FO_EquipmentConfig::SLOT_BOOTS
      return "Boots"
    when FO_EquipmentConfig::SLOT_ACCESSORY
      return "Accessory"
    end

    return slot.to_s

  end

  #===========================================================================
  # ■ equipped_name
  #===========================================================================
  def self.equipped_name(slot)

    return "vazio" unless valid_slot?(slot)

    slot = slot.to_sym

    item_id = equips[slot]

    return "vazio" if item_id.nil?

    if slot == FO_EquipmentConfig::SLOT_WEAPON

      return weapon_name(item_id)

    end

    return armor_name(item_id)

  end

  #===========================================================================
  # ■ total_atk_bonus
  #===========================================================================
  def self.total_atk_bonus

    total = 0

    weapon_id = equips[FO_EquipmentConfig::SLOT_WEAPON]

    if weapon_id != nil && weapon_database_ready?

      weapon = $data_weapons[weapon_id]

      if weapon != nil && weapon.respond_to?(:atk)

        total += weapon.atk.to_i

      end

    end

    return total

  end

  #===========================================================================
  # ■ total_pdef_bonus
  #===========================================================================
  def self.total_pdef_bonus

    total = 0

    equips.each do |slot, item_id|

      next if item_id.nil?
      next if slot == FO_EquipmentConfig::SLOT_WEAPON
      next unless armor_database_ready?

      armor = $data_armors[item_id]

      next if armor.nil?

      if armor.respond_to?(:pdef)

        total += armor.pdef.to_i

      end

    end

    return total

  end

  #===========================================================================
  # ■ total_mdef_bonus
  #===========================================================================
  def self.total_mdef_bonus

    total = 0

    equips.each do |slot, item_id|

      next if item_id.nil?
      next if slot == FO_EquipmentConfig::SLOT_WEAPON
      next unless armor_database_ready?

      armor = $data_armors[item_id]

      next if armor.nil?

      if armor.respond_to?(:mdef)

        total += armor.mdef.to_i

      end

    end

    return total

  end

  #===========================================================================
  # ■ clear
  #===========================================================================
  def self.clear

    setup

    @equips.keys.each do |slot|

      @equips[slot] = nil

    end

    log_action("Equipamentos limpos.")

    refresh_stats

    return true

  end

  #===========================================================================
  # ■ debug_print
  #===========================================================================
  def self.debug_print

    setup

    log_action("===== EQUIPAMENTOS =====")

    default_slots.each do |slot|

      item_id = equips[slot]

      if item_id.nil?

        log_action("#{slot_label(slot)}: vazio")

      else

        log_action("#{slot_label(slot)}: #{equipped_name(slot)} [ID #{item_id}]")

      end

    end

    log_action("ATK bônus: #{total_atk_bonus}")
    log_action("PDEF bônus: #{total_pdef_bonus}")
    log_action("MDEF bônus: #{total_mdef_bonus}")
    log_action("========================")

  end

  #===========================================================================
  # ■ refresh_stats
  #===========================================================================
  def self.refresh_stats

    if defined?(FO_PlayerStats) && FO_PlayerStats.respond_to?(:refresh)

      FO_PlayerStats.refresh

    end

    log_action("Stats de equipamentos atualizados.")

  end

  #===========================================================================
  # ■ inventory_ready?
  #===========================================================================
  def self.inventory_ready?

    return defined?(FO_Inventory)

  end

  #===========================================================================
  # ■ weapon_database_ready?
  #===========================================================================
  def self.weapon_database_ready?

    return defined?($data_weapons)

  end

  #===========================================================================
  # ■ armor_database_ready?
  #===========================================================================
  def self.armor_database_ready?

    return defined?($data_armors)

  end

  #===========================================================================
  # ■ log_action
  #===========================================================================
  def self.log_action(message)

    return unless FO_EquipmentConfig::ENABLE_LOG

    if defined?(FO_Logger)

      FO_Logger.action(message)

    elsif defined?(FO_Debug)

      FO_Debug.debug_log("equipment", message)

    elsif defined?(FO_CORE)

      FO_CORE.debug(message)

    else

      p message

    end

  end

end

FO_EquipmentSystem.setup

FO_EquipmentSystem.log_action(
  "FO_EquipmentSystem v1.3.1 carregado com sucesso."
)
