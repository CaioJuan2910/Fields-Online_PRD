class FO_PlayerData

  #===========================================================================
  # ■ ATTR_ACCESSOR
  #===========================================================================

  attr_accessor :strength
  attr_accessor :agility
  attr_accessor :intelligence
  attr_accessor :vitality
  attr_accessor :dexterity

  #===========================================================================
  # ■ setup_stats
  #--------------------------------------------------------------------------
  # Inicializa atributos do personagem.
  #===========================================================================
  def setup_stats

    @strength     = 10
    @agility      = 10
    @intelligence = 10
    @vitality     = 10
    @dexterity    = 10

    FO_Logger.action("#{@name} atributos inicializados.")

    FO_Debug.debug_log("engine", "Stats inicializados.")

  end

  #===========================================================================
  # ■ attack_power
  #--------------------------------------------------------------------------
  # Fórmula de ataque físico.
  #===========================================================================
  def attack_power

    return (@strength * 2) + (@level * 3)

  end

  #===========================================================================
  # ■ magic_power
  #--------------------------------------------------------------------------
  # Fórmula de ataque mágico.
  #===========================================================================
  def magic_power

    return (@intelligence * 2) + (@level * 2)

  end

  #===========================================================================
  # ■ defense
  #--------------------------------------------------------------------------
  # Fórmula de defesa.
  #===========================================================================
  def defense

    return (@vitality * 2) + (@level * 2)

  end

  #===========================================================================
  # ■ attack_speed
  #--------------------------------------------------------------------------
  # Fórmula de velocidade de ataque.
  #===========================================================================
  def attack_speed

    return 100 + (@agility * 2)

  end

  #===========================================================================
  # ■ critical_chance
  #--------------------------------------------------------------------------
  # Fórmula de crítico.
  #===========================================================================
  def critical_chance

    return (@dexterity * 0.5)

  end

  #===========================================================================
  # ■ max_hp_formula
  #--------------------------------------------------------------------------
  # Fórmula HP máximo.
  #===========================================================================
  def max_hp_formula

    return 100 + (@vitality * 10) + (@level * 5)

  end

  #===========================================================================
  # ■ max_mp_formula
  #--------------------------------------------------------------------------
  # Fórmula MP máximo.
  #===========================================================================
  def max_mp_formula

    return 50 + (@intelligence * 10) + (@level * 3)

  end

  #===========================================================================
  # ■ recalculate_stats
  #--------------------------------------------------------------------------
  # Recalcula atributos derivados.
  #===========================================================================
  def recalculate_stats

    @max_hp = max_hp_formula
    @max_mp = max_mp_formula

    @hp = FO_Utils.clamp(@hp, 0, @max_hp)
    @mp = FO_Utils.clamp(@mp, 0, @max_mp)

    FO_Debug.debug_log("engine", "Stats recalculados.")

  end

end

#==============================================================================
# ■ Expansão do initialize
#------------------------------------------------------------------------------
# Adiciona setup de atributos ao personagem.
#==============================================================================

class FO_PlayerData

  alias fo_playerstats_initialize initialize

  #===========================================================================
  # ■ initialize
  #===========================================================================
  def initialize(name = "Player")

    fo_playerstats_initialize(name)

    setup_stats

    recalculate_stats

    FO_Logger.action("#{@name} stats carregados.")

  end

end
