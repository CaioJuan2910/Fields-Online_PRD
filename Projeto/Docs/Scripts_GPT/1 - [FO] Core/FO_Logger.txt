#------------------------------------------------------------------------------
# Descrição:
# Sistema de logs do projeto Fields Online.
#------------------------------------------------------------------------------
# Funções atuais:
# - Criação automática da pasta Logs
# - Criação automática de arquivos
# - Logs separados por categoria
# - Logs com data e hora
#------------------------------------------------------------------------------
#==============================================================================

module FO_Logger

  #===========================================================================
  # ■ CONFIGURAÇÕES
  #===========================================================================

  LOG_FOLDER = "Logs"

  #===========================================================================
  # ■ create_log_folder
  #--------------------------------------------------------------------------
  # Cria a pasta de logs caso ela não exista.
  #===========================================================================
  def self.create_log_folder

    unless FileTest.exist?(LOG_FOLDER)
      Dir.mkdir(LOG_FOLDER)
    end

  end

  #===========================================================================
  # ■ current_date
  #--------------------------------------------------------------------------
  # Retorna a data atual formatada.
  #
  # Exemplo:
  # 2026_05_18
  #===========================================================================
  def self.current_date

    time = Time.now

    year  = time.year
    month = sprintf("%02d", time.month)
    day   = sprintf("%02d", time.day)

    return "#{year}_#{month}_#{day}"

  end

  #===========================================================================
  # ■ current_time
  #--------------------------------------------------------------------------
  # Retorna horário formatado.
  #
  # Exemplo:
  # 14:35:10
  #===========================================================================
  def self.current_time

    time = Time.now

    return sprintf("%02d:%02d:%02d",
      time.hour,
      time.min,
      time.sec
    )

  end

  #===========================================================================
  # ■ log
  #--------------------------------------------------------------------------
  # Método principal de gravação de logs.
  #===========================================================================
  def self.log(type, message)

    begin

      create_log_folder

      file_name = "fo_log_#{type}_#{current_date}.txt"

      file_path = "#{LOG_FOLDER}/#{file_name}"

      file = File.open(file_path, "a")

      file.write("[#{current_time}] #{message}\n")

      file.close

      FO_CORE.debug("[LOGGER] #{message}")

    rescue => error

      print "\n"
      print "========================================\n"
      print "FO_Logger ERROR\n"
      print "========================================\n"
      print "#{error}\n"
      print "========================================\n"

    end

  end

  #===========================================================================
  # ■ error
  #--------------------------------------------------------------------------
  # Log de erros.
  #===========================================================================
  def self.error(message)

    log("erros", message)

  end

  #===========================================================================
  # ■ action
  #--------------------------------------------------------------------------
  # Log de ações.
  #===========================================================================
  def self.action(message)

    log("acoes", message)

  end

  #===========================================================================
  # ■ chat
  #--------------------------------------------------------------------------
  # Log de chat.
  #===========================================================================
  def self.chat(message)

    log("chat", message)

  end

  #===========================================================================
  # ■ network
  #--------------------------------------------------------------------------
  # Log de rede/network.
  #===========================================================================
  def self.network(message)

    log("network", message)

  end

end

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_Logger.create_log_folder

FO_CORE.debug("FO_Logger carregado com sucesso.")
