module FO_TARGET
  MAX_DISTANCE = 5
  TARGET_KEY   = Input::L   # Q no teclado
end

$fo_target      = nil
$fo_target_name = "Nenhum"
$fo_target_type = nil

class FO_Target_System

  def self.update
    update_input
    update_target_validity
  end

  def self.update_input
    select_target if Input.trigger?(FO_TARGET::TARGET_KEY)
  end

  def self.select_target
    return if $game_map.nil?
    return if $game_map.events.nil?

    nearest_event    = nil
    nearest_distance = 9999

    for event in $game_map.events.values
      next if event.nil?
      event_data = event.instance_variable_get(:@event)
      next if event_data.nil?
      event_name = event_data.name
      next if event_name.nil? || event_name.empty?

      distance = ($game_player.x - event.x).abs +
                 ($game_player.y - event.y).abs
      next if distance > FO_TARGET::MAX_DISTANCE

      if distance < nearest_distance
        nearest_distance = distance
        nearest_event    = event
      end
    end

    $fo_target = nearest_event
    update_target_data
  end

  def self.update_target_data
    $fo_target_name = "Nenhum"
    $fo_target_type = nil
    return if $fo_target.nil?

    event_data = $fo_target.instance_variable_get(:@event)
    return if event_data.nil?

    original_name = event_data.name
    return if original_name.nil? || original_name.empty?

    # Protege nomes com apenas 1 caractere
    if original_name.size <= 1
      $fo_target_name = original_name
      $fo_target_type = :unknown
      return
    end

    first_char = original_name[0, 1]
    clean_name = original_name[1, original_name.size - 1]
    clean_name = original_name if clean_name.nil? || clean_name.empty?

    case first_char
    when "@" then $fo_target_type = :npc
    when "$" then $fo_target_type = :mob
    when "!" then $fo_target_type = :boss
    when "?" then $fo_target_type = :quest
    else          $fo_target_type = :unknown
    end

    $fo_target_name = clean_name
  end

  def self.update_target_validity
    return if $fo_target.nil?

    # Limpa se morto
    if $fo_target.respond_to?(:fo_dead?) && $fo_target.fo_dead?
      clear_target
      return
    end

    distance = ($game_player.x - $fo_target.x).abs +
               ($game_player.y - $fo_target.y).abs

    clear_target if distance > FO_TARGET::MAX_DISTANCE
  end

  def self.clear_target
    $fo_target      = nil
    $fo_target_name = "Nenhum"
    $fo_target_type = nil
  end

end

class Scene_Map

  alias fo_target_update_v26 update

  def update
    fo_target_update_v26
    FO_Target_System.update
  end

end
