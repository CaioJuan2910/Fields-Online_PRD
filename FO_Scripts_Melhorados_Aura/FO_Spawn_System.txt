module FO_Spawn_System
  SPAWNS = {
    1 => [
      { :event_id => 5, :amount => 3 },
      { :event_id => 2, :amount => 3 }
    ]
  }
end

class FO_Spawn_Manager

  def self.setup_map_spawns
    return if $game_map.nil?
    map_id = $game_map.map_id
    return unless FO_Spawn_System::SPAWNS.has_key?(map_id)

    FO_Spawn_System::SPAWNS[map_id].each do |spawn_data|
      create_spawn_group(spawn_data[:event_id], spawn_data[:amount])
    end
  end

  def self.create_spawn_group(template_event_id, amount)
    return if $game_map.nil?
    template = $game_map.events[template_event_id]
    return if template.nil?

    template.transparent = true

    amount.times { create_clone(template) }
  end

  def self.create_clone(template)
    return if $game_map.nil?

    new_id = ($game_map.events.keys.max || 0) + 1

    # Clona RPG::Event original (dentro do método — indentação corrigida)
    original_event = template.instance_variable_get(:@event)
    return if original_event.nil?

    event_data    = Marshal.load(Marshal.dump(original_event))
    event_data.id = new_id

    # Posição aleatória passável
    x     = rand($game_map.width)
    y     = rand($game_map.height)
    tries = 30

    while tries > 0
      break if $game_map.passable?(x, y, 2)
      x      = rand($game_map.width)
      y      = rand($game_map.height)
      tries -= 1
    end

    cloned_event = Game_Event.new($game_map.map_id, event_data)
    cloned_event.moveto(x, y)

    $game_map.events[new_id] = cloned_event
    # NOTA: Para o sprite aparecer, o Spriteset_Map precisa ser
    # refreshado após a adição. Considere chamar:
    # $scene.instance_variable_get(:@spriteset).refresh_characters
    # ou recriar o spriteset após setup_map_spawns.
  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_spawn_system_main_v11 main

  def main
    FO_Spawn_Manager.setup_map_spawns
    fo_spawn_system_main_v11
  end

end
