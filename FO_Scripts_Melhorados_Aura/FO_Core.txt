# Descrição:
# Script principal do projeto Fields Online.
# Responsável pelas configurações globais, debug e logs do jogo.
# CHANGELOG v1.3:
# - File.open agora usa bloco (garante fechamento mesmo em erro)
# - rescue em write_log evita crash por disco cheio / permissão
# - rescue em create_log_file evita crash se pasta não puder ser criada
#==============================================================================

module FO_CORE

  #===========================================================================
  # ■ CONFIGURAÇÕES GERAIS
  #===========================================================================

  GAME_NAME    = "Fields Online"
  GAME_VERSION = "0.1 Alpha"

  SCREEN_WIDTH  = 640
  SCREEN_HEIGHT = 480

  GAME_FPS = 60

  #===========================================================================
  # ■ CONFIGURAÇÕES DEBUG
  #===========================================================================

  DEBUG_MODE     = true
  ENABLE_LOGS    = true

  LOG_FILE_NAME  = "Logs/fo_debug.txt"

  # Mostrar popup de debug?
  SHOW_POPUP_DEBUG = false

  #===========================================================================
  # ■ SISTEMA DE LOGS
  #===========================================================================

  #--------------------------------------------------------------------------
  # ● Criar arquivo de log (seguro — sem crash por permissão)
  #--------------------------------------------------------------------------
  def self.create_log_file
    begin
      Dir.mkdir("Logs") unless FileTest.exist?("Logs")
      unless FileTest.exist?(LOG_FILE_NAME)
        File.open(LOG_FILE_NAME, "w") do |file|
          file.puts("==============================")
          file.puts(" Fields Online - Debug Log")
          file.puts("==============================")
        end
      end
    rescue => e
      # Silencia erro de I/O para não travar o jogo
    end
  end

  #--------------------------------------------------------------------------
  # ● Escrever log (seguro — fecha arquivo mesmo em erro)
  #--------------------------------------------------------------------------
  def self.write_log(message)
    return unless ENABLE_LOGS
    begin
      create_log_file
      File.open(LOG_FILE_NAME, "a") do |file|
        time = Time.now.strftime("%d/%m/%Y %H:%M:%S")
        file.puts("[#{time}] #{message}")
      end
    rescue => e
      # Silencia erros de I/O
    end
  end

  #===========================================================================
  # ■ DEBUG
  #===========================================================================

  #--------------------------------------------------------------------------
  # ● Debug
  #--------------------------------------------------------------------------
  def self.debug(message)
    return unless DEBUG_MODE
    print("[FO_DEBUG] #{message}") if SHOW_POPUP_DEBUG
    write_log(message)
  end

end

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_CORE.debug("===================================")
FO_CORE.debug("Fields Online iniciado")
FO_CORE.debug("Versão: #{FO_CORE::GAME_VERSION}")
FO_CORE.debug("FO_Core carregado com sucesso")
FO_CORE.debug("===================================")
