class FO_Player_HUD

  def initialize
    create_hud
  end

  def create_hud
    @sprite        = Sprite.new
    @sprite.bitmap = Bitmap.new(320, 120)
    @sprite.x      = 10
    @sprite.y      = 10
    @sprite.z      = 9999
    refresh
  end

  def refresh
    return if @sprite.nil? || @sprite.disposed?
    bitmap = @sprite.bitmap
    return if bitmap.nil?

    bitmap.clear

    # Fundo
    bitmap.fill_rect(0, 0, 320, 120, Color.new(0, 0, 0, 160))

    # Fonte
    bitmap.font.name  = "Arial"
    bitmap.font.size  = 18
    bitmap.font.bold  = true
    bitmap.font.color = Color.new(255, 255, 255)

    # Nome do jogo
    bitmap.draw_text(10, 5, 300, 24, "Fields Online")

    # Level
    bitmap.draw_text(10, 30, 300, 24, "Level: #{$fo_player_level}")

    # HP
    maxhp    = [$fo_player_maxhp.to_i, 1].max
    hp_rate  = $fo_player_hp.to_f / maxhp
    hp_width = (200 * hp_rate).to_i.abs
    bitmap.fill_rect(10, 58, 200, 16, Color.new(60, 60, 60))
    bitmap.fill_rect(10, 58, hp_width, 16, Color.new(220, 40, 40))
    bitmap.draw_text(220, 54, 90, 24, "#{$fo_player_hp}/#{$fo_player_maxhp}")

    # MP
    maxmp    = [$fo_player_maxmp.to_i, 1].max
    mp_rate  = $fo_player_mp.to_f / maxmp
    mp_width = (200 * mp_rate).to_i.abs
    bitmap.fill_rect(10, 82, 200, 16, Color.new(60, 60, 60))
    bitmap.fill_rect(10, 82, mp_width, 16, Color.new(40, 120, 255))
    bitmap.draw_text(220, 78, 90, 24, "#{$fo_player_mp}/#{$fo_player_maxmp}")

    # EXP
    next_exp  = [$fo_player_next_exp.to_i, 1].max
    exp_rate  = $fo_player_exp.to_f / next_exp
    exp_width = (300 * exp_rate).to_i.abs
    bitmap.fill_rect(10, 105, 300, 8, Color.new(60, 60, 60))
    bitmap.fill_rect(10, 105, exp_width, 8, Color.new(255, 220, 40))
  end

  def update
    refresh
  end

  def dispose
    return if @sprite.nil?
    @sprite.bitmap.dispose unless @sprite.bitmap.nil?
    @sprite.dispose
  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_player_hud_main_v31   main
  alias fo_player_hud_update_v31 update

  def main
    @fo_player_hud = FO_Player_HUD.new
    fo_player_hud_main_v31
    @fo_player_hud.dispose if @fo_player_hud
  end

  def update
    fo_player_hud_update_v31
    @fo_player_hud.update if @fo_player_hud
  end

end
