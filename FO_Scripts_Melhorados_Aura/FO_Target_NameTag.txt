class FO_Target_NameTag

  def initialize
    create_sprite
  end

  def create_sprite
    @sprite        = Sprite.new
    @sprite.bitmap = Bitmap.new(220, 64)
    @sprite.z      = 5000

    font       = @sprite.bitmap.font
    font.name  = "Arial"
    font.size  = 18
    font.bold  = true
  end

  def update
    refresh
  end

  def refresh
    return if @sprite.nil? || @sprite.disposed?
    bitmap = @sprite.bitmap
    return if bitmap.nil?

    bitmap.clear

    if $fo_target.nil?
      @sprite.visible = false
      return
    end

    @sprite.visible = true

    # Cor por tipo
    color = case $fo_target_type
            when :npc   then Color.new(255, 255, 255)
            when :mob   then Color.new(100, 255, 100)
            when :boss  then Color.new(255, 80, 80)
            when :quest then Color.new(255, 220, 80)
            else             Color.new(255, 255, 255)
            end
    bitmap.font.color = color

    # Nome
    bitmap.draw_text(0, 0, 220, 32, $fo_target_name.to_s, 1)

    # HP apenas para mobs (guard nil — NPCs não têm fo_hp)
    if $fo_target_type == :mob &&
       $fo_target.respond_to?(:fo_hp) &&
       $fo_target.respond_to?(:fo_maxhp)

      hp    = $fo_target.fo_hp.to_i
      maxhp = [$fo_target.fo_maxhp.to_i, 1].max
      rate  = hp.to_f / maxhp

      bar_width    = 120
      filled_width = (bar_width * rate).to_i

      bitmap.fill_rect(50, 34, bar_width, 8, Color.new(60, 60, 60))
      bitmap.fill_rect(50, 34, filled_width, 8, Color.new(0, 220, 0))
    end

    # Posição dinâmica
    screen_x     = $fo_target.screen_x
    screen_y     = $fo_target.screen_y
    frame_height = 48

    event_data = $fo_target.instance_variable_get(:@event)
    if event_data
      page = event_data.pages[0]
      if page
        graphic      = page.graphic
        charset_name = graphic.character_name
        charset_hue  = graphic.character_hue
        if charset_name && !charset_name.empty?
          begin
            charset      = RPG::Cache.character(charset_name, charset_hue)
            frame_height = charset.height / 4
          rescue
          end
        end
      end
    end

    @sprite.x = screen_x - 110
    @sprite.y = screen_y - frame_height - 42
  end

  def dispose
    return if @sprite.nil?
    @sprite.bitmap.dispose unless @sprite.bitmap.nil?
    @sprite.dispose
  end

end

#==============================================================================

class Scene_Map

  alias fo_target_nametag_main_v21   main
  alias fo_target_nametag_update_v21 update

  def main
    @fo_target_nametag = FO_Target_NameTag.new
    fo_target_nametag_main_v21
    @fo_target_nametag.dispose if @fo_target_nametag
  end

  def update
    fo_target_nametag_update_v21
    @fo_target_nametag.update if @fo_target_nametag
  end

end
