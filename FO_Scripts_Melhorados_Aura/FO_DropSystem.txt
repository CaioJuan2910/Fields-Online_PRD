module FO_DROP

  #--------------------------------------------------------------------------
  # Array ordenado de raridades — garante ordem correta no Ruby 1.8
  #--------------------------------------------------------------------------
  RARITY_CHANCE = [
    [:comum,      70],
    [:raro,       20],
    [:gold,        9],
    [:super_raro,  1]
  ]

  DROP_TABLE = {
    1 => {
      5 => {
        :comum      => [[:item,   1]],
        :raro       => [[:weapon, 1]],
        :gold       => [[:armor,  1]],
        :super_raro => [[:weapon, 2]]
      }
    }
  }

end

#==============================================================================
# ■ FO_DropManager
#==============================================================================

class FO_DropManager

  def self.generate_drop(map_id, event_id)
    map_data  = FO_DROP::DROP_TABLE[map_id]
    return nil if map_data.nil?

    drop_data = map_data[event_id]
    return nil if drop_data.nil?

    rarity    = roll_rarity
    item_list = drop_data[rarity]
    return nil if item_list.nil? || item_list.empty?

    loot = item_list[rand(item_list.size)]
    [loot, rarity]
  end

  #--------------------------------------------------------------------------
  # ● Sorteia raridade — Array garante ordem em Ruby 1.8
  #--------------------------------------------------------------------------
  def self.roll_rarity
    roll    = rand(100) + 1
    current = 0
    FO_DROP::RARITY_CHANCE.each do |rarity, chance|
      current += chance
      return rarity if roll <= current
    end
    :comum
  end

end
