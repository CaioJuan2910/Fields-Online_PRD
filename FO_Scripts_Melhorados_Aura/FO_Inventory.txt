class FO_Inventory

  MAX_ITEMS = 20

  def initialize
    @items = []
    FO_CORE.debug("FO_Inventory iniciado")
    show_inventory
  end

  def add_item(item_id)
    if @items.size >= MAX_ITEMS
      FO_CORE.debug("Inventário cheio")
      return false
    end
    item = FO_ItemSystem.get_item(item_id)
    if item.nil?
      FO_CORE.debug("Item inválido: #{item_id}")
      return false
    end
    @items.push(item)
    FO_CORE.debug("Item adicionado: #{item.name}")
    show_inventory
    true
  end

  def remove_item(item_id)
    item = @items.find { |i| i.id == item_id }
    if item.nil?
      FO_CORE.debug("Item não encontrado: #{item_id}")
      return false
    end
    @items.delete(item)
    FO_CORE.debug("Item removido: #{item.name}")
    show_inventory
    true
  end

  def has_item?(item_id)
    @items.any? { |i| i.id == item_id }
  end

  def count
    @items.size
  end

  def show_inventory
    FO_CORE.debug("==============================")
    FO_CORE.debug("INVENTÁRIO")
    if @items.empty?
      FO_CORE.debug("Inventário vazio")
    else
      @items.each do |item|
        FO_CORE.debug("--------------------")
        FO_CORE.debug("ID......: #{item.id}")
        FO_CORE.debug("Nome....: #{item.name}")
        FO_CORE.debug("Tipo....: #{item.type}")
        FO_CORE.debug("Raridade: #{item.rarity}")
        FO_CORE.debug("Ataque..: #{item.attack}")
        FO_CORE.debug("Defesa..: #{item.defense}")
        FO_CORE.debug("Preço...: #{item.price}")
      end
    end
    FO_CORE.debug("Slots: #{@items.size}/#{MAX_ITEMS}")
    FO_CORE.debug("==============================")
  end

end
