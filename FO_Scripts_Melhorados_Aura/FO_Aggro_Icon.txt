class Sprite_Character < RPG::Sprite

  alias fo_aggro_icon_initialize_v21 initialize
  alias fo_aggro_icon_update_v21     update
  alias fo_aggro_icon_dispose_v21    dispose

  def initialize(viewport, character = nil)
    fo_aggro_icon_initialize_v21(viewport, character)
    @fo_aggro_icon = nil
    create_fo_aggro_icon
  end

  def create_fo_aggro_icon
    return if self.viewport.nil?
    @fo_aggro_icon         = Sprite.new(self.viewport)
    @fo_aggro_icon.bitmap  = RPG::Cache.picture("IconFuria")
    @fo_aggro_icon.visible = false
    @fo_aggro_icon.z       = self.z + 50
  end

  def update
    fo_aggro_icon_update_v21
    update_fo_aggro_icon
  end

  def update_fo_aggro_icon
    return if @fo_aggro_icon.nil?

    if @character.nil?
      @fo_aggro_icon.visible = false
      return
    end

    unless @character.is_a?(Game_Event)
      @fo_aggro_icon.visible = false
      return
    end

    if @character.respond_to?(:fo_dead?) && @character.fo_dead?
      @fo_aggro_icon.visible = false
      return
    end

    if @character.transparent
      @fo_aggro_icon.visible = false
      return
    end

    if self.opacity <= 0
      @fo_aggro_icon.visible = false
      return
    end

    # Verifica @fo_aggressive — dupla compatibilidade Ruby 1.8 (String) / 1.9+ (Symbol)
    aggro = false
    ivars = @character.instance_variables
    if ivars.include?(:@fo_aggressive) || ivars.include?("@fo_aggressive")
      aggro = @character.instance_variable_get(:@fo_aggressive)
    elsif @character.respond_to?(:fo_aggressive)
      aggro = @character.fo_aggressive
    end

    unless aggro
      @fo_aggro_icon.visible = false
      return
    end

    @fo_aggro_icon.visible = true

    return if @fo_aggro_icon.bitmap.nil?

    @fo_aggro_icon.x = self.x - (@fo_aggro_icon.bitmap.width / 2)
    @fo_aggro_icon.y = self.y - self.oy - 42
    @fo_aggro_icon.z = self.z + 50
  end

  def dispose
    fo_aggro_icon_dispose_v21
    if @fo_aggro_icon
      @fo_aggro_icon.bitmap.dispose unless @fo_aggro_icon.bitmap.nil?
      @fo_aggro_icon.dispose
      @fo_aggro_icon = nil
    end
  end

end
