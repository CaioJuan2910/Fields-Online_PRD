#==============================================================================
# ■ FO_Engine
#------------------------------------------------------------------------------
# Classe principal da Engine
#==============================================================================

class FO_Engine

  #===========================================================================
  # ● Initialize
  #===========================================================================
  def initialize
    @frame_count = 0
    @start_time  = Time.now
    FO_CORE.debug("FO_Engine iniciada")
  end

  #===========================================================================
  # ● Tempo decorrido desde o início (em segundos)
  #===========================================================================
  def elapsed_time
    Time.now - @start_time
  end

  #===========================================================================
  # ● Update (chamado a cada frame)
  #===========================================================================
  def update
    @frame_count += 1
    update_second if @frame_count >= Graphics.frame_rate
  end

  #===========================================================================
  # ● Update Second (chamado 1x por segundo)
  #===========================================================================
  def update_second
    @frame_count = 0
    # Debug apenas ocasional — remova a linha abaixo em produção
    # FO_CORE.debug("FO_Engine update global — #{elapsed_time.to_i}s")
  end

end

#==============================================================================
# ■ Scene_Map
#------------------------------------------------------------------------------
# Integra Engine ao mapa principal
#==============================================================================

class Scene_Map

  alias fo_engine_main   main
  alias fo_engine_update update

  def main
    @fo_engine = FO_Engine.new
    FO_CORE.debug("FO_Engine conectada ao Scene_Map")
    fo_engine_main
  end

  def update
    fo_engine_update
    @fo_engine.update if @fo_engine
  end

end
