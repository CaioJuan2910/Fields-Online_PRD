module FO_Target_Status

  MOB_HP   = 100
  BOSS_HP  = 1000

  MOB_EXP  = 25
  BOSS_EXP = 250

  MOB_DAMAGE  = 8
  BOSS_DAMAGE = 25

  MOB_ATTACK_SPEED  = 90
  BOSS_ATTACK_SPEED = 45

  MOB_RESPAWN  = 5  * Graphics.frame_rate
  BOSS_RESPAWN = 60 * Graphics.frame_rate

  RANDOM_RESPAWN_RANGE = 6

  MAX_WANDER_DISTANCE = 5
  MOB_MOVE_CHANCE     = 25
  BOSS_MOVE_CHANCE    = 12

  AGGRO_RANGE    = 5
  MAX_FAILED_MOVES = 20
  COMBAT_TIME    = 5 * Graphics.frame_rate

end

#==============================================================================
# ■ Player Status (variáveis globais)
#==============================================================================

$fo_player_level    = 1
$fo_player_exp      = 0
$fo_player_next_exp = 100
$fo_player_maxhp    = 810
$fo_player_hp       = 810
$fo_player_maxmp    = 585
$fo_player_mp       = 585

#==============================================================================
# ■ FO_Player_Level
#==============================================================================

module FO_Player_Level

  def self.gain_exp(amount)
    $fo_player_exp += amount
    check_level_up
  end

  def self.check_level_up
    while $fo_player_exp >= $fo_player_next_exp
      level_up
    end
  end

  def self.level_up
    $fo_player_exp      -= $fo_player_next_exp
    $fo_player_level    += 1
    $fo_player_next_exp  = ($fo_player_next_exp * 1.5).to_i
    $fo_player_maxhp    += 50
    $fo_player_hp        = $fo_player_maxhp
    $fo_player_maxmp    += 25
    $fo_player_mp        = $fo_player_maxmp
  end

end

#==============================================================================
# ■ Game_Event
#==============================================================================

class Game_Event < Game_Character

  attr_accessor :fo_hp
  attr_accessor :fo_maxhp
  attr_accessor :fo_entity_type
  attr_accessor :fo_dead
  attr_accessor :fo_respawn_time
  attr_accessor :fo_in_combat
  attr_accessor :fo_aggressive     # ← NOVO: usado pelo FO_Aggro_Icon

  alias fo_target_status_initialize initialize
  alias fo_target_status_update     update

  def initialize(map_id, event)
    fo_target_status_initialize(map_id, event)
    @fo_spawn_x = @x
    @fo_spawn_y = @y
    setup_fo_entity
  end

  def update
    fo_target_status_update
    update_fo_respawn
    update_fo_combat
    update_fo_ai
    update_fo_attack
  end

  #--------------------------------------------------------------------------
  # ● Setup entidade
  #--------------------------------------------------------------------------
  def setup_fo_entity
    @fo_dead        = false
    @fo_respawn_time = 0
    @fo_in_combat   = false
    @fo_aggressive  = false
    @fo_failed_moves = 0
    @fo_attack_cooldown = 0
    @fo_combat_time = 0
    @fo_entity_type = nil

    # Guard nil para @event
    return if @event.nil?
    event_name = @event.name
    return if event_name.nil?

    if event_name.include?("$")
      @fo_entity_type = :mob
      @fo_maxhp = FO_Target_Status::MOB_HP
      @fo_hp    = @fo_maxhp
    end

    if event_name.include?("!")
      @fo_entity_type = :boss
      @fo_maxhp = FO_Target_Status::BOSS_HP
      @fo_hp    = @fo_maxhp
    end
  end

  def fo_combat_entity?
    @fo_entity_type == :mob || @fo_entity_type == :boss
  end

  def fo_dead?
    @fo_dead == true
  end

  def fo_gain_exp
    return unless fo_combat_entity?
    amount = @fo_entity_type == :boss ?
             FO_Target_Status::BOSS_EXP :
             FO_Target_Status::MOB_EXP
    FO_Player_Level.gain_exp(amount)
  end

  def fo_kill
    return if fo_dead?
    @fo_dead        = true
    @fo_hp          = 0
    @fo_in_combat   = false
    @fo_aggressive  = false
    @fo_failed_moves = 0
    @fo_combat_time = 0

    fo_gain_exp

    @fo_respawn_time = @fo_entity_type == :boss ?
      FO_Target_Status::BOSS_RESPAWN :
      FO_Target_Status::MOB_RESPAWN

    if $fo_target == self
      $fo_target      = nil
      $fo_target_name = "Nenhum"    # ← era "" antes
      $fo_target_type = nil
      $fo_boss_target = nil
    end

    @transparent = true
    @through     = true
  end

  def update_fo_respawn
    return unless fo_dead?
    return if @fo_respawn_time <= 0
    @fo_respawn_time -= 1
    fo_respawn if @fo_respawn_time <= 0
  end

  def fo_respawn
    @fo_dead        = false
    @fo_hp          = @fo_maxhp
    @fo_in_combat   = false
    @fo_aggressive  = false
    @fo_failed_moves = 0
    @fo_combat_time = 0
    random_respawn_position
    @transparent = false
    @through     = false
  end

  def update_fo_combat
    return unless @fo_in_combat
    if $fo_target == self
      @fo_combat_time = FO_Target_Status::COMBAT_TIME
      return
    end
    @fo_combat_time -= 1 if @fo_combat_time > 0
    if @fo_combat_time <= 0
      @fo_in_combat  = false
      @fo_aggressive = false
      @fo_failed_moves = 0
    end
  end

  def update_fo_attack
    return if fo_dead?
    return unless fo_combat_entity?
    return unless @fo_in_combat
    if @fo_attack_cooldown > 0
      @fo_attack_cooldown -= 1
      return
    end
    return unless (@x - $game_player.x).abs <= 1
    return unless (@y - $game_player.y).abs <= 1
    execute_fo_attack
  end

  def execute_fo_attack
    damage      = 0
    damage_type = :normal
    text        = ""
    roll        = rand(100)
    base_damage = @fo_entity_type == :boss ?
                  FO_Target_Status::BOSS_DAMAGE :
                  FO_Target_Status::MOB_DAMAGE

    if    roll < 10 then text = "MISS"; damage_type = :miss
    elsif roll < 25 then damage = (base_damage * 0.10).to_i
    elsif roll < 60 then damage = (base_damage * 0.50).to_i
    elsif roll < 85 then damage = (base_damage * 0.80).to_i
    elsif roll < 95 then damage = base_damage
    else
      critical_rate = rand(81) + 120
      damage        = (base_damage * critical_rate / 100.0).to_i
      damage_type   = :critical
      text          = "CRITICAL"
    end

    text = damage.to_s if damage > 0 && text.empty?

    if damage > 0
      $fo_player_hp -= damage
      $fo_player_hp  = 0 if $fo_player_hp < 0
    end

    FO_Floating_Damage.create_player(text, damage_type)

    @fo_attack_cooldown = @fo_entity_type == :boss ?
      FO_Target_Status::BOSS_ATTACK_SPEED :
      FO_Target_Status::MOB_ATTACK_SPEED
  end

  def update_fo_ai
    return if fo_dead?
    return unless fo_combat_entity?
    return if moving?

    if @fo_in_combat
      @fo_aggressive = true
      follow_player_diagonal
      return
    end

    # Detecção de aggro por proximidade
    distance = ($game_player.x - @x).abs + ($game_player.y - @y).abs
    if distance <= FO_Target_Status::AGGRO_RANGE
      @fo_in_combat  = true
      @fo_aggressive = true
      return
    end

    @fo_aggressive = false

    chance = @fo_entity_type == :boss ?
             FO_Target_Status::BOSS_MOVE_CHANCE :
             FO_Target_Status::MOB_MOVE_CHANCE
    return unless rand(100) < chance
    move_random
  end

  def follow_player_diagonal
    distance_x = (@x - $game_player.x).abs
    distance_y = (@y - $game_player.y).abs
    return if distance_x <= 1 && distance_y <= 1

    old_x = @x
    old_y = @y
    sx    = @x - $game_player.x
    sy    = @y - $game_player.y

    if sx != 0 && sy != 0
      horizontal = sx > 0 ? 4 : 6
      vertical   = sy > 0 ? 8 : 2
      move_diagonal(horizontal, vertical)
    else
      move_toward_player
    end

    if @x == old_x && @y == old_y
      @fo_failed_moves += 1
    else
      @fo_failed_moves = 0
    end

    if @fo_failed_moves >= FO_Target_Status::MAX_FAILED_MOVES
      @fo_in_combat    = false
      @fo_aggressive   = false
      @fo_failed_moves = 0
    end
  end

  def move_diagonal(horizontal, vertical)
    case horizontal
    when 4 then move_left
    when 6 then move_right
    end
    case vertical
    when 8 then move_up
    when 2 then move_down
    end
  end

  def random_respawn_position
    max_range = FO_Target_Status::RANDOM_RESPAWN_RANGE
    tries     = 20
    while tries > 0
      new_x = @fo_spawn_x + rand(max_range * 2 + 1) - max_range
      new_y = @fo_spawn_y + rand(max_range * 2 + 1) - max_range
      tries -= 1
      next if new_x < 0 || new_y < 0
      next if new_x >= $game_map.width || new_y >= $game_map.height
      if $game_map.passable?(new_x, new_y, 2)
        moveto(new_x, new_y)
        return
      end
    end
    moveto(@fo_spawn_x, @fo_spawn_y)
  end

end
