class FO_Floating_Damage

  @@damages = []

  def self.create(target, text, type = :normal)
    return if target.nil?
    return if text.nil? || text.to_s.empty?

    sprite                    = Sprite.new
    sprite.bitmap             = Bitmap.new(120, 32)
    sprite.bitmap.font.name   = "Arial"
    sprite.bitmap.font.size   = 22
    sprite.bitmap.font.bold   = true
    sprite.bitmap.font.color  = font_color(type)

    sprite.bitmap.draw_text(0, 0, 120, 32, text.to_s, 1)

    sprite.x = target.screen_x - 60
    sprite.y = target.screen_y - 64
    sprite.z = 9999

    @@damages.push({ :sprite => sprite, :time => 40 })
  end

  def self.create_player(text, type = :normal)
    return if text.nil? || text.to_s.empty?

    sprite                    = Sprite.new
    sprite.bitmap             = Bitmap.new(120, 32)
    sprite.bitmap.font.name   = "Arial"
    sprite.bitmap.font.size   = 22
    sprite.bitmap.font.bold   = true
    sprite.bitmap.font.color  = font_color(type)

    sprite.bitmap.draw_text(0, 0, 120, 32, text.to_s, 1)

    sprite.x = $game_player.screen_x - 60
    sprite.y = $game_player.screen_y - 64
    sprite.z = 9999

    @@damages.push({ :sprite => sprite, :time => 40 })
  end

  def self.font_color(type)
    case type
    when :miss     then Color.new(80, 180, 255)
    when :critical then Color.new(255, 60, 60)
    else                Color.new(255, 255, 255)
    end
  end

  def self.update
    @@damages.each do |damage|
      sprite = damage[:sprite]
      next if sprite.nil? || sprite.disposed?
      damage[:time] -= 1
      sprite.y      -= 1
      sprite.opacity -= 6
    end

    @@damages.delete_if do |damage|
      sprite = damage[:sprite]
      if sprite.nil? || sprite.disposed? || damage[:time] <= 0
        sprite.dispose if sprite && !sprite.disposed?
        true
      else
        false
      end
    end
  end

end

#==============================================================================

class Scene_Map

  alias fo_floating_damage_update_v21 update

  def update
    fo_floating_damage_update_v21
    FO_Floating_Damage.update
  end

end
