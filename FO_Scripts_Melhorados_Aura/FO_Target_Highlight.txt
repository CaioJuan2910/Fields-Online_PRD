class FO_Target_Highlight

  def initialize
    @viewport_assigned = false
    create_sprite
  end

  def create_sprite
    @sprite    = Sprite.new
    @sprite.bitmap = RPG::Cache.picture("target_circle")

    @sprite.ox = @sprite.bitmap.width  / 2
    @sprite.oy = @sprite.bitmap.height / 2

    @fade_min_opacity  = 220
    @fade_max_opacity  = 255
    @fade_speed        = 2
    @rotation_speed    = 0.2
    @sprite.opacity    = @fade_max_opacity
    @opacity_direction = -@fade_speed
  end

  def setup_viewport
    return if @viewport_assigned
    return unless $scene.is_a?(Scene_Map)

    spriteset = $scene.instance_variable_get(:@spriteset)
    return if spriteset.nil?

    viewport1 = spriteset.instance_variable_get(:@viewport1)
    return if viewport1.nil?
    return if @sprite.nil? || @sprite.disposed?

    x       = @sprite.x
    y       = @sprite.y
    opacity = @sprite.opacity
    angle   = @sprite.angle
    bitmap  = @sprite.bitmap

    @sprite.dispose

    @sprite    = Sprite.new(viewport1)
    @sprite.bitmap = bitmap
    @sprite.ox = @sprite.bitmap.width  / 2
    @sprite.oy = @sprite.bitmap.height / 2
    @sprite.x       = x
    @sprite.y       = y
    @sprite.opacity = opacity
    @sprite.angle   = angle

    @viewport_assigned = true
  end

  def update
    setup_viewport
    refresh
    update_animation
  end

  def refresh
    return if @sprite.nil? || @sprite.disposed?

    if $fo_target.nil?
      @sprite.visible = false
      return
    end

    unless $fo_target_type == :mob || $fo_target_type == :boss
      @sprite.visible = false
      return
    end

    @sprite.visible = true

    @sprite.x = $fo_target.screen_x + 2
    @sprite.y = $fo_target.screen_y - 8
    @sprite.z = $fo_target.screen_z - 1
  end

  def update_animation
    return if @sprite.nil? || @sprite.disposed?
    return unless @sprite.visible

    @sprite.opacity += @opacity_direction

    if    @sprite.opacity <= @fade_min_opacity then @opacity_direction =  @fade_speed
    elsif @sprite.opacity >= @fade_max_opacity then @opacity_direction = -@fade_speed
    end

    @sprite.angle += @rotation_speed
    @sprite.angle  = 0 if @sprite.angle >= 360
  end

  def dispose
    return if @sprite.nil?
    @sprite.dispose unless @sprite.disposed?
  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_target_highlight_main_v21   main
  alias fo_target_highlight_update_v21 update

  def main
    @fo_target_highlight = FO_Target_Highlight.new
    fo_target_highlight_main_v21
    @fo_target_highlight.dispose if @fo_target_highlight
  end

  def update
    fo_target_highlight_update_v21
    @fo_target_highlight.update if @fo_target_highlight
  end

end
