#==============================================================================
# ■ Game_Event
#==============================================================================

class Game_Event < Game_Character

  #--------------------------------------------------------------------------
  # Alias Initialize
  #--------------------------------------------------------------------------

  alias fo_enemy_system_initialize_v2 initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize(map_id, event)

    #--------------------------------------------------------------------------
    # Original
    #--------------------------------------------------------------------------

    fo_enemy_system_initialize_v2(
      map_id,
      event
    )

    #--------------------------------------------------------------------------
    # Enemy Setup
    #--------------------------------------------------------------------------

    setup_fo_enemy

  end

  #--------------------------------------------------------------------------
  # ● Setup Enemy
  #--------------------------------------------------------------------------
  def setup_fo_enemy

    clear_fo_enemy_data

    #--------------------------------------------------------------------------
    # Enemy ID
    #--------------------------------------------------------------------------

    enemy_id =
      find_fo_enemy_id

    return if enemy_id.nil?

    #--------------------------------------------------------------------------
    # Enemy Data
    #--------------------------------------------------------------------------

    enemy_data =
      FO_Data_Manager.enemy(
        enemy_id
      )

    return if enemy_data.nil?

    #--------------------------------------------------------------------------
    # Enemy Database
    #--------------------------------------------------------------------------

    @fo_enemy_id =
      enemy_id

    @fo_enemy_data =
      enemy_data

    #--------------------------------------------------------------------------
    # Estado
    #--------------------------------------------------------------------------

    setup_fo_enemy_state

    #--------------------------------------------------------------------------
    # Stats
    #--------------------------------------------------------------------------

    setup_fo_enemy_stats

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[ENEMY LOAD] #{@event.name} => #{@fo_enemy_data.name}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Clear Enemy Data
  #--------------------------------------------------------------------------
  def clear_fo_enemy_data

    @fo_enemy_id =
      nil

    @fo_enemy_data =
      nil

    @fo_dead =
      false

  end

  #--------------------------------------------------------------------------
  # ● Setup State
  #--------------------------------------------------------------------------
  def setup_fo_enemy_state

    @fo_dead =
      false

    @fo_in_combat =
      false

  end

  #--------------------------------------------------------------------------
  # ● Setup Stats
  #--------------------------------------------------------------------------
  def setup_fo_enemy_stats

    return if @fo_enemy_data.nil?

    #--------------------------------------------------------------------------
    # HP/SP
    #--------------------------------------------------------------------------

    @fo_maxhp =
      @fo_enemy_data.maxhp

    @fo_hp =
      @fo_maxhp

    @fo_maxsp =
      @fo_enemy_data.maxsp

    @fo_sp =
      @fo_maxsp

    #--------------------------------------------------------------------------
    # Attributes
    #--------------------------------------------------------------------------

    @fo_str =
      @fo_enemy_data.str

    @fo_dex =
      @fo_enemy_data.dex

    @fo_agi =
      @fo_enemy_data.agi

    @fo_int =
      @fo_enemy_data.int

    #--------------------------------------------------------------------------
    # Combat
    #--------------------------------------------------------------------------

    @fo_atk =
      @fo_enemy_data.atk

    @fo_pdef =
      @fo_enemy_data.pdef

    @fo_mdef =
      @fo_enemy_data.mdef

    @fo_eva =
      @fo_enemy_data.eva

    #--------------------------------------------------------------------------
    # Rewards
    #--------------------------------------------------------------------------

    @fo_exp =
      @fo_enemy_data.exp

    @fo_gold =
      @fo_enemy_data.gold

  end

  #--------------------------------------------------------------------------
  # ● Find Enemy ID
  #--------------------------------------------------------------------------
  def find_fo_enemy_id

    return nil if @list.nil?

    for command in @list

      next if command.nil?

      #----------------------------------------------------------------------
      # Comentário
      #----------------------------------------------------------------------

      if command.code == 108

        text =
          command.parameters[0]

        #--------------------------------------------------------------------
        # Tag
        #--------------------------------------------------------------------

        if text =~ /<enemy_id:(\d+)>/

          return $1.to_i

        end

      end

    end

    return nil

  end

  #--------------------------------------------------------------------------
  # ● Enemy?
  #--------------------------------------------------------------------------
  def fo_enemy?

    return !@fo_enemy_id.nil?

  end

  #--------------------------------------------------------------------------
  # ● Combat Entity?
  #--------------------------------------------------------------------------
  def fo_combat_entity?

    return fo_enemy?

  end

  #--------------------------------------------------------------------------
  # ● Kill
  #--------------------------------------------------------------------------
  def fo_kill

    @fo_dead =
      true

  end

  #--------------------------------------------------------------------------
  # ● Dead?
  #--------------------------------------------------------------------------
  def fo_dead?

    return @fo_dead

  end

  #--------------------------------------------------------------------------
  # ● Enemy ID
  #--------------------------------------------------------------------------
  def fo_enemy_id

    return @fo_enemy_id

  end

  #--------------------------------------------------------------------------
  # ● Enemy Data
  #--------------------------------------------------------------------------
  def fo_enemy_data

    return @fo_enemy_data

  end

  #--------------------------------------------------------------------------
  # ● HP
  #--------------------------------------------------------------------------
  attr_accessor :fo_hp

  #--------------------------------------------------------------------------
  # ● MaxHP
  #--------------------------------------------------------------------------
  attr_accessor :fo_maxhp

  #--------------------------------------------------------------------------
  # ● SP
  #--------------------------------------------------------------------------
  attr_accessor :fo_sp

  #--------------------------------------------------------------------------
  # ● MaxSP
  #--------------------------------------------------------------------------
  attr_accessor :fo_maxsp

  #--------------------------------------------------------------------------
  # ● STR
  #--------------------------------------------------------------------------
  attr_accessor :fo_str

  #--------------------------------------------------------------------------
  # ● DEX
  #--------------------------------------------------------------------------
  attr_accessor :fo_dex

  #--------------------------------------------------------------------------
  # ● AGI
  #--------------------------------------------------------------------------
  attr_accessor :fo_agi

  #--------------------------------------------------------------------------
  # ● INT
  #--------------------------------------------------------------------------
  attr_accessor :fo_int

  #--------------------------------------------------------------------------
  # ● ATK
  #--------------------------------------------------------------------------
  attr_accessor :fo_atk

  #--------------------------------------------------------------------------
  # ● PDEF
  #--------------------------------------------------------------------------
  attr_accessor :fo_pdef

  #--------------------------------------------------------------------------
  # ● MDEF
  #--------------------------------------------------------------------------
  attr_accessor :fo_mdef

  #--------------------------------------------------------------------------
  # ● EVA
  #--------------------------------------------------------------------------
  attr_accessor :fo_eva

  #--------------------------------------------------------------------------
  # ● EXP
  #--------------------------------------------------------------------------
  attr_accessor :fo_exp

  #--------------------------------------------------------------------------
  # ● GOLD
  #--------------------------------------------------------------------------
  attr_accessor :fo_gold

  #--------------------------------------------------------------------------
  # ● Combat State
  #--------------------------------------------------------------------------
  attr_accessor :fo_in_combat

end
