#==============================================================================
# ■ FO_Player_Status
#------------------------------------------------------------------------------
# Classe principal de status do jogador
#==============================================================================

class FO_Player_Status

  #===========================================================================
  # ■ ATTR ACCESSOR
  #===========================================================================

  attr_accessor :current_hp
  attr_accessor :current_mp
  attr_accessor :dead

  #===========================================================================
  # ● Initialize
  #===========================================================================
  def initialize(player_stats)

    # Referência dos stats
    @player_stats = player_stats

    # HP atual
    @current_hp = @player_stats.max_hp

    # MP atual
    @current_mp = @player_stats.max_mp

    # Morto?
    @dead = false

    # Debug
    FO_CORE.debug("FO_Player_Status iniciado")

    # Mostrar status
    show_status

  end

  #===========================================================================
  # ● Mostrar status
  #===========================================================================
  def show_status

    FO_CORE.debug("==============================")
    FO_CORE.debug("PLAYER STATUS")
    FO_CORE.debug("HP Atual.: #{@current_hp}")
    FO_CORE.debug("MP Atual.: #{@current_mp}")
    FO_CORE.debug("Morto?...: #{@dead}")
    FO_CORE.debug("==============================")

  end

  #===========================================================================
  # ● Receber dano
  #===========================================================================
  def take_damage(amount)

    # Não recebe dano morto
    return if @dead

    # Reduz HP
    @current_hp -= amount

    # Evita HP negativo
    @current_hp = 0 if @current_hp < 0

    FO_CORE.debug("Jogador recebeu #{amount} dano")
    FO_CORE.debug("HP Atual: #{@current_hp}")

    # Verifica morte
    check_death

  end

  #===========================================================================
  # ● Curar HP
  #===========================================================================
  def heal(amount)

    # Não cura morto
    return if @dead

    # Cura HP
    @current_hp += amount

    # Limite máximo
    if @current_hp > @player_stats.max_hp
      @current_hp = @player_stats.max_hp
    end

    FO_CORE.debug("Jogador curou #{amount} HP")
    FO_CORE.debug("HP Atual: #{@current_hp}")

  end

  #===========================================================================
  # ● Gastar MP
  #===========================================================================
  def spend_mp(amount)

    # Verifica mana suficiente
    return false if @current_mp < amount

    # Reduz MP
    @current_mp -= amount

    FO_CORE.debug("Jogador gastou #{amount} MP")
    FO_CORE.debug("MP Atual: #{@current_mp}")

    return true

  end

  #===========================================================================
  # ● Regeneração
  #===========================================================================
  def regenerate

    # Não regenera morto
    return if @dead

    # Regenera HP
    @current_hp += 1

    # Regenera MP
    @current_mp += 1

    # Limites máximos
    @current_hp = @player_stats.max_hp if @current_hp > @player_stats.max_hp
    @current_mp = @player_stats.max_mp if @current_mp > @player_stats.max_mp

    FO_CORE.debug("Regeneração automática")

  end

  #===========================================================================
  # ● Verificar morte
  #===========================================================================
  def check_death

    # Verifica HP
    if @current_hp <= 0

      # Morre
      @dead = true

      FO_CORE.debug("O jogador morreu")

    end

  end

end

#==============================================================================
# ■ FO_Engine
#------------------------------------------------------------------------------
# Integra Status à Engine
#==============================================================================

class FO_Engine

  #--------------------------------------------------------------------------
  # ● Alias Initialize
  #--------------------------------------------------------------------------
  alias fo_status_initialize initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize

    # Executa original
    fo_status_initialize

    # Cria status
    @player_status = FO_Player_Status.new(@player_stats)

    # Testes
    @player_status.take_damage(30)

    @player_status.heal(10)

    @player_status.spend_mp(5)

  end

  #--------------------------------------------------------------------------
  # ● Alias Update Second
  #--------------------------------------------------------------------------
  alias fo_status_update_second update_second

  #--------------------------------------------------------------------------
  # ● Update Second
  #--------------------------------------------------------------------------
  def update_second

    # Executa original
    fo_status_update_second

    # Regeneração automática
    @player_status.regenerate if @player_status

  end

end
