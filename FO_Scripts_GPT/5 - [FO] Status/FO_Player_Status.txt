#==============================================================================
# ■ FO_PLAYER_STATUS
#==============================================================================

module FO_PLAYER_STATUS

  #============================================================================
  # ■ CONFIGURAÇÕES
  #============================================================================

  #--------------------------------------------------------------------------
  # ● Regen
  #--------------------------------------------------------------------------

  HP_REGEN_INTERVAL =
    3 * Graphics.frame_rate

  MP_REGEN_INTERVAL =
    2 * Graphics.frame_rate

  #--------------------------------------------------------------------------
  # ● Valores Base
  #--------------------------------------------------------------------------

  BASE_HP_REGEN = 2

  BASE_MP_REGEN = 3

  #--------------------------------------------------------------------------
  # ● Multiplicadores
  #--------------------------------------------------------------------------

  VITALITY_HP_BONUS = 5

  INTELLIGENCE_MP_BONUS = 4

end

#==============================================================================
# ■ FO_Player_Status
#==============================================================================

module FO_Player_Status

  #--------------------------------------------------------------------------
  # ■ Timers
  #--------------------------------------------------------------------------

  @hp_regen_timer = 0
  @mp_regen_timer = 0

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_hp_regen

    update_mp_regen

    update_dead_state

  end

  #--------------------------------------------------------------------------
  # ● Update HP Regen
  #--------------------------------------------------------------------------

  def self.update_hp_regen

    return unless defined?(FO_Player_Data)

    return if FO_Player_Data.dead?

    @hp_regen_timer += 1

    return if @hp_regen_timer <
              FO_PLAYER_STATUS::HP_REGEN_INTERVAL

    @hp_regen_timer = 0

    regenerate_hp

  end

  #--------------------------------------------------------------------------
  # ● Update MP Regen
  #--------------------------------------------------------------------------

  def self.update_mp_regen

    return unless defined?(FO_Player_Data)

    return if FO_Player_Data.dead?

    @mp_regen_timer += 1

    return if @mp_regen_timer <
              FO_PLAYER_STATUS::MP_REGEN_INTERVAL

    @mp_regen_timer = 0

    regenerate_mp

  end

  #--------------------------------------------------------------------------
  # ● Regenerate HP
  #--------------------------------------------------------------------------

  def self.regenerate_hp

    amount =
      FO_PLAYER_STATUS::BASE_HP_REGEN

    if defined?(FO_Player_Stats)

      amount +=
        (
          FO_Player_Stats.vit /
          FO_PLAYER_STATUS::VITALITY_HP_BONUS
        )

    end

    heal(amount)

  end

  #--------------------------------------------------------------------------
  # ● Regenerate MP
  #--------------------------------------------------------------------------

  def self.regenerate_mp

    amount =
      FO_PLAYER_STATUS::BASE_MP_REGEN

    if defined?(FO_Player_Stats)

      amount +=
        (
          FO_Player_Stats.int /
          FO_PLAYER_STATUS::INTELLIGENCE_MP_BONUS
        )

    end

    recover_mFO_Core.logamount)

  end

  #--------------------------------------------------------------------------
  # ● Heal
  #--------------------------------------------------------------------------

  def self.heal(amount)

    return unless defined?(FO_Player_Data)

    return if amount <= 0

    current_hp =
      FO_Player_Data.hp

    max_hp =
      FO_Player_Data.maxhp

    current_hp += amount

    if current_hp > max_hp

      current_hp = max_hp

    end

    FO_Player_Data.hp = current_hp

  end

  #--------------------------------------------------------------------------
  # ● Damage
  #--------------------------------------------------------------------------

  def self.damage(amount)

    return unless defined?(FO_Player_Data)

    return if amount <= 0

    current_hp =
      FO_Player_Data.hp

    current_hp -= amount

    if current_hp < 0

      current_hp = 0

    end

    FO_Player_Data.hp = current_hp

    check_death

  end

  #--------------------------------------------------------------------------
  # ● Recover MP
  #--------------------------------------------------------------------------

  def self.recover_mFO_Core.logamount)

    return unless defined?(FO_Player_Data)

    return if amount <= 0

    current_mp =
      FO_Player_Data.mp

    max_mp =
      FO_Player_Data.maxmp

    current_mp += amount

    if current_mp > max_mp

      current_mp = max_mp

    end

    FO_Player_Data.mp = current_mp

  end

  #--------------------------------------------------------------------------
  # ● Spend MP
  #--------------------------------------------------------------------------

  def self.spend_mFO_Core.logamount)

    return false unless defined?(FO_Player_Data)

    return false if amount <= 0

    current_mp =
      FO_Player_Data.mp

    return false if current_mp < amount

    current_mp -= amount

    if current_mp < 0

      current_mp = 0

    end

    FO_Player_Data.mp = current_mp

    return true

  end

  #--------------------------------------------------------------------------
  # ● Full Recover
  #--------------------------------------------------------------------------

  def self.full_recover

    return unless defined?(FO_Player_Data)

    FO_Player_Data.hp =
      FO_Player_Data.maxhp

    FO_Player_Data.mp =
      FO_Player_Data.maxmp

  end

  #--------------------------------------------------------------------------
  # ● Check Death
  #--------------------------------------------------------------------------

  def self.check_death

    return unless defined?(FO_Player_Data)

    if FO_Player_Data.hp <= 0

      FO_Player_Data.hp = 0

      FO_Player_Data.dead = true

      on_player_death

    end

  end

  #--------------------------------------------------------------------------
  # ● Update Dead State
  #--------------------------------------------------------------------------

  def self.update_dead_state

    return unless defined?(FO_Player_Data)

    if FO_Player_Data.hp > 0

      FO_Player_Data.dead = false

    end

  end

  #--------------------------------------------------------------------------
  # ● Revive
  #--------------------------------------------------------------------------

  def self.revive(hp_percent = 50)

    return unless defined?(FO_Player_Data)

    max_hp =
      FO_Player_Data.maxhp

    recover_hp =
      (
        max_hp *
        hp_percent / 100.0
      ).to_i

    if recover_hp <= 0

      recover_hp = 1

    end

    FO_Player_Data.hp = recover_hp

    FO_Player_Data.dead = false

    FO_Player_Data.mp =
      FO_Player_Data.maxmp

    if defined?(FO_DEBUG)

      FO_DEBUG.warning(
        "[PLAYER REVIVE] #{recover_hp} HP"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Player Death
  #--------------------------------------------------------------------------

  def self.on_player_death

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.error(
        "[PLAYER DEAD]"
      )

    end

    #--------------------------------------------------------------------------
    # Limpa target
    #--------------------------------------------------------------------------

    if defined?(FO_Target_System)

      FO_Target_System.clear_target

    end

  end

  #--------------------------------------------------------------------------
  # ● HP Percent
  #--------------------------------------------------------------------------

  def self.hp_percent

    return 0 unless defined?(FO_Player_Data)

    max_hp =
      FO_Player_Data.maxhp

    return 0 if max_hp <= 0

    hp =
      FO_Player_Data.hp

    return (
      hp * 100 / max_hp
    ).to_i

  end

  #--------------------------------------------------------------------------
  # ● MP Percent
  #--------------------------------------------------------------------------

  def self.mp_percent

    return 0 unless defined?(FO_Player_Data)

    max_mp =
      FO_Player_Data.maxmp

    return 0 if max_mp <= 0

    mp =
      FO_Player_Data.mp

    return (
      mp * 100 / max_mp
    ).to_i

  end

end

  begin

    FO_ENGINE.register_system(
      FO_Player_Status
    )

  rescue

  end
