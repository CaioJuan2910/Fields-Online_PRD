module FO_TARGET_SYSTEM

  #============================================================================
  # ■ CONFIGURAÇÕES
  #============================================================================

  TARGET_KEY =
    Input::SHIFT

  TARGET_RANGE = 8

end

#==============================================================================
# ■ FO_Target_System
#==============================================================================

module FO_Target_System

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @target = nil

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_target_input

    update_target_validity

  end

  #--------------------------------------------------------------------------
  # ● Target
  #--------------------------------------------------------------------------

  def self.target

    return @target

  end

  #--------------------------------------------------------------------------
  # ● Set Target
  #--------------------------------------------------------------------------

  def self.set_target(target)

    @target = target

    if defined?(FO_DEBUG)

      if target != nil

        FO_DEBUG.log(
          "[TARGET] #{target.event.name}"
        )

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear Target
  #--------------------------------------------------------------------------

  def self.clear_target

    @target = nil

  end

  #--------------------------------------------------------------------------
  # ● Has Target?
  #--------------------------------------------------------------------------

  def self.has_target?

    return @target != nil

  end

  #--------------------------------------------------------------------------
  # ● Valid Target?
  #--------------------------------------------------------------------------

  def self.valid_target?

    return false if @target.nil?

    return false unless @target.respond_to?(
      :fo_combat_entity?
    )

    return false unless @target.fo_combat_entity?

    return false if @target.fo_dead?

    return false if @target.transparent

    return true

  end

  #--------------------------------------------------------------------------
  # ● Update Input
  #--------------------------------------------------------------------------

  def self.update_target_input

    return unless Input.trigger?(
      FO_TARGET_SYSTEM::TARGET_KEY
    )

    select_nearest_target

  end

  #--------------------------------------------------------------------------
  # ● Select Nearest Target
  #--------------------------------------------------------------------------

  def self.select_nearest_target

    return if $game_map.nil?

    return if $game_player.nil?

    nearest_target = nil

    nearest_distance = nil

    #--------------------------------------------------------------------------
    # ● Busca eventos
    #--------------------------------------------------------------------------

    for event in $game_map.events.values

      next if event.nil?

      next unless valid_event_target?(event)

      distance =
        calculate_distance(event)

      next if distance >
              FO_TARGET_SYSTEM::TARGET_RANGE

      if nearest_target.nil?

        nearest_target = event

        nearest_distance = distance

        next

      end

      if distance < nearest_distance

        nearest_target = event

        nearest_distance = distance

      end

    end

    #--------------------------------------------------------------------------
    # ● Resultado
    #--------------------------------------------------------------------------

    if nearest_target != nil

      set_target(nearest_target)

    else

      clear_target

    end

  end

  #--------------------------------------------------------------------------
  # ● Valid Event Target?
  #--------------------------------------------------------------------------

  def self.valid_event_target?(event)

    return false if event.nil?

    return false unless event.respond_to?(
      :fo_combat_entity?
    )

    return false unless event.fo_combat_entity?

    return false if event.fo_dead?

    return false if event.transparent

    return true

  end

  #--------------------------------------------------------------------------
  # ● Calculate Distance
  #--------------------------------------------------------------------------

  def self.calculate_distance(target)

    player = $game_player

    return 999 unless player

    dx = (target.x - player.x).abs
    dy = (target.y - player.y).abs

    return dx + dy

  end

  #--------------------------------------------------------------------------
  # ● Update Validity
  #--------------------------------------------------------------------------

  def self.update_target_validity

    return if @target.nil?

    #--------------------------------------------------------------------------
    # ● Removido mapa
    #--------------------------------------------------------------------------

    if $game_map.nil?

      clear_target

      return

    end

    #--------------------------------------------------------------------------
    # ● Evento removido
    #--------------------------------------------------------------------------

    unless $game_map.events.values.include?(@target)

      clear_target

      return

    end

    #--------------------------------------------------------------------------
    # ● Morto
    #--------------------------------------------------------------------------

    if @target.respond_to?(:fo_dead?)

      if @target.fo_dead?

        clear_target

        return

      end

    end

    #--------------------------------------------------------------------------
    # ● Transparente
    #--------------------------------------------------------------------------

    if @target.transparent

      clear_target

      return

    end

    #--------------------------------------------------------------------------
    # ● Distância
    #--------------------------------------------------------------------------

    distance =
      calculate_distance(@target)

    if distance >
       FO_TARGET_SYSTEM::TARGET_RANGE

      clear_target

      return

    end

  end

  #--------------------------------------------------------------------------
  # ● Target Name
  #--------------------------------------------------------------------------

  def self.target_name

    return "" if @target.nil?

    return "" unless @target.event

    return @target.event.name

  end

  #--------------------------------------------------------------------------
  # ● Target HP
  #--------------------------------------------------------------------------

  def self.target_hp

    return 0 if @target.nil?

    return 0 unless @target.respond_to?(:fo_hp)

    return @target.fo_hp

  end

  #--------------------------------------------------------------------------
  # ● Target MaxHP
  #--------------------------------------------------------------------------

  def self.target_maxhp

    return 0 if @target.nil?

    return 0 unless @target.respond_to?(:fo_maxhp)

    return @target.fo_maxhp

  end

  #--------------------------------------------------------------------------
  # ● Target HP Percent
  #--------------------------------------------------------------------------

  def self.target_hp_percent

    hp =
      target_hp

    maxhp =
      target_maxhp

    return 0 if maxhp <= 0

    return (
      hp * 100 / maxhp
    ).to_i

  end

  #--------------------------------------------------------------------------
  # ● Boss?
  #--------------------------------------------------------------------------

  def self.target_boss?

    return false if @target.nil?

    return false unless @target.respond_to?(
      :fo_boss?
    )

    return @target.fo_boss?

  end

end

  begin

    FO_ENGINE.register_system(
      FO_Target_System
    )

  rescue

  end

end

  )

end
