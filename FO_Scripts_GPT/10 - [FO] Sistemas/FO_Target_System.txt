#==============================================================================
# ■ FO_Target_System
#==============================================================================

module FO_Target_System

  #--------------------------------------------------------------------------
  # ■ Current Target
  #--------------------------------------------------------------------------

  @target = nil

  #--------------------------------------------------------------------------
  # ● Target
  #--------------------------------------------------------------------------
  def self.target

    return @target

  end

  #--------------------------------------------------------------------------
  # ● Set Target
  #--------------------------------------------------------------------------
  def self.set_target(target)

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if target.nil?

    #--------------------------------------------------------------------------
    # Combat Entity?
    #--------------------------------------------------------------------------

    return unless target.respond_to?(
      :fo_combat_entity?
    )

    return unless target.fo_combat_entity?

    #--------------------------------------------------------------------------
    # Dead?
    #--------------------------------------------------------------------------

    return if target.fo_dead?

    #--------------------------------------------------------------------------
    # Define
    #--------------------------------------------------------------------------

    @target =
      target

    #--------------------------------------------------------------------------
    # Compatibilidade antiga
    #--------------------------------------------------------------------------

    $fo_target =
      target

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[TARGET SET] #{@target.event.name}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Clear Target
  #--------------------------------------------------------------------------
  def self.clear_target

    return if @target.nil?

    FO_CORE.debug(
      "[TARGET CLEAR]"
    )

    @target = nil

    #--------------------------------------------------------------------------
    # Compatibilidade antiga
    #--------------------------------------------------------------------------

    $fo_target = nil

  end

  #--------------------------------------------------------------------------
  # ● Valid Target?
  #--------------------------------------------------------------------------
  def self.valid_target?

    return false if @target.nil?

    #--------------------------------------------------------------------------
    # Existe?
    #--------------------------------------------------------------------------

    begin

      @target.event

    rescue

      clear_target

      return false

    end

    #--------------------------------------------------------------------------
    # Morto?
    #--------------------------------------------------------------------------

    if @target.fo_dead?

      clear_target

      return false

    end

    #--------------------------------------------------------------------------
    # HP
    #--------------------------------------------------------------------------

    if @target.fo_hp <= 0

      clear_target

      return false

    end

    return true

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    #--------------------------------------------------------------------------
    # Target inválido
    #--------------------------------------------------------------------------

    unless valid_target?

      clear_target

      return

    end

  end

end

#==============================================================================
# ■ Registro no System Manager
#==============================================================================

FO_System_Manager.register(
  FO_Target_System
)

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_CORE.debug(
  "FO_Target_System_v2 carregado com sucesso"
)
