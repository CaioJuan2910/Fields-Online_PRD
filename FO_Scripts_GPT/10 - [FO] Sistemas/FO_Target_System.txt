#==============================================================================
# ■ FO_Target_System
#==============================================================================

module FO_Target_System

  #--------------------------------------------------------------------------
  # ■ Current Target
  #--------------------------------------------------------------------------

  @target = nil

  #--------------------------------------------------------------------------
  # ● Target
  #--------------------------------------------------------------------------
  def self.target

    return @target

  end

  #--------------------------------------------------------------------------
  # ● Set Target
  #--------------------------------------------------------------------------
  def self.set_target(target)

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if target.nil?

    #--------------------------------------------------------------------------
    # Combat Entity?
    #--------------------------------------------------------------------------

    return unless target.respond_to?(
      :fo_combat_entity?
    )

    return unless target.fo_combat_entity?

    #--------------------------------------------------------------------------
    # Entity válida?
    #--------------------------------------------------------------------------

    return unless FO_Entity_Manager.valid_entity?(
      target
    )

    #--------------------------------------------------------------------------
    # Define
    #--------------------------------------------------------------------------

    @target =
      target

    #--------------------------------------------------------------------------
    # Compatibilidade temporária
    #--------------------------------------------------------------------------

    $fo_target =
      target

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    begin

      FO_CORE.debug(
        "[TARGET SET] #{target.event.name}"
      )

    rescue

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear Target
  #--------------------------------------------------------------------------
  def self.clear_target

    return if @target.nil?

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    begin

      FO_CORE.debug(
        "[TARGET CLEAR] #{@target.event.name}"
      )

    rescue

    end

    #--------------------------------------------------------------------------
    # Remove
    #--------------------------------------------------------------------------

    @target = nil

    #--------------------------------------------------------------------------
    # Compatibilidade antiga
    #--------------------------------------------------------------------------

    $fo_target = nil

  end

  #--------------------------------------------------------------------------
  # ● Valid Target?
  #--------------------------------------------------------------------------
  def self.valid_target?

    return false if @target.nil?

    #--------------------------------------------------------------------------
    # Entity válida?
    #--------------------------------------------------------------------------

    unless FO_Entity_Manager.valid_entity?(
      @target
    )

      clear_target

      return false

    end

    return true

  end

  #--------------------------------------------------------------------------
  # ● Target Dead?
  #--------------------------------------------------------------------------
  def self.target_dead?

    return true unless valid_target?

    return @target.fo_dead?

  end

  #--------------------------------------------------------------------------
  # ● Target HP
  #--------------------------------------------------------------------------
  def self.target_hp

    return 0 unless valid_target?

    return @target.fo_hp

  end

  #--------------------------------------------------------------------------
  # ● Target MaxHP
  #--------------------------------------------------------------------------
  def self.target_maxhp

    return 0 unless valid_target?

    return @target.fo_maxhp

  end

  #--------------------------------------------------------------------------
  # ● Target Name
  #--------------------------------------------------------------------------
  def self.target_name

    return "" unless valid_target?

    begin

      return @target.fo_enemy_data.name

    rescue

      return ""

    end

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    #--------------------------------------------------------------------------
    # Target inválido
    #--------------------------------------------------------------------------

    unless valid_target?

      clear_target

      return

    end

  end

end

#==============================================================================
# ■ Registro no System Manager
#==============================================================================

FO_System_Manager.register(
  FO_Target_System
)

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_CORE.debug(
  "FO_Target_System_v3 carregado com sucesso"
)
