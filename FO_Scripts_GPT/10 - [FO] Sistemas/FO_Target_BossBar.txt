#==============================================================================
# ■ FO_TARGET_BOSSBAR
#==============================================================================

module FO_TARGET_BOSSBAR

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  WIDTH = 420

  HEIGHT = 22

  X = 110

  Y = 20

  Z = 99999

  #--------------------------------------------------------------------------
  # ● Cores
  #--------------------------------------------------------------------------

  BACK_COLOR =
    Color.new(0,0,0)

  HP_COLOR =
    Color.new(220,40,40)

  BORDER_COLOR =
    Color.new(255,255,255)

  TEXT_COLOR =
    Color.new(255,255,255)

  #--------------------------------------------------------------------------
  # ● Fonte
  #--------------------------------------------------------------------------

  FONT_NAME = "Arial"

  FONT_SIZE = 18

end

#==============================================================================
# ■ FO_Target_BossBar
#==============================================================================

module FO_Target_BossBar

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @sprite = nil

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_visibility

    update_bar

  end

  #--------------------------------------------------------------------------
  # ● Update Visibility
  #--------------------------------------------------------------------------

  def self.update_visibility

    target =
      FO_Target_System.target

    #--------------------------------------------------------------------------
    # ● Sem target
    #--------------------------------------------------------------------------

    if target.nil?

      dispose

      return

    end

    #--------------------------------------------------------------------------
    # ● Não é boss
    #--------------------------------------------------------------------------

    unless FO_Target_System.target_boss?

      dispose

      return

    end

    #--------------------------------------------------------------------------
    # ● Cria sprite
    #--------------------------------------------------------------------------

    if @sprite.nil?

      create_sprite

    end

  end

  #--------------------------------------------------------------------------
  # ● Create Sprite
  #--------------------------------------------------------------------------

  def self.create_sprite

    dispose

    @sprite =
      Sprite.new

    @sprite.bitmap =
      Bitmap.new(
        FO_TARGET_BOSSBAR::WIDTH,
        48
      )

    @sprite.x =
      FO_TARGET_BOSSBAR::X

    @sprite.y =
      FO_TARGET_BOSSBAR::Y

    @sprite.z =
      FO_TARGET_BOSSBAR::Z

  end

  #--------------------------------------------------------------------------
  # ● Update Bar
  #--------------------------------------------------------------------------

  def self.update_bar

    return if @sprite.nil?

    bitmap =
      @sprite.bitmap

    bitmap.clear

    #--------------------------------------------------------------------------
    # ● Dados
    #--------------------------------------------------------------------------

    hp =
      FO_Target_System.target_hp

    maxhp =
      FO_Target_System.target_maxhp

    name =
      FO_Target_System.target_name

    return if maxhp <= 0

    #--------------------------------------------------------------------------
    # ● Percentual
    #--------------------------------------------------------------------------

    percent =
      hp.to_f / maxhp.to_f

    fill_width =
      (
        (
          FO_TARGET_BOSSBAR::WIDTH - 4
        ) * percent
      ).to_i

    #--------------------------------------------------------------------------
    # ● Fundo
    #--------------------------------------------------------------------------

    bitmap.fill_rect(
      0,
      20,
      FO_TARGET_BOSSBAR::WIDTH,
      FO_TARGET_BOSSBAR::HEIGHT,
      FO_TARGET_BOSSBAR::BACK_COLOR
    )

    #--------------------------------------------------------------------------
    # ● HP
    #--------------------------------------------------------------------------

    bitmap.fill_rect(
      2,
      22,
      fill_width,
      FO_TARGET_BOSSBAR::HEIGHT - 4,
      FO_TARGET_BOSSBAR::HP_COLOR
    )

    #--------------------------------------------------------------------------
    # ● Bordas
    #--------------------------------------------------------------------------

    draw_border(bitmap)

    #--------------------------------------------------------------------------
    # ● Fonte
    #--------------------------------------------------------------------------

    font =
      bitmap.font

    font.name =
      FO_TARGET_BOSSBAR::FONT_NAME

    font.size =
      FO_TARGET_BOSSBAR::FONT_SIZE

    font.bold = true

    font.color =
      FO_TARGET_BOSSBAR::TEXT_COLOR

    #--------------------------------------------------------------------------
    # ● Nome
    #--------------------------------------------------------------------------

    clean =
      clean_name(name)

    bitmap.draw_text(
      0,
      0,
      FO_TARGET_BOSSBAR::WIDTH,
      20,
      clean,
      1
    )

  end

  #--------------------------------------------------------------------------
  # ● Draw Border
  #--------------------------------------------------------------------------

  def self.draw_border(bitmap)

    color =
      FO_TARGET_BOSSBAR::BORDER_COLOR

    width =
      FO_TARGET_BOSSBAR::WIDTH

    height =
      FO_TARGET_BOSSBAR::HEIGHT

    y = 20

    # Top
    bitmap.fill_rect(
      0,
      y,
      width,
      1,
      color
    )

    # Bottom
    bitmap.fill_rect(
      0,
      y + height - 1,
      width,
      1,
      color
    )

    # Left
    bitmap.fill_rect(
      0,
      y,
      1,
      height,
      color
    )

    # Right
    bitmap.fill_rect(
      width - 1,
      y,
      1,
      height,
      color
    )

  end

  #--------------------------------------------------------------------------
  # ● Clean Name
  #--------------------------------------------------------------------------

  def self.clean_name(name)

    return "" if name.nil?

    text = name.clone

    text.gsub!("!", "")
    text.gsub!("$", "")
    text.gsub!("@", "")
    text.gsub!("?", "")

    return text

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def self.dispose

    return if @sprite.nil?

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    @sprite.dispose

    @sprite = nil

  end

end

#==============================================================================
# ■ Registro no FO_ENGINE
#==============================================================================

if defined?(FO_ENGINE)

  begin

    FO_ENGINE.register_system(
      FO_Target_BossBar
    )

  rescue

  end

end

#==============================================================================
# ■ Inicialização
#==============================================================================

if defined?(FO_DEBUG)

  FO_DEBUG.log(
    "FO_Target_BossBar_v4 carregado com sucesso"
  )

end
