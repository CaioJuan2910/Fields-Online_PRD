class FO_Player_Status

  #===========================================================================
  # ● Initialize
  #===========================================================================

  def initialize

    #--------------------------------------------------------------------------
    # Player Actor
    #--------------------------------------------------------------------------

    @player =
      $game_party.actors[0]

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if @player.nil?

    #--------------------------------------------------------------------------
    # HP/SP iniciais
    #--------------------------------------------------------------------------

    @player.hp =
      @player.maxhp

    @player.sp =
      @player.maxsp

    #--------------------------------------------------------------------------
    # Dead
    #--------------------------------------------------------------------------

    @dead =
      false

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "FO_Player_Status_v2 iniciado"
    )

    show_status

  end

  #===========================================================================
  # ● Show Status
  #===========================================================================

  def show_status

    return if @player.nil?

    FO_CORE.debug("==============================")
    FO_CORE.debug("PLAYER STATUS")
    FO_CORE.debug("HP Atual.: #{@player.hp}")
    FO_CORE.debug("HP Máximo: #{@player.maxhp}")
    FO_CORE.debug("MP Atual.: #{@player.sp}")
    FO_CORE.debug("MP Máximo: #{@player.maxsp}")
    FO_CORE.debug("==============================")

  end

  #===========================================================================
  # ● Take Damage
  #===========================================================================

  def take_damage(amount)

    return if @player.nil?

    return if dead?

    #--------------------------------------------------------------------------
    # Reduz HP
    #--------------------------------------------------------------------------

    @player.hp -= amount

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    if @player.hp < 0

      @player.hp = 0

    end

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[PLAYER DAMAGE] #{amount}"
    )

    #--------------------------------------------------------------------------
    # Death
    #--------------------------------------------------------------------------

    check_death

  end

  #===========================================================================
  # ● Heal
  #===========================================================================

  def heal(amount)

    return if @player.nil?

    return if dead?

    #--------------------------------------------------------------------------
    # Cura
    #--------------------------------------------------------------------------

    @player.hp += amount

    #--------------------------------------------------------------------------
    # Limite
    #--------------------------------------------------------------------------

    if @player.hp > @player.maxhp

      @player.hp = @player.maxhp

    end

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[PLAYER HEAL] #{amount}"
    )

  end

  #===========================================================================
  # ● Spend MP
  #===========================================================================

  def spend_mp(amount)

    return false if @player.nil?

    #--------------------------------------------------------------------------
    # MP insuficiente
    #--------------------------------------------------------------------------

    if @player.sp < amount

      return false

    end

    #--------------------------------------------------------------------------
    # Reduz
    #--------------------------------------------------------------------------

    @player.sp -= amount

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[PLAYER MP SPEND] #{amount}"
    )

    return true

  end

  #===========================================================================
  # ● Regenerate
  #===========================================================================

  def regenerate

    return if @player.nil?

    return if dead?

    #--------------------------------------------------------------------------
    # HP
    #--------------------------------------------------------------------------

    @player.hp += 1

    #--------------------------------------------------------------------------
    # SP
    #--------------------------------------------------------------------------

    @player.sp += 1

    #--------------------------------------------------------------------------
    # Limites
    #--------------------------------------------------------------------------

    if @player.hp > @player.maxhp

      @player.hp = @player.maxhp

    end

    if @player.sp > @player.maxsp

      @player.sp = @player.maxsp

    end

  end

  #===========================================================================
  # ● Dead?
  #===========================================================================

  def dead?

    return @dead

  end

  #===========================================================================
  # ● Check Death
  #===========================================================================

  def check_death

    return if @player.nil?

    #--------------------------------------------------------------------------
    # HP <= 0
    #--------------------------------------------------------------------------

    if @player.hp <= 0

      @dead = true

      FO_CORE.debug(
        "[PLAYER DEAD]"
      )

    end

  end

end

#==============================================================================
# ■ FO_Engine
#==============================================================================

class FO_Engine

  #--------------------------------------------------------------------------
  # Alias Initialize
  #--------------------------------------------------------------------------

  alias fo_player_status_v2_initialize initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize

    fo_player_status_v2_initialize

    #--------------------------------------------------------------------------
    # Player Status
    #--------------------------------------------------------------------------

    @player_status =
      FO_Player_Status.new

  end

  #--------------------------------------------------------------------------
  # Alias Update Second
  #--------------------------------------------------------------------------

  alias fo_player_status_v2_update_second update_second

  #--------------------------------------------------------------------------
  # ● Update Second
  #--------------------------------------------------------------------------

  def update_second

    fo_player_status_v2_update_second

    #--------------------------------------------------------------------------
    # Regen
    #--------------------------------------------------------------------------

    if @player_status != nil

      @player_status.regenerate

    end

  end

end
