module FO_COMBAT_MANAGER

  MIN_DAMAGE = 1

  MAX_VARIANCE = 6

  CRITICAL_MULTIPLIER = 1.5

end

#==============================================================================
# ■ FO_Combat_Manager
#==============================================================================

module FO_Combat_Manager

  #--------------------------------------------------------------------------
  # ● Attack Target
  #--------------------------------------------------------------------------

  def self.attack_target(attacker, target)

    return if attacker == nil
    return if target == nil

    return unless valid_target?(target)

    result = calculate_damage(attacker, target)

    damage = result[:damage]
    text   = result[:text]
    type   = result[:type]

    if damage <= 0

      show_damage(target, text, type)

      return

    end

    #--------------------------------------------------------------------------
    # ● Combat State
    #--------------------------------------------------------------------------

    target.fo_in_combat = true

    if defined?(FO_TARGET_STATUS)

      target.fo_combat_time =
        FO_TARGET_STATUS::COMBAT_MEMORY_TIME

    end

    #--------------------------------------------------------------------------
    # ● Apply Damage
    #--------------------------------------------------------------------------

    apply_damage(target, damage)

    #--------------------------------------------------------------------------
    # ● Floating Damage
    #--------------------------------------------------------------------------

    show_damage(target, text, type)

    #--------------------------------------------------------------------------
    # ● Death
    #--------------------------------------------------------------------------

    if target.fo_hp <= 0

      process_death(attacker, target)

    end

  end

  #--------------------------------------------------------------------------
  # ● Valid Target?
  #--------------------------------------------------------------------------

  def self.valid_target?(target)

    return false unless target.respond_to?(
      :fo_combat_entity?
    )

    return false unless target.fo_combat_entity?

    return false if target.fo_dead?

    return false if target.fo_hp <= 0

    return true

  end

  #--------------------------------------------------------------------------
  # ● Calculate Damage
  #--------------------------------------------------------------------------

  def self.calculate_damage(attacker, target)

    result = {

      :damage => 0,
      :text   => "",
      :type   => :normal

    }

    #--------------------------------------------------------------------------
    # ● Accuracy
    #--------------------------------------------------------------------------

    accuracy = FO_Player_Stats.accuracy

    roll = rand(100)

    if roll > accuracy

      result[:damage] = 0

      result[:text] = "MISS"

      result[:type] = :miss

      return result

    end

    #--------------------------------------------------------------------------
    # ● Attack / Defense
    #--------------------------------------------------------------------------

    attack  = FO_Player_Stats.attack

    defense = target.fo_pdef

    #--------------------------------------------------------------------------
    # ● Damage Formula
    #--------------------------------------------------------------------------

    damage = attack - (defense / 2)

    if damage < FO_COMBAT_MANAGER::MIN_DAMAGE

      damage = FO_COMBAT_MANAGER::MIN_DAMAGE

    end

    #--------------------------------------------------------------------------
    # ● Variance
    #--------------------------------------------------------------------------

    variance = rand(
      FO_COMBAT_MANAGER::MAX_VARIANCE
    )

    damage += variance

    #--------------------------------------------------------------------------
    # ● Critical
    #--------------------------------------------------------------------------

    critical_rate = FO_Player_Stats.critical

    critical_roll = rand(100)

    if critical_roll < critical_rate

      damage = (
        damage *
        FO_COMBAT_MANAGER::CRITICAL_MULTIPLIER
      ).to_i

      result[:type] = :critical

      result[:text] =
        "CRITICAL " + damage.to_s

    else

      result[:text] =
        damage.to_s

    end

    result[:damage] = damage

    return result

  end

  #--------------------------------------------------------------------------
  # ● Apply Damage
  #--------------------------------------------------------------------------

  def self.apply_damage(target, damage)

    return if target == nil

    target.fo_hp -= damage

    if target.fo_hp < 0

      target.fo_hp = 0

    end

  end

  #--------------------------------------------------------------------------
  # ● Show Damage
  #--------------------------------------------------------------------------

  def self.show_damage(target, text, type)

    return unless defined?(FO_Floating_Damage)

    FO_Floating_Damage.create(
      target,
      text,
      type
    )

  end

  #--------------------------------------------------------------------------
  # ● Process Death
  #--------------------------------------------------------------------------

  def self.process_death(attacker, target)

    return if target == nil

    give_exp(target)

    process_drops(target)

    process_kill(target)

  end

  #--------------------------------------------------------------------------
  # ● Give EXP
  #--------------------------------------------------------------------------

  def self.give_exp(target)

    return unless defined?(FO_Player_Data)

    return unless target.respond_to?(:fo_exp)

    exp = target.fo_exp

    return if exp == nil

    return if exp <= 0

    FO_Player_Data.gain_exp(exp)

    FO_Core.debug(
      "[GAIN EXP] " + exp.to_s
    )

  end

  #--------------------------------------------------------------------------
  # ● Process Drops
  #--------------------------------------------------------------------------

  def self.process_drops(target)

    return unless defined?(FO_DropManager)

    begin

      map_id = $game_map.map_id

      event_id = target.id

      drops = FO_DropManager.generate_drops(
        map_id,
        event_id
      )

      FO_DropManager.deliver_drops(
        drops
      )

    rescue => e

      FO_Core.error(
        "[DROP ERROR] " + e.to_s
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Process Kill
  #--------------------------------------------------------------------------

  def self.process_kill(target)

    begin

      target.fo_kill

      target.transparent = true

      target.through = true

    rescue => e

      FO_Core.error(
        "[KILL ERROR] " + e.to_s
      )

    end

  end

end
