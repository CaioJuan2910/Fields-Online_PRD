#==============================================================================
# ■ Game_Event
#==============================================================================

class Game_Event < Game_Character

  #--------------------------------------------------------------------------
  # ● Alias
  #--------------------------------------------------------------------------

  alias fo_enemy_system_initialize_v4 initialize

  #--------------------------------------------------------------------------
  # ● Accessors
  #--------------------------------------------------------------------------

  attr_accessor :fo_level

  attr_accessor :fo_hp
  attr_accessor :fo_maxhp

  attr_accessor :fo_sp
  attr_accessor :fo_maxsp

  attr_accessor :fo_atk

  attr_accessor :fo_pdef
  attr_accessor :fo_mdef

  attr_accessor :fo_exp

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize(map_id, event)

    fo_enemy_system_initialize_v4(
      map_id,
      event
    )

    setup_fo_enemy

  end

  #--------------------------------------------------------------------------
  # ● Setup Enemy
  #--------------------------------------------------------------------------

  def setup_fo_enemy

    return unless fo_combat_entity?

    enemy_data =
      FO_ENEMY_SYSTEM::ENEMIES[@id]

    return if enemy_data.nil?

    #--------------------------------------------------------------------------
    # ● Nome
    #--------------------------------------------------------------------------

    @fo_enemy_name =
      enemy_data[:name]

    #--------------------------------------------------------------------------
    # ● Level
    #--------------------------------------------------------------------------

    @fo_level =
      enemy_data[:level]

    #--------------------------------------------------------------------------
    # ● HP/SP
    #--------------------------------------------------------------------------

    @fo_maxhp =
      enemy_data[:maxhp]

    @fo_hp =
      @fo_maxhp

    @fo_maxsp =
      enemy_data[:maxsp]

    @fo_sp =
      @fo_maxsp

    #--------------------------------------------------------------------------
    # ● Combate
    #--------------------------------------------------------------------------

    @fo_atk =
      enemy_data[:atk]

    @fo_pdef =
      enemy_data[:pdef]

    @fo_mdef =
      enemy_data[:mdef]

    #--------------------------------------------------------------------------
    # ● EXP
    #--------------------------------------------------------------------------

    @fo_exp =
      enemy_data[:exp]

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[ENEMY SETUP] #{@event.name}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Combat Entity?
  #--------------------------------------------------------------------------

  def fo_combat_entity?

    return false if @event.nil?

    name =
      @event.name

    return true if name.include?("$")
    return true if name.include?("!")

    return false

  end

  #--------------------------------------------------------------------------
  # ● NPC?
  #--------------------------------------------------------------------------

  def fo_npc?

    return false if @event.nil?

    return @event.name.include?("@")

  end

  #--------------------------------------------------------------------------
  # ● Quest?
  #--------------------------------------------------------------------------

  def fo_quest?

    return false if @event.nil?

    return @event.name.include?("?")

  end

  #--------------------------------------------------------------------------
  # ● Alive?
  #--------------------------------------------------------------------------

  def fo_alive?

    return false if fo_dead?

    return false if @fo_hp <= 0

    return true

  end

  #--------------------------------------------------------------------------
  # ● HP Percent
  #--------------------------------------------------------------------------

  def fo_hp_percent

    return 0 if @fo_maxhp.nil?

    return 0 if @fo_maxhp <= 0

    return (
      @fo_hp * 100 / @fo_maxhp
    ).to_i

  end

  #--------------------------------------------------------------------------
  # ● Recover Full
  #--------------------------------------------------------------------------

  def fo_full_recover

    @fo_hp =
      @fo_maxhp

    @fo_sp =
      @fo_maxsp

  end

  #--------------------------------------------------------------------------
  # ● Damage
  #--------------------------------------------------------------------------

  def fo_damage(value)

    return if value <= 0

    @fo_hp -= value

    if @fo_hp < 0

      @fo_hp = 0

    end

  end

  #--------------------------------------------------------------------------
  # ● Heal
  #--------------------------------------------------------------------------

  def fo_heal(value)

    return if value <= 0

    @fo_hp += value

    if @fo_hp > @fo_maxhp

      @fo_hp = @fo_maxhp

    end

  end

end
