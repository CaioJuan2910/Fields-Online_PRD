module FO_SPAWN_SYSTEM

  #--------------------------------------------------------------------------
  # ● Configuração dos Spawns
  #--------------------------------------------------------------------------
  #
  # MAP_ID => [
  #
  #   {
  #     :template_event_id => ID,
  #     :amount => QUANTIDADE
  #   }
  #
  # ]
  #
  #--------------------------------------------------------------------------

  SPAWNS = {

    #--------------------------------------------------------------------------
    # ● Mapa 1
    #--------------------------------------------------------------------------

    1 => [

      {
        :template_event_id => 5,
        :amount => 3
      },

      {
        :template_event_id => 2,
        :amount => 2
      }

    ]

  }

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  RANDOM_SPAWN_RANGE = 8

  MAX_SPAWN_TRIES = 30

end

#==============================================================================
# ■ FO_Spawn_Manager
#==============================================================================

module FO_Spawn_Manager

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @spawned_maps = {}

  #--------------------------------------------------------------------------
  # ● Setup Map Spawns
  #--------------------------------------------------------------------------

  def self.setup_map_spawns

    return if $game_map.nil?

    map_id =
      $game_map.map_id

    return if map_id.nil?

    #--------------------------------------------------------------------------
    # ● Evita duplicação
    #--------------------------------------------------------------------------

    return if @spawned_maps[map_id]

    #--------------------------------------------------------------------------
    # ● Configuração
    #--------------------------------------------------------------------------

    spawn_data =
      FO_SPAWN_SYSTEM::SPAWNS[map_id]

    return if spawn_data.nil?

    #--------------------------------------------------------------------------
    # ● Cria grupos
    #--------------------------------------------------------------------------

    for data in spawn_data

      next if data.nil?

      template_event_id =
        data[:template_event_id]

      amount =
        data[:amount]

      create_spawn_grouFO_Core.log
        template_event_id,
        amount
      )

    end

    #--------------------------------------------------------------------------
    # ● Marca mapa
    #--------------------------------------------------------------------------

    @spawned_maps[map_id] = true

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[SPAWN MAP SETUP] #{map_id}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Create Spawn Group
  #--------------------------------------------------------------------------

  def self.create_spawn_grouFO_Core.logtemplate_id, amount)

    return if $game_map.nil?

    template =
      $game_map.events[template_id]

    return if template.nil?

    #--------------------------------------------------------------------------
    # ● Esconde template
    #--------------------------------------------------------------------------

    template.transparent = true

    template.through = true

    #--------------------------------------------------------------------------
    # ● Clones
    #--------------------------------------------------------------------------

    for i in 1..amount

      create_clone(template)

    end

  end

  #--------------------------------------------------------------------------
  # ● Create Clone
  #--------------------------------------------------------------------------

  def self.create_clone(template)

    return if template.nil?

    #--------------------------------------------------------------------------
    # ● Novo ID
    #--------------------------------------------------------------------------

    new_id =
      next_event_id

    #--------------------------------------------------------------------------
    # ● Evento original
    #--------------------------------------------------------------------------

    original_event =
      template.instance_variable_get(
        :@event
      )

    return if original_event.nil?

    #--------------------------------------------------------------------------
    # ● Clone seguro
    #--------------------------------------------------------------------------

    event_data =
      Marshal.load(
        Marshal.dumFO_Core.logoriginal_event)
      )

    event_data.id = new_id

    #--------------------------------------------------------------------------
    # ● Cria Game_Event
    #--------------------------------------------------------------------------

    cloned_event =
      Game_Event.new(
        $game_map.map_id,
        event_data
      )

    #--------------------------------------------------------------------------
    # ● Spawn Position
    #--------------------------------------------------------------------------

    x, y =
      random_spawn_position(
        template.x,
        template.y
      )

    cloned_event.moveto(x, y)

    #--------------------------------------------------------------------------
    # ● Adiciona mapa
    #--------------------------------------------------------------------------

    $game_map.events[new_id] =
      cloned_event

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[SPAWN CLONE] #{new_id}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Random Spawn Position
  #--------------------------------------------------------------------------

  def self.random_spawn_position(origin_x, origin_y)

    tries =
      FO_SPAWN_SYSTEM::MAX_SPAWN_TRIES

    max_range =
      FO_SPAWN_SYSTEM::RANDOM_SPAWN_RANGE

    while tries > 0

      x =
        origin_x +
        rand(max_range * 2 + 1) -
        max_range

      y =
        origin_y +
        rand(max_range * 2 + 1) -
        max_range

      #--------------------------------------------------------------------------
      # ● Segurança
      #--------------------------------------------------------------------------

      if invalid_position?(x, y)

        tries -= 1

        next

      end

      #--------------------------------------------------------------------------
      # ● Passável
      #--------------------------------------------------------------------------

      if $game_map.passable?(x, y, 2)

        return [x, y]

      end

      tries -= 1

    end

    #--------------------------------------------------------------------------
    # ● Fallback
    #--------------------------------------------------------------------------

    return [origin_x, origin_y]

  end

  #--------------------------------------------------------------------------
  # ● Invalid Position?
  #--------------------------------------------------------------------------

  def self.invalid_position?(x, y)

    return true if x < 0
    return true if y < 0

    return true if x >= $game_map.width
    return true if y >= $game_map.height

    return false

  end

  #--------------------------------------------------------------------------
  # ● Next Event ID
  #--------------------------------------------------------------------------

  def self.next_event_id

    ids =
      $game_map.events.keys

    return 1 if ids.empty?

    return ids.max + 1

  end

  #--------------------------------------------------------------------------
  # ● Reset Map
  #--------------------------------------------------------------------------

  def self.reset_maFO_Core.logmap_id)

    @spawned_maps.delete(map_id)

  end

end
