module FO_DROPSYSTEM

  #--------------------------------------------------------------------------
  # ● Raridades
  #--------------------------------------------------------------------------

  COMMON     = :common
  RARE       = :rare
  GOLD       = :gold
  SUPER_RARE = :super_rare

  #--------------------------------------------------------------------------
  # ● Drop Tables
  #--------------------------------------------------------------------------

  DROP_TABLES = {

    1 => {

      2 => [

        {
          :type => :item,
          :id => 1,
          :amount => 1,
          :chance => 65,
          :rarity => COMMON
        },

        {
          :type => :gold,
          :id => 0,
          :amount => 25,
          :chance => 80,
          :rarity => COMMON
        },

        {
          :type => :weapon,
          :id => 1,
          :amount => 1,
          :chance => 10,
          :rarity => RARE
        },

        {
          :type => :armor,
          :id => 1,
          :amount => 1,
          :chance => 3,
          :rarity => GOLD
        },

        {
          :type => :weapon,
          :id => 2,
          :amount => 1,
          :chance => 1,
          :rarity => SUPER_RARE
        }

      ]

    }

  }

end

#==============================================================================
# ■ FO_DropManager
#==============================================================================

module FO_DropManager

  #--------------------------------------------------------------------------
  # ● Generate Drops
  #--------------------------------------------------------------------------

  def self.generate_drops(map_id, event_id)

    drops = []

    return drops if map_id == nil
    return drops if event_id == nil

    map_table =
      FO_DROPSYSTEM::DROP_TABLES[map_id]

    return drops if map_table == nil

    event_table =
      map_table[event_id]

    return drops if event_table == nil

    #--------------------------------------------------------------------------
    # ● Processa Drops
    #--------------------------------------------------------------------------

    for drop_data in event_table

      next if drop_data == nil

      chance =
        drop_data[:chance]

      roll =
        rand(100) + 1

      next if roll > chance

      drops.push(drop_data)

    end

    return drops

  end

  #--------------------------------------------------------------------------
  # ● Deliver Drops
  #--------------------------------------------------------------------------

  def self.deliver_drops(drops)

    return if drops == nil

    for drop in drops

      next if drop == nil

      deliver_drop(drop)

    end

  end

  #--------------------------------------------------------------------------
  # ● Deliver Drop
  #--------------------------------------------------------------------------

  def self.deliver_drop(drop)

    type =
      drop[:type]

    id =
      drop[:id]

    amount =
      drop[:amount]

    rarity =
      drop[:rarity]

    #--------------------------------------------------------------------------
    # ● Entrega
    #--------------------------------------------------------------------------

    case type

    when :item

      FO_Inventory.gain_item(
        id,
        amount
      )

    when :weapon

      FO_Inventory.gain_weapon(
        id,
        amount
      )

    when :armor

      FO_Inventory.gain_armor(
        id,
        amount
      )

    when :gold

      FO_Inventory.gain_gold(
        amount
      )

    end

    #--------------------------------------------------------------------------
    # ● Nome
    #--------------------------------------------------------------------------

    name =
      drop_name(
        type,
        id
      )

    #--------------------------------------------------------------------------
    # ● Loot Popup
    #--------------------------------------------------------------------------

    if defined?(FO_Loot_Popup_Manager)

      FO_Loot_Popup_Manager.show_loot(
        name,
        rarity,
        amount
      )

    end

    FO_Core.debug(
      "[DROP] " +
      name.to_s +
      " x" +
      amount.to_s
    )

  end

  #--------------------------------------------------------------------------
  # ● Drop Name
  #--------------------------------------------------------------------------

  def self.drop_name(type, id)

    case type

    when :item

      item =
        $data_items[id]

      return "Unknown Item" if item == nil

      return item.name

    when :weapon

      weapon =
        $data_weapons[id]

      return "Unknown Weapon" if weapon == nil

      return weapon.name

    when :armor

      armor =
        $data_armors[id]

      return "Unknown Armor" if armor == nil

      return armor.name

    when :gold

      return "Gold"

    end

    return "Unknown"

  end

  #--------------------------------------------------------------------------
  # ● Rarity Color
  #--------------------------------------------------------------------------

  def self.rarity_color(rarity)

    case rarity

    when :rare

      return Color.new(80,180,255)

    when :gold

      return Color.new(255,220,80)

    when :super_rare

      return Color.new(255,80,255)

    end

    return Color.new(255,255,255)

  end

end
