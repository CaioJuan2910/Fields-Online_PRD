module FO_PLAYER_DATA

  SAVE_FILE =
    "FO_Player_Data.rxdata"

  AUTO_SAVE_INTERVAL =
    60 * Graphics.frame_rate

end

#==============================================================================
# ■ FO_Player_Data
#==============================================================================

module FO_Player_Data

  #--------------------------------------------------------------------------
  # ■ Accessors
  #--------------------------------------------------------------------------

  class << self

    attr_accessor :name

    attr_accessor :level

    attr_accessor :exp
    attr_accessor :next_exp

    attr_accessor :hp
    attr_accessor :maxhp

    attr_accessor :mp
    attr_accessor :maxmp

    attr_accessor :dead

    attr_accessor :map_id
    attr_accessor :x
    attr_accessor :y

  end

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @auto_save_timer = 0

  #--------------------------------------------------------------------------
  # ● Setup
  #--------------------------------------------------------------------------

  def self.setup

    @name =
      "Player"

    @level =
      1

    @exp =
      0

    @next_exp =
      100

    @maxhp =
      100

    @hp =
      @maxhp

    @maxmp =
      50

    @mp =
      @maxmp

    @dead =
      false

    @map_id =
      1

    @x = 0
    @y = 0

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_position

    update_auto_save

  end

  #--------------------------------------------------------------------------
  # ● Update Position
  #--------------------------------------------------------------------------

  def self.update_position

    return if $game_player.nil?

    return if $game_map.nil?

    @map_id =
      $game_map.map_id

    @x =
      $game_player.x

    @y =
      $game_player.y

  end

  #--------------------------------------------------------------------------
  # ● Update Auto Save
  #--------------------------------------------------------------------------

  def self.update_auto_save

    @auto_save_timer += 1

    return if @auto_save_timer <
              FO_PLAYER_DATA::AUTO_SAVE_INTERVAL

    @auto_save_timer = 0

    save

  end

  #--------------------------------------------------------------------------
  # ● Gain EXP
  #--------------------------------------------------------------------------

  def self.gain_exFO_Core.logamount)

    return if amount <= 0

    @exp += amount

    check_level_up

  end

  #--------------------------------------------------------------------------
  # ● Check Level Up
  #--------------------------------------------------------------------------

  def self.check_level_up

    while @exp >= @next_exp

      level_up

    end

  end

  #--------------------------------------------------------------------------
  # ● Level Up
  #--------------------------------------------------------------------------

  def self.level_up

    @level += 1

    @exp -= @next_exp

    @next_exp =
      calculate_next_exp

    #--------------------------------------------------------------------------
    # ● Crescimento
    #--------------------------------------------------------------------------

    @maxhp += 15

    @maxmp += 8

    @hp =
      @maxhp

    @mp =
      @maxmp

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.warning(
        "[LEVEL UP] #{@level}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Calculate Next EXP
  #--------------------------------------------------------------------------

  def self.calculate_next_exp

    return (
      100 +
      (@level * 75)
    )

  end

  #--------------------------------------------------------------------------
  # ● Dead?
  #--------------------------------------------------------------------------

  def self.dead?

    return @dead == true

  end

  #--------------------------------------------------------------------------
  # ● Save
  #--------------------------------------------------------------------------

  def self.save

    data = {

      :name     => @name,

      :level    => @level,

      :exp      => @exp,
      :next_exp => @next_exp,

      :hp       => @hp,
      :maxhp    => @maxhp,

      :mp       => @mp,
      :maxmp    => @maxmp,

      :dead     => @dead,

      :map_id   => @map_id,
      :x        => @x,
      :y        => @y

    }

    #--------------------------------------------------------------------------
    # ● Inventory
    #--------------------------------------------------------------------------

    if defined?(FO_Inventory)

      data[:inventory] =
        FO_Inventory.save_data

    end

    #--------------------------------------------------------------------------
    # ● Save File
    #--------------------------------------------------------------------------

    begin

      file =
        File.open(
          FO_PLAYER_DATA::SAVE_FILE,
          "wb"
        )

      Marshal.dumFO_Core.logdata, file)

      file.close

      if defined?(FO_DEBUG)

        FO_DEBUG.log(
          "[PLAYER SAVE]"
        )

      end

    rescue => e

      if defined?(FO_DEBUG)

        FO_DEBUG.error(
          "[SAVE ERROR] #{e}"
        )

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Load
  #--------------------------------------------------------------------------

  def self.load

    setup

    return unless File.exist?(
      FO_PLAYER_DATA::SAVE_FILE
    )

    begin

      file =
        File.open(
          FO_PLAYER_DATA::SAVE_FILE,
          "rb"
        )

      data =
        Marshal.load(file)

      file.close

      #--------------------------------------------------------------------------
      # ● Dados
      #--------------------------------------------------------------------------

      @name =
        data[:name]

      @level =
        data[:level]

      @exp =
        data[:exp]

      @next_exp =
        data[:next_exp]

      @hp =
        data[:hp]

      @maxhp =
        data[:maxhp]

      @mp =
        data[:mp]

      @maxmp =
        data[:maxmp]

      @dead =
        data[:dead]

      @map_id =
        data[:map_id]

      @x =
        data[:x]

      @y =
        data[:y]

      #--------------------------------------------------------------------------
      # ● Inventory
      #--------------------------------------------------------------------------

      if defined?(FO_Inventory)

        inventory_data =
          data[:inventory]

        FO_Inventory.load_data(
          inventory_data
        )

      end

      #--------------------------------------------------------------------------
      # ● Debug
      #--------------------------------------------------------------------------

      if defined?(FO_DEBUG)

        FO_DEBUG.log(
          "[PLAYER LOAD]"
        )

      end

    rescue => e

      if defined?(FO_DEBUG)

        FO_DEBUG.error(
          "[LOAD ERROR] #{e}"
        )

      end

      setup

    end

  end

  #--------------------------------------------------------------------------
  # ● Teleport Player
  #--------------------------------------------------------------------------

  def self.transfer_player

    return if $game_player.nil?

    $game_player.moveto(
      @x,
      @y
    )

  end

end

#==============================================================================
# ■ Registro no FO_ENGINE
#==============================================================================

if defined?(FO_ENGINE)

  begin

    FO_ENGINE.register_system(
      FO_Player_Data
    )

  rescue

  end

end

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_Player_Data.load

if defined?(FO_DEBUG)

  FO_DEBUG.log(
    "FO_Player_Data_v4 carregado com sucesso"
  )

end
