#==============================================================================
# ■ FO_TARGET_STATUS
#==============================================================================

module FO_TARGET_STATUS

  #--------------------------------------------------------------------------
  # ● Tempo de combate
  #--------------------------------------------------------------------------

  COMBAT_MEMORY_TIME =
    5 * Graphics.frame_rate

  #--------------------------------------------------------------------------
  # ● Tempo de respawn
  #--------------------------------------------------------------------------

  DEFAULT_RESPAWN_TIME =
    10 * Graphics.frame_rate

end

#==============================================================================
# ■ Game_Event
#==============================================================================

class Game_Event < Game_Character

  #--------------------------------------------------------------------------
  # ● Accessors
  #--------------------------------------------------------------------------

  attr_accessor :fo_dead

  attr_accessor :fo_respawn_timer

  attr_accessor :fo_in_combat

  attr_accessor :fo_combat_time

  attr_accessor :fo_aggressive

  #--------------------------------------------------------------------------
  # ● Alias
  #--------------------------------------------------------------------------

  alias fo_target_status_initialize_v4 initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize(map_id, event)

    fo_target_status_initialize_v4(
      map_id,
      event
    )

    setup_fo_status

  end

  #--------------------------------------------------------------------------
  # ● Setup FO Status
  #--------------------------------------------------------------------------

  def setup_fo_status

    @fo_dead = false

    @fo_respawn_timer = 0

    @fo_in_combat = false

    @fo_combat_time = 0

    @fo_aggressive = false

    setup_aggressive_state

  end

  #--------------------------------------------------------------------------
  # ● Setup Aggressive State
  #--------------------------------------------------------------------------

  def setup_aggressive_state

    return unless fo_combat_entity?

    name =
      @event.name

    #--------------------------------------------------------------------------
    # ● Boss sempre agressivo
    #--------------------------------------------------------------------------

    if name.include?("!")

      @fo_aggressive = true

      return

    end

    #--------------------------------------------------------------------------
    # ● Mob agressivo
    #--------------------------------------------------------------------------

    if name.include?("$")

      @fo_aggressive = true

    end

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  alias fo_target_status_update_v4 update

  def update

    fo_target_status_update_v4

    update_fo_status

  end

  #--------------------------------------------------------------------------
  # ● Update FO Status
  #--------------------------------------------------------------------------

  def update_fo_status

    return unless fo_combat_entity?

    update_combat_timer

    update_respawn

  end

  #--------------------------------------------------------------------------
  # ● Update Combat Timer
  #--------------------------------------------------------------------------

  def update_combat_timer

    return unless @fo_in_combat

    @fo_combat_time -= 1

    if @fo_combat_time <= 0

      @fo_combat_time = 0

      @fo_in_combat = false

    end

  end

  #--------------------------------------------------------------------------
  # ● Update Respawn
  #--------------------------------------------------------------------------

  def update_respawn

    return unless @fo_dead

    return if @fo_respawn_timer <= 0

    @fo_respawn_timer -= 1

    if @fo_respawn_timer <= 0

      fo_respawn

    end

  end

  #--------------------------------------------------------------------------
  # ● Dead?
  #--------------------------------------------------------------------------

  def fo_dead?

    return @fo_dead == true

  end

  #--------------------------------------------------------------------------
  # ● Kill
  #--------------------------------------------------------------------------

  def fo_kill

    return if @fo_dead

    @fo_dead = true

    @fo_in_combat = false

    @fo_combat_time = 0

    @fo_respawn_timer =
      FO_TARGET_STATUS::DEFAULT_RESPAWN_TIME

    @fo_hp = 0 if respond_to?(:fo_hp)

    #--------------------------------------------------------------------------
    # ● Visual
    #--------------------------------------------------------------------------

    self.transparent = true

    self.through = true

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.warning(
        "[ENTITY DEAD] #{@event.name}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Respawn
  #--------------------------------------------------------------------------

  def fo_respawn

    @fo_dead = false

    @fo_in_combat = false

    @fo_combat_time = 0

    @fo_respawn_timer = 0

    #--------------------------------------------------------------------------
    # ● Recover
    #--------------------------------------------------------------------------

    if respond_to?(:fo_full_recover)

      fo_full_recover

    end

    #--------------------------------------------------------------------------
    # ● Visual
    #--------------------------------------------------------------------------

    self.transparent = false

    self.through = false

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[ENTITY RESPAWN] #{@event.name}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Force Respawn
  #--------------------------------------------------------------------------

  def fo_force_respawn

    @fo_respawn_timer = 1

  end

  #--------------------------------------------------------------------------
  # ● Combat State
  #--------------------------------------------------------------------------

  def fo_enter_combat

    @fo_in_combat = true

    @fo_combat_time =
      FO_TARGET_STATUS::COMBAT_MEMORY_TIME

  end

  #--------------------------------------------------------------------------
  # ● Leave Combat
  #--------------------------------------------------------------------------

  def fo_leave_combat

    @fo_in_combat = false

    @fo_combat_time = 0

  end

  #--------------------------------------------------------------------------
  # ● Respawn Remaining
  #--------------------------------------------------------------------------

  def fo_respawn_remaining

    return @fo_respawn_timer

  end

  #--------------------------------------------------------------------------
  # ● Boss?
  #--------------------------------------------------------------------------

  def fo_boss?

    return false if @event.nil?

    return @event.name.include?("!")

  end

  #--------------------------------------------------------------------------
  # ● Mob?
  #--------------------------------------------------------------------------

  def fo_mob?

    return false if @event.nil?

    return @event.name.include?("$")

  end

end

#==============================================================================
# ■ Inicialização
#==============================================================================

if defined?(FO_DEBUG)

  FO_DEBUG.log(
    "FO_Target_Status_v4 carregado com sucesso"
  )

end
