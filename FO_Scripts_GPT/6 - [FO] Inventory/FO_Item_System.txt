module FO_ItemSystem

  #====================================
  # Classe do Item
  #====================================
  class Item
    attr_accessor :id
    attr_accessor :name
    attr_accessor :type
    attr_accessor :rarity
    attr_accessor :attack
    attr_accessor :defense
    attr_accessor :price

    def initialize(id, name, type, rarity, attack, defense, price)
      @id = id
      @name = name
      @type = type
      @rarity = rarity
      @attack = attack
      @defense = defense
      @price = price
    end
  end

  #====================================
  # Banco Global de Itens
  #====================================
  ITEMS = {

    1 => Item.new(
      1,
      "Iron Sword",
      :weapon,
      :common,
      10,
      0,
      100
    ),

    2 => Item.new(
      2,
      "Wood Shield",
      :shield,
      :common,
      0,
      5,
      80
    ),

    3 => Item.new(
      3,
      "Health Potion",
      :consumable,
      :common,
      0,
      0,
      25
    )

  }

  #====================================
  # Buscar item por ID
  #====================================
  def self.get_item(item_id)
    return ITEMS[item_id]
  end

end
