module FO_DROP

  #===========================================================================
  # ■ DROP TABLE
  #===========================================================================
  #
  # MAP_ID => {
  #
  #   EVENT_ID => {
  #
  #     :drops => [
  #
  #       {
  #         :type => :item,
  #         :id => 1,
  #         :chance => 70,
  #         :amount => 1..3,
  #         :rarity => :common
  #       }
  #
  #     ]
  #
  #   }
  #
  # }
  #
  #===========================================================================

  DROP_TABLE = {

    #=========================================================================
    # MAPA 1
    #=========================================================================
    1 => {

      #-----------------------------------------------------------------------
      # EVENTO 5 = $Slime
      #-----------------------------------------------------------------------
      5 => {

        :drops => [

          #---------------------------------------------------------------
          # Potion
          #---------------------------------------------------------------

          {
            :type => :item,
            :id => 1,
            :chance => 80,
            :amount => 1..3,
            :rarity => :common
          },

          #---------------------------------------------------------------
          # Bronze Sword
          #---------------------------------------------------------------

          {
            :type => :weapon,
            :id => 1,
            :chance => 25,
            :amount => 1,
            :rarity => :rare
          },

          #---------------------------------------------------------------
          # Bronze Shield
          #---------------------------------------------------------------

          {

            :id => 1,
            :chance => 10,
            :amount => 1,
            :rarity => :gold
          },

          #---------------------------------------------------------------
          # Gold
          #---------------------------------------------------------------

          {
            :type => :gold,
            :id => 0,
            :chance => 100,
            :amount => 25..75,
            :rarity => :common
          }

        ]

      }

    }

  }

end

#==============================================================================
# ■ FO_DropManager
#==============================================================================

module FO_DropManager

  #===========================================================================
  # ● Generate Drops
  #===========================================================================

  def self.generate_drops(map_id, event_id)

    #--------------------------------------------------------------------------
    # Resultado final
    #--------------------------------------------------------------------------

    results = []

    #--------------------------------------------------------------------------
    # Busca mapa
    #--------------------------------------------------------------------------

    map_data =
      FO_DROP::DROP_TABLE[map_id]

    return results if map_data.nil?

    #--------------------------------------------------------------------------
    # Busca evento
    #--------------------------------------------------------------------------

    event_data =
      map_data[event_id]

    return results if event_data.nil?

    #--------------------------------------------------------------------------
    # Lista de drops
    #--------------------------------------------------------------------------

    drops =
      event_data[:drops]

    return results if drops.nil?

    #--------------------------------------------------------------------------
    # Processa cada drop
    #--------------------------------------------------------------------------

    for drop_data in drops

      #----------------------------------------------------------------------
      # Chance
      #----------------------------------------------------------------------

      roll =
        rand(100) + 1

      chance =
        drop_data[:chance]

      #----------------------------------------------------------------------
      # Falhou
      #----------------------------------------------------------------------

      next if roll > chance

      #----------------------------------------------------------------------
      # Tipo
      #----------------------------------------------------------------------

      type =
        drop_data[:type]

      #----------------------------------------------------------------------
      # ID
      #----------------------------------------------------------------------

      id =
        drop_data[:id]

      #----------------------------------------------------------------------
      # Quantidade
      #----------------------------------------------------------------------

      amount_data =
        drop_data[:amount]

      amount = 1

      # Range
      if amount_data.is_a?(Range)

        amount =
          rand(
            amount_data.end -
            amount_data.begin + 1
          ) + amount_data.begin

      else

        amount =
          amount_data

      end

      #----------------------------------------------------------------------
      # Raridade
      #----------------------------------------------------------------------

      rarity =
        drop_data[:rarity]

      #----------------------------------------------------------------------
      # Adiciona resultado
      #----------------------------------------------------------------------

      results.push({

        :type => type,
        :id => id,
        :amount => amount,
        :rarity => rarity

      })

    end

    return results

  end

  #===========================================================================
  # ● Deliver Drops
  #===========================================================================

  def self.deliver_drops(drop_results)

    return if drop_results.nil?

    #--------------------------------------------------------------------------
    # Processa
    #--------------------------------------------------------------------------

    for drop in drop_results

      #----------------------------------------------------------------------
      # Dados
      #----------------------------------------------------------------------

      type =
        drop[:type]

      id =
        drop[:id]

      amount =
        drop[:amount]

      rarity =
        drop[:rarity]

      #----------------------------------------------------------------------
      # Entrega inventário
      #----------------------------------------------------------------------

      FO_Inventory_Manager.gain_loot(
        type,
        id,
        amount
      )

      #----------------------------------------------------------------------
      # Nome
      #----------------------------------------------------------------------

      name = ""

      if type == :gold

        name =
          "#{amount} Gold"

      else

        name =
          FO_Inventory_Manager.get_name(
            type,
            id
          )

      end

      #----------------------------------------------------------------------
      # Popup
      #----------------------------------------------------------------------

      FO_Loot_Popup_Manager.show_loot(
        name,
        rarity,
        amount
      )

      #----------------------------------------------------------------------
      # Log
      #----------------------------------------------------------------------

      FO_CORE.debug(
        "[DROP] #{name}"
      )

    end

  end

end
