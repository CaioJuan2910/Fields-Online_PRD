#==============================================================================
# ■ FO_Engine
#------------------------------------------------------------------------------
# Classe principal da Engine
#==============================================================================

class FO_Engine

  #===========================================================================
  # ● Initialize
  #===========================================================================
  def initialize

    # Contador de frames
    @frame_count = 0

    # Tempo inicial
    @start_time = Time.now

    # Debug
    FO_CORE.debug("FO_Engine iniciada")

  end

  #===========================================================================
  # ● Update
  #===========================================================================
  def update

    # Tick da engine
    #FO_CORE.debug("Tick da Engine")
    
    # Atualiza contador
    @frame_count += 1

    # Executa atualização por segundo
    update_second if @frame_count >= Graphics.frame_rate

  end

  #===========================================================================
  # ● Update Second
  #===========================================================================
  def update_second

    # Reseta contador
    @frame_count = 0

    # Debug
    FO_CORE.debug("FO_Engine executando update global")

  end

end

#==============================================================================
# ■ Scene_Map
#------------------------------------------------------------------------------
# Integra Engine ao mapa principal
#==============================================================================

class Scene_Map

  #--------------------------------------------------------------------------
  # ● Alias do main
  #--------------------------------------------------------------------------
  alias fo_engine_main main

  #--------------------------------------------------------------------------
  # ● Main
  #--------------------------------------------------------------------------
  def main

    # Cria engine
    @fo_engine = FO_Engine.new

    # Debug
    FO_CORE.debug("FO_Engine conectada ao Scene_Map")

    # Executa main original
    fo_engine_main

  end

  #--------------------------------------------------------------------------
  # ● Alias do update
  #--------------------------------------------------------------------------
  alias fo_engine_update update

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    # Executa update original
    fo_engine_update

    # Atualiza engine
    @fo_engine.update if @fo_engine

  end

end
