#==============================================================================
# ■ FO_TIMER
#==============================================================================

module FO_TIMER

  ENABLE_DEBUG = false

end

#==============================================================================
# ■ FO_Timer_Object
#==============================================================================

class FO_Timer_Object

  #--------------------------------------------------------------------------
  # ● Accessors
  #--------------------------------------------------------------------------

  attr_accessor :name
  attr_accessor :duration
  attr_accessor :remaining
  attr_accessor :looped
  attr_accessor :callback
  attr_accessor :finished

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize(name, duration, looped = false, &block)

    @name = name

    @duration = duration

    @remaining = duration

    @looped = looped

    @callback = block

    @finished = false

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def update

    return if @finished

    @remaining -= 1

    if @remaining <= 0

      execute

    end

  end

  #--------------------------------------------------------------------------
  # ● Execute
  #--------------------------------------------------------------------------

  def execute

    begin

      @callback.call if @callback != nil

    rescue => e

      FO_Core.error(
        "[FO_Timer] " + e.to_s
      )

    end

    if @looped

      reset

    else

      @finished = true

    end

  end

  #--------------------------------------------------------------------------
  # ● Reset
  #--------------------------------------------------------------------------

  def reset

    @remaining = @duration

  end

  #--------------------------------------------------------------------------
  # ● Finished?
  #--------------------------------------------------------------------------

  def finished?

    return @finished

  end

end

#==============================================================================
# ■ FO_Timer
#==============================================================================

module FO_Timer

  #--------------------------------------------------------------------------
  # ● Timers
  #--------------------------------------------------------------------------

  @timers = []

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_timers

    clear_finished

  end

  #--------------------------------------------------------------------------
  # ● Update Timers
  #--------------------------------------------------------------------------

  def self.update_timers

    for timer in @timers

      next if timer == nil

      timer.update

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear Finished
  #--------------------------------------------------------------------------

  def self.clear_finished

    @timers.delete_if do |timer|

      timer.finished?

    end

  end

  #--------------------------------------------------------------------------
  # ● Add Timer
  #--------------------------------------------------------------------------

  def self.add_timer(name, duration, &block)

    return nil if duration <= 0

    timer = FO_Timer_Object.new(
      name,
      duration,
      false,
      &block
    )

    @timers.push(timer)

    debug_add(name, duration)

    return timer

  end

  #--------------------------------------------------------------------------
  # ● Add Loop
  #--------------------------------------------------------------------------

  def self.add_loop(name, duration, &block)

    return nil if duration <= 0

    timer = FO_Timer_Object.new(
      name,
      duration,
      true,
      &block
    )

    @timers.push(timer)

    debug_add(name, duration)

    return timer

  end

  #--------------------------------------------------------------------------
  # ● Remove Timer
  #--------------------------------------------------------------------------

  def self.remove_timer(name)

    @timers.delete_if do |timer|

      timer.name == name

    end

  end

  #--------------------------------------------------------------------------
  # ● Exists?
  #--------------------------------------------------------------------------

  def self.exists?(name)

    for timer in @timers

      next if timer == nil

      return true if timer.name == name

    end

    return false

  end

  #--------------------------------------------------------------------------
  # ● Get Timer
  #--------------------------------------------------------------------------

  def self.get_timer(name)

    for timer in @timers

      next if timer == nil

      return timer if timer.name == name

    end

    return nil

  end

  #--------------------------------------------------------------------------
  # ● Remaining
  #--------------------------------------------------------------------------

  def self.remaining(name)

    timer = get_timer(name)

    return 0 if timer == nil

    return timer.remaining

  end

  #--------------------------------------------------------------------------
  # ● Reset
  #--------------------------------------------------------------------------

  def self.reset(name)

    timer = get_timer(name)

    return if timer == nil

    timer.reset

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    @timers.clear

  end

  #--------------------------------------------------------------------------
  # ● Count
  #--------------------------------------------------------------------------

  def self.count

    return @timers.size

  end

  #--------------------------------------------------------------------------
  # ● Debug Add
  #--------------------------------------------------------------------------

  def self.debug_add(name, duration)

    return unless FO_TIMER::ENABLE_DEBUG

    FO_Core.debug(
      "[FO_Timer] " +
      name.to_s +
      " (" +
      duration.to_s +
      ")"
    )

  end

end
