#==============================================================================
# ■ [FO] Configurações da Engine
#==============================================================================
module FO_ENGINE_CONFIG

  #--------------------------------------------------------------------------
  # FPS alvo
  #--------------------------------------------------------------------------
  TARGET_FPS = 60

  #--------------------------------------------------------------------------
  # Delta time mínimo
  #--------------------------------------------------------------------------
  MIN_DELTA = 0.001

  #--------------------------------------------------------------------------
  # Delta time máximo
  #--------------------------------------------------------------------------
  MAX_DELTA = 0.100

  #--------------------------------------------------------------------------
  # Ativar monitoramento de FPS
  #--------------------------------------------------------------------------
  ENABLE_FPS_MONITOR = true

  #--------------------------------------------------------------------------
  # Ativar logs da Engine
  #--------------------------------------------------------------------------
  ENABLE_ENGINE_LOGS = true

  #--------------------------------------------------------------------------
  # Ativar prioridade de updates
  #--------------------------------------------------------------------------
  ENABLE_UPDATE_PRIORITY = true

  #--------------------------------------------------------------------------
  # Ativar sistema pause
  #--------------------------------------------------------------------------
  ENABLE_PAUSE_SYSTEM = true

  #--------------------------------------------------------------------------
  # Ativar update seguro
  #--------------------------------------------------------------------------
  ENABLE_SAFE_UPDATE = true

  #--------------------------------------------------------------------------
  # Ativar profiler simples
  #--------------------------------------------------------------------------
  ENABLE_UPDATE_PROFILER = true

end

#==============================================================================
# ■ [FO] Engine Principal
#==============================================================================
module FO_ENGINE

  @initialized = false

  @systems = []

  @paused = false

  @last_time = Time.now

  @delta_time = 0.016

  @fps = 60

  @frame_counter = 0

  @fps_timer = Time.now

  @session_started = false

  #--------------------------------------------------------------------------
  # Inicialização
  #--------------------------------------------------------------------------
  def self.initialize

    return if @initialized

    log("========================================")
    log("Inicializando FO_Engine v2.0")
    log("========================================")

    @systems = []

    @paused = false

    @last_time = Time.now

    @delta_time = 0.016

    @fps = 60

    @frame_counter = 0

    @fps_timer = Time.now

    @session_started = true

    @initialized = true

    validate_engine

    log("FO_Engine inicializada com sucesso.")

  end

  #--------------------------------------------------------------------------
  # Validação
  #--------------------------------------------------------------------------
  def self.validate_engine

    log("Validando Engine...")

    if FO_ENGINE_CONFIG::TARGET_FPS <= 0
      FO_DEBUG.warning("TARGET_FPS inválido.")
    end

  end

  #--------------------------------------------------------------------------
  # Atualização principal
  #--------------------------------------------------------------------------
  def self.update

    return unless @initialized

    update_delta_time

    update_fps

    return if paused?

    update_systems

    late_update_systems

    render_systems

  end

  #--------------------------------------------------------------------------
  # Atualiza delta time
  #--------------------------------------------------------------------------
  def self.update_delta_time

    current_time = Time.now

    delta = current_time - @last_time

    delta = FO_UTILS.clamp(
      delta,
      FO_ENGINE_CONFIG::MIN_DELTA,
      FO_ENGINE_CONFIG::MAX_DELTA
    )

    @delta_time = delta

    @last_time = current_time

  end

  #--------------------------------------------------------------------------
  # Atualiza FPS
  #--------------------------------------------------------------------------
  def self.update_fps

    return unless FO_ENGINE_CONFIG::ENABLE_FPS_MONITOR

    @frame_counter += 1

    elapsed = Time.now - @fps_timer

    if elapsed >= 1.0

      @fps = @frame_counter

      @frame_counter = 0

      @fps_timer = Time.now

    end

  end

  #--------------------------------------------------------------------------
  # Atualiza sistemas
  #--------------------------------------------------------------------------
  def self.update_systems

    sorted_systems.each do |system|

      next if system.nil?
      next unless system.enabled

      if FO_ENGINE_CONFIG::ENABLE_SAFE_UPDATE

        FO_FAIL_SAFE.safe_execute(
          "Update #{system.system_name}"
        ) do

          profile_update(system.system_name) do
            system.update(@delta_time)
          end

        end

      else

        system.update(@delta_time)

      end

    end

  end

  #--------------------------------------------------------------------------
  # Late update
  #--------------------------------------------------------------------------
  def self.late_update_systems

    sorted_systems.each do |system|

      next if system.nil?
      next unless system.enabled

      FO_FAIL_SAFE.safe_execute(
        "LateUpdate #{system.system_name}"
      ) do

        system.late_update(@delta_time)

      end

    end

  end

  #--------------------------------------------------------------------------
  # Render
  #--------------------------------------------------------------------------
  def self.render_systems

    sorted_systems.each do |system|

      next if system.nil?
      next unless system.enabled

      FO_FAIL_SAFE.safe_execute(
        "Render #{system.system_name}"
      ) do

        system.render(@delta_time)

      end

    end

  end

  #--------------------------------------------------------------------------
  # Sistemas ordenados
  #--------------------------------------------------------------------------
  def self.sorted_systems

    return @systems unless FO_ENGINE_CONFIG::ENABLE_UPDATE_PRIORITY

    return @systems.sort_by { |system| system.priority }

  end

  #--------------------------------------------------------------------------
  # Registra sistema
  #--------------------------------------------------------------------------
  def self.register_system(system)

    return if system.nil?

    @systems << system

    log("Sistema registrado: #{system.system_name}")

  end

  #--------------------------------------------------------------------------
  # Remove sistema
  #--------------------------------------------------------------------------
  def self.unregister_system(system)

    return if system.nil?

    @systems.delete(system)

    log("Sistema removido: #{system.system_name}")

  end

  #--------------------------------------------------------------------------
  # Pausa engine
  #--------------------------------------------------------------------------
  def self.pause

    return unless FO_ENGINE_CONFIG::ENABLE_PAUSE_SYSTEM

    return if @paused

    @paused = true

    @systems.each do |system|

      next if system.nil?

      FO_FAIL_SAFE.safe_execute(
        "Pause #{system.system_name}"
      ) do

        system.pause

      end

    end

    log("Engine pausada.")

  end

  #--------------------------------------------------------------------------
  # Resume engine
  #--------------------------------------------------------------------------
  def self.resume

    return unless @paused

    @paused = false

    @systems.each do |system|

      next if system.nil?

      FO_FAIL_SAFE.safe_execute(
        "Resume #{system.system_name}"
      ) do

        system.resume

      end

    end

    log("Engine resumida.")

  end

  #--------------------------------------------------------------------------
  # Pausada?
  #--------------------------------------------------------------------------
  def self.paused?

    return @paused

  end

  #--------------------------------------------------------------------------
  # Delta time
  #--------------------------------------------------------------------------
  def self.delta_time

    return @delta_time

  end

  #--------------------------------------------------------------------------
  # FPS atual
  #--------------------------------------------------------------------------
  def self.fps

    return @fps

  end

  #--------------------------------------------------------------------------
  # Sistemas registrados
  #--------------------------------------------------------------------------
  def self.systems

    return @systems

  end

  #--------------------------------------------------------------------------
  # Quantidade de sistemas
  #--------------------------------------------------------------------------
  def self.system_count

    return @systems.size

  end

  #--------------------------------------------------------------------------
  # Sessão iniciada?
  #--------------------------------------------------------------------------
  def self.session_started?

    return @session_started

  end

  #--------------------------------------------------------------------------
  # Engine inicializada?
  #--------------------------------------------------------------------------
  def self.initialized?

    return @initialized

  end

  #--------------------------------------------------------------------------
  # Profile update
  #--------------------------------------------------------------------------
  def self.profile_update(system_name)

    return yield unless FO_ENGINE_CONFIG::ENABLE_UPDATE_PROFILER

    start_time = Time.now

    yield

    elapsed = ((Time.now - start_time) * 1000.0).to_i

    if elapsed >= 5
      FO_DEBUG.warning(
        "#{system_name} consumiu #{elapsed}ms no update."
      )
    end

  end

  #--------------------------------------------------------------------------
  # Log da engine
  #--------------------------------------------------------------------------
  def self.log(message)

    return unless FO_ENGINE_CONFIG::ENABLE_ENGINE_LOGS

    FO_LOG.write("[FO_ENGINE] #{message}")

  end

end

class Scene_Map

  alias fo_engine_main_update update

  def update

    FO_FAIL_SAFE.safe_execute("FO_ENGINE Scene_Map Update") do

      FO_ENGINE.update

    end

    fo_engine_main_update

  end

end
