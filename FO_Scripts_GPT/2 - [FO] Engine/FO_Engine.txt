#==============================================================================
# ■ FO_ENGINE
#==============================================================================

module FO_ENGINE

  #--------------------------------------------------------------------------
  # ■ Sistemas
  #--------------------------------------------------------------------------

  @systems = []

  #--------------------------------------------------------------------------
  # ■ Estado
  #--------------------------------------------------------------------------

  @started = false

  #--------------------------------------------------------------------------
  # ● Start
  #--------------------------------------------------------------------------

  def self.start

    return if @started

    @started = true

    if defined?(FO_DEBUG)

      FO_DEBUG.warning(
        "[FO_ENGINE STARTED]"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Started?
  #--------------------------------------------------------------------------

  def self.started?

    return @started == true

  end

  #--------------------------------------------------------------------------
  # ● Register System
  #--------------------------------------------------------------------------

  def self.register_system(system)

    return if system.nil?

    return if @systems.include?(system)

    @systems.push(system)

    debug_register(system)

  end

  #--------------------------------------------------------------------------
  # ● Remove System
  #--------------------------------------------------------------------------

  def self.remove_system(system)

    return if system.nil?

    @systems.delete(system)

  end

  #--------------------------------------------------------------------------
  # ● Systems
  #--------------------------------------------------------------------------

  def self.systems

    return @systems

  end

  #--------------------------------------------------------------------------
  # ● Count
  #--------------------------------------------------------------------------

  def self.count

    return @systems.size

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    @systems.clear

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    return unless started?

    for system in @systems

      next if system.nil?

      begin

        if system.respond_to?(:update)

          system.update

        end

      rescue => e

        handle_system_error(
          system,
          e
        )

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Handle System Error
  #--------------------------------------------------------------------------

  def self.handle_system_error(system, error)

    return unless defined?(FO_DEBUG)

    FO_DEBUG.error(
      "[FO_ENGINE ERROR] #{system} => #{error}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Debug Register
  #--------------------------------------------------------------------------

  def self.debug_register(system)

    return unless defined?(FO_DEBUG)

    return unless FO_ENGINE::ENABLE_DEBUG

    FO_DEBUG.log(
      "[SYSTEM REGISTERED] #{system}"
    )

  end

end
