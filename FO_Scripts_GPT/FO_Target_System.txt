module FO_TARGET

  #--------------------------------------------------------------------------
  # ● Distância máxima
  #--------------------------------------------------------------------------

  MAX_DISTANCE = 5

  #--------------------------------------------------------------------------
  # ● Tecla de Target
  #--------------------------------------------------------------------------
  # Input::L = tecla Q
  #--------------------------------------------------------------------------

  TARGET_KEY = Input::L

end

#==============================================================================
# ■ Variáveis Globais
#==============================================================================

$fo_target      = nil
$fo_target_name = "Nenhum"
$fo_target_type = nil

#==============================================================================
# ■ FO_Target_System
#==============================================================================

class FO_Target_System

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    update_input

    update_target_validity

  end

  #--------------------------------------------------------------------------
  # ● Input
  #--------------------------------------------------------------------------
  def self.update_input

    if Input.trigger?(FO_TARGET::TARGET_KEY)

      select_target

    end

  end

  #--------------------------------------------------------------------------
  # ● Selecionar Target
  #--------------------------------------------------------------------------
  def self.select_target

    nearest_event = nil

    nearest_distance = 9999

    # Percorre eventos
    for event in $game_map.events.values

      next if event.nil?

      # Segurança
      event_data = event.instance_variable_get(:@event)

      next if event_data.nil?

      # Nome
      event_name = event_data.name

      next if event_name.nil?
      next if event_name == ""

      # Distância
      distance =
        ($game_player.x - event.x).abs +
        ($game_player.y - event.y).abs

      next if distance > FO_TARGET::MAX_DISTANCE

      # Mais próximo
      if distance < nearest_distance

        nearest_distance = distance

        nearest_event = event

      end

    end

    # Define target
    $fo_target = nearest_event

    # Atualiza dados
    update_target_data

  end

  #--------------------------------------------------------------------------
  # ● Atualizar dados
  #--------------------------------------------------------------------------
  def self.update_target_data

    # Limpa
    $fo_target_name = "Nenhum"
    $fo_target_type = nil

    return if $fo_target.nil?

    # Dados
    event_data = $fo_target.instance_variable_get(:@event)

    return if event_data.nil?

    # Nome original
    original_name = event_data.name

    return if original_name.nil?

    # Primeiro símbolo
    first_char = original_name[0,1]

    # Nome limpo
    clean_name = original_name[1, original_name.size]

    # Tipo
    case first_char

    when "@"
      $fo_target_type = :npc

    when "$"
      $fo_target_type = :mob

    when "!"
      $fo_target_type = :boss

    when "?"
      $fo_target_type = :quest

    else
      $fo_target_type = :unknown

    end

    # Nome final
    $fo_target_name = clean_name

  end

  #--------------------------------------------------------------------------
  # ● Validar target
  #--------------------------------------------------------------------------
  def self.update_target_validity

    return if $fo_target.nil?

    # Distância
    distance =
      ($game_player.x - $fo_target.x).abs +
      ($game_player.y - $fo_target.y).abs

    # Muito longe
    if distance > FO_TARGET::MAX_DISTANCE

      clear_target

    end

  end

  #--------------------------------------------------------------------------
  # ● Limpar target
  #--------------------------------------------------------------------------
  def self.clear_target

    $fo_target      = nil
    $fo_target_name = "Nenhum"
    $fo_target_type = nil

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_target_update_v25 update

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    fo_target_update_v25

    FO_Target_System.update

  end

end
