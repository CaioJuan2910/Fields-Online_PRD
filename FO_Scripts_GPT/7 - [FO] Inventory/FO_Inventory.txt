# ■ FO_INVENTORY
#==============================================================================

module FO_INVENTORY

  #============================================================================
  # ■ CONFIGURAÇÕES
  #============================================================================

  MAX_SLOTS = 60

  MAX_STACK = 999

  INITIAL_GOLD = 0

end

#==============================================================================
# ■ FO_Inventory
#==============================================================================

module FO_Inventory

  #--------------------------------------------------------------------------
  # ■ Estrutura
  #--------------------------------------------------------------------------

  @items = {}
  @weapons = {}
  @armors = {}

  @gold = FO_INVENTORY::INITIAL_GOLD

  #--------------------------------------------------------------------------
  # ● Setup
  #--------------------------------------------------------------------------

  def self.setup

    @items ||= {}
    @weapons ||= {}
    @armors ||= {}

    @gold ||= 0

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    @items.clear
    @weapons.clear
    @armors.clear

    @gold = 0

  end

  #--------------------------------------------------------------------------
  # ● Items
  #--------------------------------------------------------------------------

  def self.items

    return @items

  end

  #--------------------------------------------------------------------------
  # ● Weapons
  #--------------------------------------------------------------------------

  def self.weapons

    return @weapons

  end

  #--------------------------------------------------------------------------
  # ● Armors
  #--------------------------------------------------------------------------

  def self.armors

    return @armors

  end

  #--------------------------------------------------------------------------
  # ● Gold
  #--------------------------------------------------------------------------

  def self.gold

    return @gold

  end

  #--------------------------------------------------------------------------
  # ● Gain Gold
  #--------------------------------------------------------------------------

  def self.gain_gold(amount)

    return if amount <= 0

    @gold += amount

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[GAIN GOLD] #{amount}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Lose Gold
  #--------------------------------------------------------------------------

  def self.lose_gold(amount)

    return if amount <= 0

    @gold -= amount

    if @gold < 0

      @gold = 0

    end

  end

  #--------------------------------------------------------------------------
  # ● Gain Item
  #--------------------------------------------------------------------------

  def self.gain_item(id, amount = 1)

    setup

    return if id.nil?
    return if amount <= 0

    @items[id] ||= 0

    @items[id] += amount

    if @items[id] > FO_INVENTORY::MAX_STACK

      @items[id] =
        FO_INVENTORY::MAX_STACK

    end

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[GAIN ITEM] ID #{id} x#{amount}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Gain Weapon
  #--------------------------------------------------------------------------

  def self.gain_weapon(id, amount = 1)

    setup

    return if id.nil?
    return if amount <= 0

    @weapons[id] ||= 0

    @weapons[id] += amount

    if @weapons[id] > FO_INVENTORY::MAX_STACK

      @weapons[id] =
        FO_INVENTORY::MAX_STACK

    end

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[GAIN WEAPON] ID #{id} x#{amount}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Gain Armor
  #--------------------------------------------------------------------------

  def self.gain_armor(id, amount = 1)

    setup

    return if id.nil?
    return if amount <= 0

    @armors[id] ||= 0

    @armors[id] += amount

    if @armors[id] > FO_INVENTORY::MAX_STACK

      @armors[id] =
        FO_INVENTORY::MAX_STACK

    end

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[GAIN ARMOR] ID #{id} x#{amount}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Lose Item
  #--------------------------------------------------------------------------

  def self.lose_item(id, amount = 1)

    return if id.nil?
    return if amount <= 0

    return unless @items.has_key?(id)

    @items[id] -= amount

    if @items[id] <= 0

      @items.delete(id)

    end

  end

  #--------------------------------------------------------------------------
  # ● Lose Weapon
  #--------------------------------------------------------------------------

  def self.lose_weapon(id, amount = 1)

    return if id.nil?
    return if amount <= 0

    return unless @weapons.has_key?(id)

    @weapons[id] -= amount

    if @weapons[id] <= 0

      @weapons.delete(id)

    end

  end

  #--------------------------------------------------------------------------
  # ● Lose Armor
  #--------------------------------------------------------------------------

  def self.lose_armor(id, amount = 1)

    return if id.nil?
    return if amount <= 0

    return unless @armors.has_key?(id)

    @armors[id] -= amount

    if @armors[id] <= 0

      @armors.delete(id)

    end

  end

  #--------------------------------------------------------------------------
  # ● Item Amount
  #--------------------------------------------------------------------------

  def self.item_amount(id)

    return 0 unless @items.has_key?(id)

    return @items[id]

  end

  #--------------------------------------------------------------------------
  # ● Weapon Amount
  #--------------------------------------------------------------------------

  def self.weapon_amount(id)

    return 0 unless @weapons.has_key?(id)

    return @weapons[id]

  end

  #--------------------------------------------------------------------------
  # ● Armor Amount
  #--------------------------------------------------------------------------

  def self.armor_amount(id)

    return 0 unless @armors.has_key?(id)

    return @armors[id]

  end

  #--------------------------------------------------------------------------
  # ● Has Item?
  #--------------------------------------------------------------------------

  def self.has_item?(id)

    return item_amount(id) > 0

  end

  #--------------------------------------------------------------------------
  # ● Has Weapon?
  #--------------------------------------------------------------------------

  def self.has_weapon?(id)

    return weapon_amount(id) > 0

  end

  #--------------------------------------------------------------------------
  # ● Has Armor?
  #--------------------------------------------------------------------------

  def self.has_armor?(id)

    return armor_amount(id) > 0

  end

  #--------------------------------------------------------------------------
  # ● Slot Count
  #--------------------------------------------------------------------------

  def self.slot_count

    total = 0

    total += @items.size
    total += @weapons.size
    total += @armors.size

    return total

  end

  #--------------------------------------------------------------------------
  # ● Inventory Full?
  #--------------------------------------------------------------------------

  def self.full?

    return slot_count >=
           FO_INVENTORY::MAX_SLOTS

  end

  #--------------------------------------------------------------------------
  # ● Save Data
  #--------------------------------------------------------------------------

  def self.save_data

    return {

      :items   => @items,
      :weapons => @weapons,
      :armors  => @armors,
      :gold    => @gold

    }

  end

  #--------------------------------------------------------------------------
  # ● Load Data
  #--------------------------------------------------------------------------

  def self.load_data(data)

    return if data.nil?

    @items =
      data[:items] || {}

    @weapons =
      data[:weapons] || {}

    @armors =
      data[:armors] || {}

    @gold =
      data[:gold] || 0

  end

end

#==============================================================================
# ■ FO_Inventory_Manager
#==============================================================================

module FO_Inventory_Manager

  #--------------------------------------------------------------------------
  # ● Gain Loot
  #--------------------------------------------------------------------------

  def self.gain_loot(type, id, amount)

    case type

    when :item

      FO_Inventory.gain_item(
        id,
        amount
      )

    when :weapon

      FO_Inventory.gain_weapon(
        id,
        amount
      )

    when :armor

      FO_Inventory.gain_armor(
        id,
        amount
      )

    when :gold

      FO_Inventory.gain_gold(
        amount
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Get Name
  #--------------------------------------------------------------------------

  def self.get_name(type, id)

    case type

    when :item

      item = $data_items[id]

      return "Unknown Item" if item.nil?

      return item.name

    when :weapon

      weapon = $data_weapons[id]

      return "Unknown Weapon" if weapon.nil?

      return weapon.name

    when :armor

      armor = $data_armors[id]

      return "Unknown Armor" if armor.nil?

      return armor.name

    end

    return "Unknown"

  end

end

FO_Inventory.setup
