#==============================================================================
# ■ FO_Inventory_Manager
#==============================================================================

module FO_Inventory_Manager

  #--------------------------------------------------------------------------
  # ● Party
  #--------------------------------------------------------------------------
  def self.party

    return FO_Data_Manager.party

  end

  #--------------------------------------------------------------------------
  # ● Gain Loot
  #--------------------------------------------------------------------------
  def self.gain_loot(type, id, amount = 1)

    return if amount <= 0

    case type

    when :item

      gain_item(
        id,
        amount
      )

    when :weapon

      gain_weapon(
        id,
        amount
      )

    when :armor

      gain_armor(
        id,
        amount
      )

    when :gold

      gain_gold(
        amount
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Gain Item
  #--------------------------------------------------------------------------
  def self.gain_item(item_id, amount)

    #--------------------------------------------------------------------------
    # Item
    #--------------------------------------------------------------------------

    item =
      FO_Data_Manager.item(item_id)

    return if item.nil?

    #--------------------------------------------------------------------------
    # Party
    #--------------------------------------------------------------------------

    return if party.nil?

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    party.gain_item(
      item_id,
      amount
    )

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[ITEM GAIN] #{item.name} x#{amount}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Gain Weapon
  #--------------------------------------------------------------------------
  def self.gain_weapon(weapon_id, amount)

    #--------------------------------------------------------------------------
    # Weapon
    #--------------------------------------------------------------------------

    weapon =
      FO_Data_Manager.weapon(
        weapon_id
      )

    return if weapon.nil?

    #--------------------------------------------------------------------------
    # Party
    #--------------------------------------------------------------------------

    return if party.nil?

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    party.gain_weapon(
      weapon_id,
      amount
    )

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[WEAPON GAIN] #{weapon.name} x#{amount}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Gain Armor
  #--------------------------------------------------------------------------
  def self.gain_armor(armor_id, amount)

    #--------------------------------------------------------------------------
    # Armor
    #--------------------------------------------------------------------------

    armor =
      FO_Data_Manager.armor(
        armor_id
      )

    return if armor.nil?

    #--------------------------------------------------------------------------
    # Party
    #--------------------------------------------------------------------------

    return if party.nil?

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    party.gain_armor(
      armor_id,
      amount
    )

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[ARMOR GAIN] #{armor.name} x#{amount}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Gain Gold
  #--------------------------------------------------------------------------
  def self.gain_gold(amount)

    return if amount <= 0

    return if party.nil?

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    party.gain_gold(amount)

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[GOLD GAIN] #{amount}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Get Object
  #--------------------------------------------------------------------------
  def self.get_object(type, id)

    case type

    when :item

      return FO_Data_Manager.item(id)

    when :weapon

      return FO_Data_Manager.weapon(id)

    when :armor

      return FO_Data_Manager.armor(id)

    end

    return nil

  end

  #--------------------------------------------------------------------------
  # ● Get Name
  #--------------------------------------------------------------------------
  def self.get_name(type, id)

    object =
      get_object(
        type,
        id
      )

    return "Unknown" if object.nil?

    return object.name

  end

end
