#==============================================================================
# ■ Game_Event
#==============================================================================

class Game_Event < Game_Character

  #--------------------------------------------------------------------------
  # Alias initialize
  #--------------------------------------------------------------------------

  alias fo_enemy_initialize initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize(map_id, event)

    #--------------------------------------------------------------------------
    # Original
    #--------------------------------------------------------------------------

    fo_enemy_initialize(map_id, event)

    #--------------------------------------------------------------------------
    # Enemy setup
    #--------------------------------------------------------------------------

    setup_fo_enemy

  end

  #--------------------------------------------------------------------------
  # ● Setup Enemy
  #--------------------------------------------------------------------------

  def setup_fo_enemy

    #--------------------------------------------------------------------------
    # Default
    #--------------------------------------------------------------------------

    @fo_enemy_id = nil

    @fo_enemy_data = nil

    #--------------------------------------------------------------------------
    # Busca enemy_id
    #--------------------------------------------------------------------------

    enemy_id =
      find_fo_enemy_id

    return if enemy_id.nil?

    #--------------------------------------------------------------------------
    # Database
    #--------------------------------------------------------------------------

    enemy =
      $data_enemies[enemy_id]

    return if enemy.nil?

    #--------------------------------------------------------------------------
    # Define enemy
    #--------------------------------------------------------------------------

    @fo_enemy_id =
      enemy_id

    @fo_enemy_data =
      enemy

    #--------------------------------------------------------------------------
    # Stats MMORPG
    #--------------------------------------------------------------------------

    @fo_hp =
      enemy.maxhp

    @fo_maxhp =
      enemy.maxhp

    @fo_sp =
      enemy.maxsp

    @fo_maxsp =
      enemy.maxsp

    @fo_str =
      enemy.str

    @fo_dex =
      enemy.dex

    @fo_agi =
      enemy.agi

    @fo_int =
      enemy.int

    @fo_atk =
      enemy.atk

    @fo_pdef =
      enemy.pdef

    @fo_mdef =
      enemy.mdef

    @fo_eva =
      enemy.eva

    @fo_exp =
      enemy.exp

    @fo_gold =
      enemy.gold

    #--------------------------------------------------------------------------
    # Status
    #--------------------------------------------------------------------------

    @fo_dead =
      false

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[ENEMY LOAD] #{@event.name} => #{enemy.name}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Buscar enemy_id
  #--------------------------------------------------------------------------

  def find_fo_enemy_id

    return nil if @list.nil?

    for command in @list

      next if command.nil?

      #----------------------------------------------------------------------
      # Comentário
      #----------------------------------------------------------------------

      if command.code == 108

        text =
          command.parameters[0]

        #--------------------------------------------------------------------
        # <enemy_id:X>
        #--------------------------------------------------------------------

        if text =~ /<enemy_id:(\d+)>/

          return $1.to_i

        end

      end

    end

    return nil

  end

  #--------------------------------------------------------------------------
  # ● Enemy?
  #--------------------------------------------------------------------------

  def fo_enemy?

    return !@fo_enemy_id.nil?

  end

  #--------------------------------------------------------------------------
  # ● Enemy ID
  #--------------------------------------------------------------------------

  def fo_enemy_id

    return @fo_enemy_id

  end

  #--------------------------------------------------------------------------
  # ● Enemy Data
  #--------------------------------------------------------------------------

  def fo_enemy_data

    return @fo_enemy_data

  end

  #--------------------------------------------------------------------------
  # ● Dead?
  #--------------------------------------------------------------------------

  def fo_dead?

    return @fo_dead

  end

  #--------------------------------------------------------------------------
  # ● Kill
  #--------------------------------------------------------------------------

  def fo_kill

    @fo_dead = true

  end

  #--------------------------------------------------------------------------
  # ● Combat Entity?
  #--------------------------------------------------------------------------

  def fo_combat_entity?

    return fo_enemy?

  end

  #--------------------------------------------------------------------------
  # ● HP
  #--------------------------------------------------------------------------

  attr_accessor :fo_hp

  #--------------------------------------------------------------------------
  # ● MaxHP
  #--------------------------------------------------------------------------

  attr_accessor :fo_maxhp

  #--------------------------------------------------------------------------
  # ● SP
  #--------------------------------------------------------------------------

  attr_accessor :fo_sp

  #--------------------------------------------------------------------------
  # ● MaxSP
  #--------------------------------------------------------------------------

  attr_accessor :fo_maxsp

  #--------------------------------------------------------------------------
  # ● STR
  #--------------------------------------------------------------------------

  attr_accessor :fo_str

  #--------------------------------------------------------------------------
  # ● DEX
  #--------------------------------------------------------------------------

  attr_accessor :fo_dex

  #--------------------------------------------------------------------------
  # ● AGI
  #--------------------------------------------------------------------------

  attr_accessor :fo_agi

  #--------------------------------------------------------------------------
  # ● INT
  #--------------------------------------------------------------------------

  attr_accessor :fo_int

  #--------------------------------------------------------------------------
  # ● ATK
  #--------------------------------------------------------------------------

  attr_accessor :fo_atk

  #--------------------------------------------------------------------------
  # ● PDEF
  #--------------------------------------------------------------------------

  attr_accessor :fo_pdef

  #--------------------------------------------------------------------------
  # ● MDEF
  #--------------------------------------------------------------------------

  attr_accessor :fo_mdef

  #--------------------------------------------------------------------------
  # ● EVA
  #--------------------------------------------------------------------------

  attr_accessor :fo_eva

  #--------------------------------------------------------------------------
  # ● EXP
  #--------------------------------------------------------------------------

  attr_accessor :fo_exp

  #--------------------------------------------------------------------------
  # ● GOLD
  #--------------------------------------------------------------------------

  attr_accessor :fo_gold

end
