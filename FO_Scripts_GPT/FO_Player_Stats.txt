#==============================================================================
# ■ FO_Player_Stats
#------------------------------------------------------------------------------
# Classe principal de atributos do jogador
#==============================================================================

class FO_Player_Stats

  #===========================================================================
  # ■ ATTR ACCESSOR
  #===========================================================================

  attr_accessor :strength
  attr_accessor :agility
  attr_accessor :intelligence
  attr_accessor :vitality
  attr_accessor :defense

  #===========================================================================
  # ● Initialize
  #===========================================================================
  def initialize

    # Força
    @strength = 10

    # Agilidade
    @agility = 8

    # Inteligência
    @intelligence = 6

    # Vitalidade
    @vitality = 12

    # Defesa
    @defense = 5

    # Debug
    FO_CORE.debug("FO_Player_Stats iniciado")

    # Mostrar atributos
    show_stats

  end

  #===========================================================================
  # ● Mostrar atributos
  #===========================================================================
  def show_stats

    FO_CORE.debug("==============================")
    FO_CORE.debug("PLAYER STATS")
    FO_CORE.debug("Strength....: #{@strength}")
    FO_CORE.debug("Agility.....: #{@agility}")
    FO_CORE.debug("Intelligence: #{@intelligence}")
    FO_CORE.debug("Vitality....: #{@vitality}")
    FO_CORE.debug("Defense.....: #{@defense}")
    FO_CORE.debug("==============================")

  end

  #===========================================================================
  # ● HP Máximo
  #===========================================================================
  def max_hp

    return @vitality * 10

  end

  #===========================================================================
  # ● MP Máximo
  #===========================================================================
  def max_mp

    return @intelligence * 5

  end

  #===========================================================================
  # ● Ataque Físico
  #===========================================================================
  def attack_power

    return @strength * 2

  end

  #===========================================================================
  # ● Defesa Total
  #===========================================================================
  def total_defense

    return @defense * 2

  end

  #===========================================================================
  # ● Mostrar cálculos
  #===========================================================================
  def show_calculations

    FO_CORE.debug("===== RPG FORMULAS =====")
    FO_CORE.debug("HP Máximo...: #{max_hp}")
    FO_CORE.debug("MP Máximo...: #{max_mp}")
    FO_CORE.debug("Ataque......: #{attack_power}")
    FO_CORE.debug("Defesa Total: #{total_defense}")
    FO_CORE.debug("========================")

  end

end

#==============================================================================
# ■ FO_Engine
#------------------------------------------------------------------------------
# Integra Stats à Engine
#==============================================================================

class FO_Engine

  #--------------------------------------------------------------------------
  # ● Alias Initialize
  #--------------------------------------------------------------------------
  alias fo_stats_initialize initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize

    # Executa original
    fo_stats_initialize

    # Cria stats
    @player_stats = FO_Player_Stats.new

    # Mostrar cálculos
    @player_stats.show_calculations

  end

end
