#==============================================================================
# ■ Game_Actor
#==============================================================================

class Game_Actor < Game_Battler

  #--------------------------------------------------------------------------
  # Alias Setup
  #--------------------------------------------------------------------------

  alias fo_player_stats_setup setup

  #--------------------------------------------------------------------------
  # ● Setup
  #--------------------------------------------------------------------------

  def setup(actor_id)

    #--------------------------------------------------------------------------
    # Original
    #--------------------------------------------------------------------------

    fo_player_stats_setup(actor_id)

    #--------------------------------------------------------------------------
    # MMORPG Stats
    #--------------------------------------------------------------------------

    setup_fo_stats

  end

  #--------------------------------------------------------------------------
  # ● Setup FO Stats
  #--------------------------------------------------------------------------

  def setup_fo_stats

    #--------------------------------------------------------------------------
    # Base Stats
    #--------------------------------------------------------------------------

    @fo_str = 10
    @fo_dex = 10
    @fo_agi = 10
    @fo_int = 10

    #--------------------------------------------------------------------------
    # Combat Stats
    #--------------------------------------------------------------------------

    @fo_attack = 5

    @fo_pdef = 2

    @fo_mdef = 2

    @fo_accuracy = 95

    @fo_critical = 5

    @fo_dodge = 3

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[PLAYER STATS] Inicializados"
    )

  end

  #--------------------------------------------------------------------------
  # ● STR
  #--------------------------------------------------------------------------

  def fo_str

    return @fo_str

  end

  #--------------------------------------------------------------------------
  # ● DEX
  #--------------------------------------------------------------------------

  def fo_dex

    return @fo_dex

  end

  #--------------------------------------------------------------------------
  # ● AGI
  #--------------------------------------------------------------------------

  def fo_agi

    return @fo_agi

  end

  #--------------------------------------------------------------------------
  # ● INT
  #--------------------------------------------------------------------------

  def fo_int

    return @fo_int

  end

  #--------------------------------------------------------------------------
  # ● ATTACK
  #--------------------------------------------------------------------------

  def fo_attack

    total =
      @fo_attack

    #--------------------------------------------------------------------------
    # Weapon bonus
    #--------------------------------------------------------------------------

    weapon =
      weapons[0]

    if weapon != nil

      total += weapon.atk

    end

    return total

  end

  #--------------------------------------------------------------------------
  # ● PDEF
  #--------------------------------------------------------------------------

  def fo_pdef

    total =
      @fo_pdef

    for armor in armors

      next if armor.nil?

      total += armor.pdef

    end

    return total

  end

  #--------------------------------------------------------------------------
  # ● MDEF
  #--------------------------------------------------------------------------

  def fo_mdef

    total =
      @fo_mdef

    for armor in armors

      next if armor.nil?

      total += armor.mdef

    end

    return total

  end

  #--------------------------------------------------------------------------
  # ● Accuracy
  #--------------------------------------------------------------------------

  def fo_accuracy

    return @fo_accuracy

  end

  #--------------------------------------------------------------------------
  # ● Critical
  #--------------------------------------------------------------------------

  def fo_critical

    return @fo_critical

  end

  #--------------------------------------------------------------------------
  # ● Dodge
  #--------------------------------------------------------------------------

  def fo_dodge

    return @fo_dodge

  end

end
