module FO_Player_Combat

  #===========================================================================
  # ■ CONFIGURAÇÕES
  #===========================================================================

  ATTACK_KEY =
    Input::CTRL

  BASE_DAMAGE =
    25

end

#==============================================================================
# ■ FO_Player_Combat_System
#==============================================================================

class FO_Player_Combat_System

  #===========================================================================
  # ● Update
  #===========================================================================

  def self.update

    update_attack_input

  end

  #===========================================================================
  # ● Input
  #===========================================================================

  def self.update_attack_input

    if Input.trigger?(
      FO_Player_Combat::ATTACK_KEY
    )

      attack_target

    end

  end

  #===========================================================================
  # ● Attack Target
  #===========================================================================

  def self.attack_target

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if $fo_target.nil?

    return unless $fo_target.respond_to?(
      :fo_combat_entity?
    )

    return unless $fo_target.fo_combat_entity?

    return if $fo_target.fo_dead?

    return if $fo_target.fo_hp <= 0

    #--------------------------------------------------------------------------
    # Sistema MMORPG
    #--------------------------------------------------------------------------

    roll =
      rand(100)

    damage = 0

    damage_type =
      :normal

    text = ""

    #--------------------------------------------------------------------------
    # MISS
    #--------------------------------------------------------------------------

    if roll < 10

      text =
        "MISS"

      damage_type =
        :miss

    #--------------------------------------------------------------------------
    # DANO 10%
    #--------------------------------------------------------------------------

    elsif roll < 25

      damage =
        (
          FO_Player_Combat::BASE_DAMAGE *
          0.10
        ).to_i

    #--------------------------------------------------------------------------
    # DANO 50%
    #--------------------------------------------------------------------------

    elsif roll < 60

      damage =
        (
          FO_Player_Combat::BASE_DAMAGE *
          0.50
        ).to_i

    #--------------------------------------------------------------------------
    # DANO 80%
    #--------------------------------------------------------------------------

    elsif roll < 85

      damage =
        (
          FO_Player_Combat::BASE_DAMAGE *
          0.80
        ).to_i

    #--------------------------------------------------------------------------
    # DANO 100%
    #--------------------------------------------------------------------------

    elsif roll < 95

      damage =
        FO_Player_Combat::BASE_DAMAGE

    #--------------------------------------------------------------------------
    # CRITICAL
    #--------------------------------------------------------------------------

    else

      critical_rate =
        rand(81) + 120

      damage =
        (
          FO_Player_Combat::BASE_DAMAGE *
          critical_rate / 100.0
        ).to_i

      damage_type =
        :critical

      text =
        "CRITICAL"

    end

    #--------------------------------------------------------------------------
    # Texto
    #--------------------------------------------------------------------------

    if damage > 0 &&
       text == ""

      text =
        damage.to_s

    end

    #--------------------------------------------------------------------------
    # Aplica dano
    #--------------------------------------------------------------------------

    if damage > 0

      $fo_target.fo_in_combat =
        true

      $fo_target.instance_variable_set(
        :@fo_failed_moves,
        0
      )

      $fo_target.fo_hp -=
        damage

      #--------------------------------------------------------------------------
      # Segurança HP
      #--------------------------------------------------------------------------

      if $fo_target.fo_hp < 0

        $fo_target.fo_hp = 0

      end

      #--------------------------------------------------------------------------
      # MORTE
      #--------------------------------------------------------------------------

      if $fo_target.fo_hp <= 0

        dead_target =
          $fo_target

        #----------------------------------------------------------------------
        # DROP SYSTEM
        #----------------------------------------------------------------------

        begin

          #--------------------------------------------------------------------
          # MAP ID
          #--------------------------------------------------------------------

          map_id =
            $game_map.map_id

          #--------------------------------------------------------------------
          # EVENT ID
          #--------------------------------------------------------------------

          event_id =
            dead_target.id

          #--------------------------------------------------------------------
          # Gera drops
          #--------------------------------------------------------------------

          drops =
            FO_DropManager.generate_drops(
              map_id,
              event_id
            )

          #--------------------------------------------------------------------
          # Entrega drops
          #--------------------------------------------------------------------

          FO_DropManager.deliver_drops(
            drops
          )

        rescue => e

          FO_CORE.debug(
            "[DROP ERROR] #{e}"
          )

        end

        #----------------------------------------------------------------------
        # Floating Damage Final
        #----------------------------------------------------------------------

        FO_Floating_Damage.create(
          dead_target,
          text,
          damage_type
        )

        #----------------------------------------------------------------------
# Kill
#----------------------------------------------------------------------

begin

  #--------------------------------------------------------------------------
  # Mata entidade
  #--------------------------------------------------------------------------

  dead_target.fo_kill

  #--------------------------------------------------------------------------
  # Remove visual imediatamente
  #--------------------------------------------------------------------------

  dead_target.transparent = true

  #--------------------------------------------------------------------------
  # Desativa movimento
  #--------------------------------------------------------------------------

  dead_target.through = true

  #--------------------------------------------------------------------------
  # Apaga evento do mapa
  #--------------------------------------------------------------------------

  dead_target.erase

rescue => e

  FO_CORE.debug(
    "[KILL ERROR] #{e}"
  )

end

return

      end

    end

    #--------------------------------------------------------------------------
    # Floating Damage
    #--------------------------------------------------------------------------

    FO_Floating_Damage.create(
      $fo_target,
      text,
      damage_type
    )

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_player_combat_update_v6 update

  #===========================================================================
  # ● Update
  #===========================================================================

  def update

    fo_player_combat_update_v6

    FO_Player_Combat_System.update

  end

end
