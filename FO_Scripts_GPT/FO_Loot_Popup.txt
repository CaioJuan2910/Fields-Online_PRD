#==============================================================================
# ■ FO_Loot_Popup
#==============================================================================

class FO_Loot_Popup

  #===========================================================================
  # ● Initialize
  #===========================================================================

  def initialize(text, color)

    @finished = false

    create_sprite(text, color)

  end

  #===========================================================================
  # ● Criar sprite
  #===========================================================================

  def create_sprite(text, color)

    @sprite = Sprite.new

    @sprite.bitmap = Bitmap.new(300, 32)

    @sprite.z = 99999

    #--------------------------------------------------------------------------
    # Fonte
    #--------------------------------------------------------------------------

    font = @sprite.bitmap.font

    font.name = "Arial"
    font.size = 20
    font.bold = true
    font.color = color

    #--------------------------------------------------------------------------
    # Texto
    #--------------------------------------------------------------------------

    @sprite.bitmap.draw_text(
      0,
      0,
      300,
      32,
      text,
      1
    )

    #--------------------------------------------------------------------------
    # Posição MMORPG
    #--------------------------------------------------------------------------

    @sprite.x = 170
    @sprite.y = 260

    #--------------------------------------------------------------------------
    # Opacidade
    #--------------------------------------------------------------------------

    @sprite.opacity = 255

    #--------------------------------------------------------------------------
    # Movimento
    #--------------------------------------------------------------------------

    @speed_y = 1

  end

  #===========================================================================
  # ● Update
  #===========================================================================

  def update

    return if @finished

    #--------------------------------------------------------------------------
    # Movimento
    #--------------------------------------------------------------------------

    @sprite.y -= @speed_y

    #--------------------------------------------------------------------------
    # Fade
    #--------------------------------------------------------------------------

    @sprite.opacity -= 4

    #--------------------------------------------------------------------------
    # Finaliza
    #--------------------------------------------------------------------------

    if @sprite.opacity <= 0

      dispose

      @finished = true

    end

  end

  #===========================================================================
  # ● Finished?
  #===========================================================================

  def finished?

    return @finished

  end

  #===========================================================================
  # ● Dispose
  #===========================================================================

  def dispose

    if @sprite != nil

      @sprite.bitmap.dispose
      @sprite.dispose

    end

  end

end

#==============================================================================
# ■ FO_Loot_Popup_Manager
#==============================================================================

module FO_Loot_Popup_Manager

  @popups = []

  #===========================================================================
  # ● Show Loot
  #===========================================================================

  def self.show_loot(name, rarity, amount = 1)

    #--------------------------------------------------------------------------
    # Texto
    #--------------------------------------------------------------------------

    text = ""

    if amount > 1

      text =
        "#{name} x#{amount}"

    else

      text =
        name

    end

    #--------------------------------------------------------------------------
    # Raridade
    #--------------------------------------------------------------------------

    color = Color.new(255,255,255)

    case rarity

    when :common
      color = Color.new(255,255,255)

    when :rare
      color = Color.new(80,180,255)

    when :gold
      color = Color.new(255,220,60)

    when :legendary
      color = Color.new(255,80,255)

    end

    #--------------------------------------------------------------------------
    # Prefixo
    #--------------------------------------------------------------------------

    prefix = "+ "

    case rarity

    when :rare
      prefix = "RARE + "

    when :gold
      prefix = "GOLD + "

    when :legendary
      prefix = "LEGENDARY + "

    end

    #--------------------------------------------------------------------------
    # Popup
    #--------------------------------------------------------------------------

    popup =
      FO_Loot_Popup.new(
        prefix + text,
        color
      )

    #--------------------------------------------------------------------------
    # Offset vertical
    #--------------------------------------------------------------------------

    popup.instance_variable_get(:@sprite).y -= (
      @popups.size * 28
    )

    @popups.push(popup)

  end

  #===========================================================================
  # ● Update
  #===========================================================================

  def self.update

    #--------------------------------------------------------------------------
    # Atualiza
    #--------------------------------------------------------------------------

    for popup in @popups

      popup.update

    end

    #--------------------------------------------------------------------------
    # Remove finalizados
    #--------------------------------------------------------------------------

    @popups.delete_if do |popup|

      popup.finished?

    end

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_loot_popup_update_v2 update

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    fo_loot_popup_update_v2

    FO_Loot_Popup_Manager.update

  end

end
