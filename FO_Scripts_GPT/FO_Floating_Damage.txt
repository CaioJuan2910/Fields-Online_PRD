class FO_Floating_Damage

  @@damages = []

  #--------------------------------------------------------------------------
  # ● Create dano entidade
  #--------------------------------------------------------------------------
  def self.create(target, text, type = :normal)

    return if target.nil?

    sprite = Sprite.new

    sprite.bitmap = Bitmap.new(120, 32)

    sprite.bitmap.font.name = "Arial"
    sprite.bitmap.font.size = 22
    sprite.bitmap.font.bold = true

    #--------------------------------------------------------------------------
    # ● Cor
    #--------------------------------------------------------------------------

    case type

    when :miss

      sprite.bitmap.font.color =
        Color.new(80, 180, 255)

    when :critical

      sprite.bitmap.font.color =
        Color.new(255, 60, 60)

    else

      sprite.bitmap.font.color =
        Color.new(255, 255, 255)

    end

    #--------------------------------------------------------------------------
    # ● Texto
    #--------------------------------------------------------------------------

    sprite.bitmap.draw_text(
      0,
      0,
      120,
      32,
      text,
      1
    )

    #--------------------------------------------------------------------------
    # ● Posição
    #--------------------------------------------------------------------------

    sprite.x =
      target.screen_x - 60

    sprite.y =
      target.screen_y - 64

    sprite.z = 9999

    #--------------------------------------------------------------------------
    # ● Dados
    #--------------------------------------------------------------------------

    data = {
      :sprite => sprite,
      :time => 40
    }

    @@damages.push(data)

  end

  #--------------------------------------------------------------------------
  # ● Create dano player
  #--------------------------------------------------------------------------
  def self.create_player(text, type = :normal)

    return if text.nil?

    sprite = Sprite.new

    sprite.bitmap = Bitmap.new(120, 32)

    sprite.bitmap.font.name = "Arial"
    sprite.bitmap.font.size = 22
    sprite.bitmap.font.bold = true

    #--------------------------------------------------------------------------
    # ● Cor
    #--------------------------------------------------------------------------

    case type

    when :miss

      sprite.bitmap.font.color =
        Color.new(80, 180, 255)

    when :critical

      sprite.bitmap.font.color =
        Color.new(255, 60, 60)

    else

      sprite.bitmap.font.color =
        Color.new(255, 255, 255)

    end

    #--------------------------------------------------------------------------
    # ● Texto
    #--------------------------------------------------------------------------

    sprite.bitmap.draw_text(
      0,
      0,
      120,
      32,
      text,
      1
    )

    #--------------------------------------------------------------------------
    # ● Posição player
    #--------------------------------------------------------------------------

    sprite.x =
      $game_player.screen_x - 60

    sprite.y =
      $game_player.screen_y - 64

    sprite.z = 9999

    #--------------------------------------------------------------------------
    # ● Dados
    #--------------------------------------------------------------------------

    data = {
      :sprite => sprite,
      :time => 40
    }

    @@damages.push(data)

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    for damage in @@damages

      sprite = damage[:sprite]

      next if sprite.nil?

      damage[:time] -= 1

      sprite.y -= 1

      sprite.opacity -= 6

    end

    #--------------------------------------------------------------------------
    # ● Remove finalizados
    #--------------------------------------------------------------------------

    @@damages.delete_if do |damage|

      if damage[:time] <= 0

        damage[:sprite].dispose

        true

      else

        false

      end

    end

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_floating_damage_update_v20 update

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    fo_floating_damage_update_v20

    FO_Floating_Damage.update

  end

end
