#==============================================================================
# ■ FO_TARGET_NAMETAG
#==============================================================================

module FO_TARGET_NAMETAG

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  FONT_NAME = "Arial"

  FONT_SIZE = 18

  FONT_BOLD = true

  FONT_OUTLINE = true

  OFFSET_Y = -48

  Z = 99999

  #--------------------------------------------------------------------------
  # ● Cores
  #--------------------------------------------------------------------------

  MOB_COLOR =
    Color.new(255,255,255)

  BOSS_COLOR =
    Color.new(255,120,120)

end

#==============================================================================
# ■ FO_Target_NameTag
#==============================================================================

module FO_Target_NameTag

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @sprite = nil

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_visibility

    update_position

    update_text

  end

  #--------------------------------------------------------------------------
  # ● Create Sprite
  #--------------------------------------------------------------------------

  def self.create_sprite

    dispose

    @sprite =
      Sprite.new

    @sprite.bitmap =
      Bitmap.new(220, 32)

    @sprite.z =
      FO_TARGET_NAMETAG::Z

  end

  #--------------------------------------------------------------------------
  # ● Update Visibility
  #--------------------------------------------------------------------------

  def self.update_visibility

    if FO_Target_System.target.nil?

      dispose

      return

    end

    if @sprite.nil?

      create_sprite

    end

  end

  #--------------------------------------------------------------------------
  # ● Update Position
  #--------------------------------------------------------------------------

  def self.update_position

    return if @sprite.nil?

    target =
      FO_Target_System.target

    return if target.nil?

    @sprite.x =
      target.screen_x - 110

    @sprite.y =
      target.screen_y +
      FO_TARGET_NAMETAG::OFFSET_Y

  end

  #--------------------------------------------------------------------------
  # ● Update Text
  #--------------------------------------------------------------------------

  def self.update_text

    return if @sprite.nil?

    target =
      FO_Target_System.target

    return if target.nil?

    bitmap =
      @sprite.bitmap

    bitmap.clear

    #--------------------------------------------------------------------------
    # ● Fonte
    #--------------------------------------------------------------------------

    font =
      bitmap.font

    font.name =
      FO_TARGET_NAMETAG::FONT_NAME

    font.size =
      FO_TARGET_NAMETAG::FONT_SIZE

    font.bold =
      FO_TARGET_NAMETAG::FONT_BOLD

    #--------------------------------------------------------------------------
    # ● Cor
    #--------------------------------------------------------------------------

    if FO_Target_System.target_boss?

      font.color =
        FO_TARGET_NAMETAG::BOSS_COLOR

    else

      font.color =
        FO_TARGET_NAMETAG::MOB_COLOR

    end

    #--------------------------------------------------------------------------
    # ● Nome
    #--------------------------------------------------------------------------

    name =
      clean_name(
        FO_Target_System.target_name
      )

    bitmap.draw_text(
      0,
      0,
      220,
      32,
      name,
      1
    )

  end

  #--------------------------------------------------------------------------
  # ● Clean Name
  #--------------------------------------------------------------------------

  def self.clean_name(name)

    return "" if name.nil?

    text = name.clone

    #--------------------------------------------------------------------------
    # ● Remove tags
    #--------------------------------------------------------------------------

    text.gsub!("$", "")
    text.gsub!("!", "")
    text.gsub!("@", "")
    text.gsub!("?", "")

    #--------------------------------------------------------------------------
    # ● Segurança
    #--------------------------------------------------------------------------

    if text.size <= 0

      return "Unknown"

    end

    return text

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def self.dispose

    return if @sprite.nil?

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    @sprite.dispose

    @sprite = nil

  end

end

#==============================================================================

  begin

    FO_ENGINE.register_system(
      FO_Target_NameTag
    )

  rescue

  end

end
