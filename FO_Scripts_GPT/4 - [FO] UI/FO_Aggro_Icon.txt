#==============================================================================
# ■ FO_AGGRO_ICON
#==============================================================================

module FO_AGGRO_ICON

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  ICON_SIZE = 20

  OFFSET_X = 6

  OFFSET_Y = -54

  Z = 99998

  #--------------------------------------------------------------------------
  # ● Cores
  #--------------------------------------------------------------------------

  AGGRO_COLOR =
    Color.new(255,60,60)

  COMBAT_COLOR =
    Color.new(255,180,60)

end

#==============================================================================
# ■ FO_Aggro_Icon_Object
#==============================================================================

class FO_Aggro_Icon_Object

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize(target)

    @target = target

    create_sprite

  end

  #--------------------------------------------------------------------------
  # ● Create Sprite
  #--------------------------------------------------------------------------

  def create_sprite

    @sprite =
      Sprite.new

    @sprite.bitmap =
      Bitmap.new(
        FO_AGGRO_ICON::ICON_SIZE,
        FO_AGGRO_ICON::ICON_SIZE
      )

    @sprite.z =
      FO_AGGRO_ICON::Z

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def update

    return if @target.nil?

    return dispose if invalid_target?

    update_position

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Invalid Target?
  #--------------------------------------------------------------------------

  def invalid_target?

    return true if @target.nil?

    return true if @target.fo_dead?

    return true if @target.transparent

    return false

  end

  #--------------------------------------------------------------------------
  # ● Update Position
  #--------------------------------------------------------------------------

  def update_position

    @sprite.x =
      @target.screen_x +
      FO_AGGRO_ICON::OFFSET_X

    @sprite.y =
      @target.screen_y +
      FO_AGGRO_ICON::OFFSET_Y

  end

  #--------------------------------------------------------------------------
  # ● Refresh
  #--------------------------------------------------------------------------

  def refresh

    bitmap =
      @sprite.bitmap

    bitmap.clear

    #--------------------------------------------------------------------------
    # ● Cor
    #--------------------------------------------------------------------------

    color =
      FO_AGGRO_ICON::AGGRO_COLOR

    if @target.fo_in_combat

      color =
        FO_AGGRO_ICON::COMBAT_COLOR

    end

    #--------------------------------------------------------------------------
    # ● Desenha círculo
    #--------------------------------------------------------------------------

    draw_circle(
      bitmap,
      10,
      10,
      8,
      color
    )

  end

  #--------------------------------------------------------------------------
  # ● Draw Circle
  #--------------------------------------------------------------------------

  def draw_circle(bitmap, cx, cy, radius, color)

    for x in 0...bitmap.width

      for y in 0...bitmap.height

        dx = x - cx
        dy = y - cy

        distance =
          Math.sqrt(
            dx * dx + dy * dy
          )

        if distance <= radius

          bitmap.set_pixel(
            x,
            y,
            color
          )

        end

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def dispose

    return if @sprite.nil?

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    @sprite.dispose

    @sprite = nil

  end

end

#==============================================================================
# ■ FO_Aggro_Icon
#==============================================================================

module FO_Aggro_Icon

  #--------------------------------------------------------------------------
  # ■ Objetos
  #--------------------------------------------------------------------------

  @objects = []

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_targets

    update_objects

  end

  #--------------------------------------------------------------------------
  # ● Update Targets
  #--------------------------------------------------------------------------

  def self.update_targets

    return if $game_map.nil?

    for event in $game_map.events.values

      next if event.nil?

      next unless valid_target?(event)

      unless already_exists?(event)

        object =
          FO_Aggro_Icon_Object.new(
            event
          )

        @objects.push(object)

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Valid Target?
  #--------------------------------------------------------------------------

  def self.valid_target?(event)

    return false unless event.respond_to?(
      :fo_combat_entity?
    )

    return false unless event.fo_combat_entity?

    return false if event.fo_dead?

    return false unless event.fo_aggressive

    return true

  end

  #--------------------------------------------------------------------------
  # ● Already Exists?
  #--------------------------------------------------------------------------

  def self.already_exists?(target)

    for object in @objects

      next if object.nil?

      current =
        object.instance_variable_get(
          :@target
        )

      return true if current == target

    end

    return false

  end

  #--------------------------------------------------------------------------
  # ● Update Objects
  #--------------------------------------------------------------------------

  def self.update_objects

    for object in @objects

      next if object.nil?

      object.update

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    for object in @objects

      next if object.nil?

      object.dispose

    end

    @objects.clear

  end

end
