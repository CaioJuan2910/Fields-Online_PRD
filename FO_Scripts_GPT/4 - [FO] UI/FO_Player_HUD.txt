#==============================================================================
# ■ FO_PLAYER_HUD
#==============================================================================

module FO_PLAYER_HUD

  #--------------------------------------------------------------------------
  # ● Posição
  #--------------------------------------------------------------------------

  X = 12
  Y = 12

  WIDTH = 260
  HEIGHT = 110

  Z = 99999

  #--------------------------------------------------------------------------
  # ● Barras
  #--------------------------------------------------------------------------

  BAR_WIDTH = 180
  BAR_HEIGHT = 14

  HP_BAR_Y = 34
  MP_BAR_Y = 58
  EXP_BAR_Y = 82

  #--------------------------------------------------------------------------
  # ● Fonte
  #--------------------------------------------------------------------------

  FONT_NAME = "Arial"

  FONT_SIZE = 16

  #--------------------------------------------------------------------------
  # ● Cores
  #--------------------------------------------------------------------------

  BACKGROUND_COLOR =
    Color.new(0,0,0,160)

  BORDER_COLOR =
    Color.new(255,255,255)

  HP_COLOR =
    Color.new(220,60,60)

  MP_COLOR =
    Color.new(80,120,255)

  EXP_COLOR =
    Color.new(255,220,80)

  TEXT_COLOR =
    Color.new(255,255,255)

end

#==============================================================================
# ■ FO_Player_HUD
#==============================================================================

module FO_Player_HUD

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @sprite = nil

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    create_sprite if @sprite.nil?

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Create Sprite
  #--------------------------------------------------------------------------

  def self.create_sprite

    dispose

    @sprite =
      Sprite.new

    @sprite.bitmap =
      Bitmap.new(
        FO_PLAYER_HUD::WIDTH,
        FO_PLAYER_HUD::HEIGHT
      )

    @sprite.x =
      FO_PLAYER_HUD::X

    @sprite.y =
      FO_PLAYER_HUD::Y

    @sprite.z =
      FO_PLAYER_HUD::Z

  end

  #--------------------------------------------------------------------------
  # ● Refresh
  #--------------------------------------------------------------------------

  def self.refresh

    return if @sprite.nil?

    bitmap =
      @sprite.bitmap

    bitmap.clear

    draw_background(bitmap)

    draw_hFO_Core.logbitmap)

    draw_mFO_Core.logbitmap)

    draw_exFO_Core.logbitmap)

    draw_info(bitmap)

  end

  #--------------------------------------------------------------------------
  # ● Draw Background
  #--------------------------------------------------------------------------

  def self.draw_background(bitmap)

    bitmap.fill_rect(
      0,
      0,
      FO_PLAYER_HUD::WIDTH,
      FO_PLAYER_HUD::HEIGHT,
      FO_PLAYER_HUD::BACKGROUND_COLOR
    )

    draw_border(bitmap)

  end

  #--------------------------------------------------------------------------
  # ● Draw Border
  #--------------------------------------------------------------------------

  def self.draw_border(bitmap)

    color =
      FO_PLAYER_HUD::BORDER_COLOR

    width =
      FO_PLAYER_HUD::WIDTH

    height =
      FO_PLAYER_HUD::HEIGHT

    # Top
    bitmap.fill_rect(
      0,
      0,
      width,
      1,
      color
    )

    # Bottom
    bitmap.fill_rect(
      0,
      height - 1,
      width,
      1,
      color
    )

    # Left
    bitmap.fill_rect(
      0,
      0,
      1,
      height,
      color
    )

    # Right
    bitmap.fill_rect(
      width - 1,
      0,
      1,
      height,
      color
    )

  end

  #--------------------------------------------------------------------------
  # ● Draw HP
  #--------------------------------------------------------------------------

  def self.draw_hFO_Core.logbitmap)

    return unless defined?(FO_Player_Data)

    hp =
      FO_Player_Data.hp

    maxhp =
      FO_Player_Data.maxhp

    return if maxhp <= 0

    percent =
      hp.to_f / maxhp.to_f

    width =
      (
        FO_PLAYER_HUD::BAR_WIDTH *
        percent
      ).to_i

    # Fundo
    bitmap.fill_rect(
      70,
      FO_PLAYER_HUD::HP_BAR_Y,
      FO_PLAYER_HUD::BAR_WIDTH,
      FO_PLAYER_HUD::BAR_HEIGHT,
      Color.new(40,40,40)
    )

    # Barra
    bitmap.fill_rect(
      70,
      FO_PLAYER_HUD::HP_BAR_Y,
      width,
      FO_PLAYER_HUD::BAR_HEIGHT,
      FO_PLAYER_HUD::HP_COLOR
    )

    draw_text(
      bitmap,
      "HP",
      10,
      FO_PLAYER_HUD::HP_BAR_Y - 2
    )

    draw_text(
      bitmap,
      "#{hp}/#{maxhp}",
      120,
      FO_PLAYER_HUD::HP_BAR_Y - 2
    )

  end

  #--------------------------------------------------------------------------
  # ● Draw MP
  #--------------------------------------------------------------------------

  def self.draw_mFO_Core.logbitmap)

    return unless defined?(FO_Player_Data)

    mp =
      FO_Player_Data.mp

    maxmp =
      FO_Player_Data.maxmp

    return if maxmp <= 0

    percent =
      mp.to_f / maxmp.to_f

    width =
      (
        FO_PLAYER_HUD::BAR_WIDTH *
        percent
      ).to_i

    # Fundo
    bitmap.fill_rect(
      70,
      FO_PLAYER_HUD::MP_BAR_Y,
      FO_PLAYER_HUD::BAR_WIDTH,
      FO_PLAYER_HUD::BAR_HEIGHT,
      Color.new(40,40,40)
    )

    # Barra
    bitmap.fill_rect(
      70,
      FO_PLAYER_HUD::MP_BAR_Y,
      width,
      FO_PLAYER_HUD::BAR_HEIGHT,
      FO_PLAYER_HUD::MP_COLOR
    )

    draw_text(
      bitmap,
      "MP",
      10,
      FO_PLAYER_HUD::MP_BAR_Y - 2
    )

    draw_text(
      bitmap,
      "#{mp}/#{maxmp}",
      120,
      FO_PLAYER_HUD::MP_BAR_Y - 2
    )

  end

  #--------------------------------------------------------------------------
  # ● Draw EXP
  #--------------------------------------------------------------------------

  def self.draw_exFO_Core.logbitmap)

    return unless defined?(FO_Player_Data)

    exp =
      FO_Player_Data.exp

    next_exp =
      FO_Player_Data.next_exp

    return if next_exp <= 0

    percent =
      exp.to_f / next_exp.to_f

    width =
      (
        FO_PLAYER_HUD::BAR_WIDTH *
        percent
      ).to_i

    # Fundo
    bitmap.fill_rect(
      70,
      FO_PLAYER_HUD::EXP_BAR_Y,
      FO_PLAYER_HUD::BAR_WIDTH,
      FO_PLAYER_HUD::BAR_HEIGHT,
      Color.new(40,40,40)
    )

    # Barra
    bitmap.fill_rect(
      70,
      FO_PLAYER_HUD::EXP_BAR_Y,
      width,
      FO_PLAYER_HUD::BAR_HEIGHT,
      FO_PLAYER_HUD::EXP_COLOR
    )

    draw_text(
      bitmap,
      "EXP",
      10,
      FO_PLAYER_HUD::EXP_BAR_Y - 2
    )

    draw_text(
      bitmap,
      "#{exp}/#{next_exp}",
      120,
      FO_PLAYER_HUD::EXP_BAR_Y - 2
    )

  end

  #--------------------------------------------------------------------------
  # ● Draw Info
  #--------------------------------------------------------------------------

  def self.draw_info(bitmap)

    return unless defined?(FO_Player_Data)

    level =
      FO_Player_Data.level

    gold = 0

    if defined?(FO_Inventory)

      gold =
        FO_Inventory.gold

    end

    draw_text(
      bitmap,
      "LV #{level}",
      10,
      6
    )

    draw_text(
      bitmap,
      "#{gold} Gold",
      150,
      6
    )

  end

  #--------------------------------------------------------------------------
  # ● Draw Text
  #--------------------------------------------------------------------------

  def self.draw_text(bitmap, text, x, y)

    font =
      bitmap.font

    font.name =
      FO_PLAYER_HUD::FONT_NAME

    font.size =
      FO_PLAYER_HUD::FONT_SIZE

    font.bold = true

    font.color =
      FO_PLAYER_HUD::TEXT_COLOR

    bitmap.draw_text(
      x,
      y,
      220,
      24,
      text
    )

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def self.dispose

    return if @sprite.nil?

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    @sprite.dispose

    @sprite = nil

  end

end

  begin

    FO_ENGINE.register_system(

    )

  rescue

  end
