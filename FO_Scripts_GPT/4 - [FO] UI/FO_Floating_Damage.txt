#==============================================================================
# ■ FO_FLOATING_DAMAGE
#==============================================================================

module FO_FLOATING_DAMAGE

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  FONT_NAME = "Arial"

  FONT_SIZE = 22

  FONT_BOLD = true

  START_OFFSET_Y = -48

  FLOAT_SPEED = 1

  FADE_SPEED = 5

  Z = 99999

  #--------------------------------------------------------------------------
  # ● Cores
  #--------------------------------------------------------------------------

  NORMAL_COLOR =
    Color.new(255,255,255)

  CRITICAL_COLOR =
    Color.new(255,80,80)

  MISS_COLOR =
    Color.new(180,180,180)

  PLAYER_DAMAGE_COLOR =
    Color.new(255,120,120)

end

#==============================================================================
# ■ FO_Floating_Damage_Object
#==============================================================================

class FO_Floating_Damage_Object

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize(x, y, text, type)

    @finished = false

    create_sprite(
      x,
      y,
      text,
      type
    )

  end

  #--------------------------------------------------------------------------
  # ● Create Sprite
  #--------------------------------------------------------------------------

  def create_sprite(x, y, text, type)

    @sprite =
      Sprite.new

    @sprite.bitmap =
      Bitmap.new(160, 32)

    @sprite.z =
      FO_FLOATING_DAMAGE::Z

    #--------------------------------------------------------------------------
    # ● Fonte
    #--------------------------------------------------------------------------

    font =
      @sprite.bitmap.font

    font.name =
      FO_FLOATING_DAMAGE::FONT_NAME

    font.size =
      FO_FLOATING_DAMAGE::FONT_SIZE

    font.bold =
      FO_FLOATING_DAMAGE::FONT_BOLD

    #--------------------------------------------------------------------------
    # ● Cor
    #--------------------------------------------------------------------------

    case type

    when :critical

      font.color =
        FO_FLOATING_DAMAGE::CRITICAL_COLOR

    when :miss

      font.color =
        FO_FLOATING_DAMAGE::MISS_COLOR

    when :player_damage

      font.color =
        FO_FLOATING_DAMAGE::PLAYER_DAMAGE_COLOR

    else

      font.color =
        FO_FLOATING_DAMAGE::NORMAL_COLOR

    end

    #--------------------------------------------------------------------------
    # ● Texto
    #--------------------------------------------------------------------------

    @sprite.bitmap.draw_text(
      0,
      0,
      160,
      32,
      text,
      1
    )

    #--------------------------------------------------------------------------
    # ● Posição
    #--------------------------------------------------------------------------

    @sprite.x =
      x - 80

    @sprite.y =
      y +
      FO_FLOATING_DAMAGE::START_OFFSET_Y

    @sprite.opacity = 255

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def update

    return if @finished

    return if @sprite.nil?

    #--------------------------------------------------------------------------
    # ● Movimento
    #--------------------------------------------------------------------------

    @sprite.y -=
      FO_FLOATING_DAMAGE::FLOAT_SPEED

    #--------------------------------------------------------------------------
    # ● Fade
    #--------------------------------------------------------------------------

    @sprite.opacity -=
      FO_FLOATING_DAMAGE::FADE_SPEED

    #--------------------------------------------------------------------------
    # ● Finaliza
    #--------------------------------------------------------------------------

    if @sprite.opacity <= 0

      dispose

      @finished = true

    end

  end

  #--------------------------------------------------------------------------
  # ● Finished?
  #--------------------------------------------------------------------------

  def finished?

    return @finished

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def dispose

    return if @sprite.nil?

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    @sprite.dispose

  end

end

#==============================================================================
# ■ FO_Floating_Damage
#==============================================================================

module FO_Floating_Damage

  #--------------------------------------------------------------------------
  # ■ Objetos
  #--------------------------------------------------------------------------

  @objects = []

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    for object in @objects

      next if object.nil?

      object.update

    end

    @objects.delete_if do |object|

      object.finished?

    end

  end

  #--------------------------------------------------------------------------
  # ● Create
  #--------------------------------------------------------------------------

  def self.create(target, text, type = :normal)

    return if target.nil?

    object =
      FO_Floating_Damage_Object.new(
        target.screen_x,
        target.screen_y,
        text,
        type
      )

    @objects.push(object)

  end

  #--------------------------------------------------------------------------
  # ● Create Player
  #--------------------------------------------------------------------------

  def self.create_player(text, type = :player_damage)

    player =
      $game_player

    return if player.nil?

    object =
      FO_Floating_Damage_Object.new(
        player.screen_x,
        player.screen_y,
        text,
        type
      )

    @objects.push(object)

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    for object in @objects

      next if object.nil?

      object.dispose

    end

    @objects.clear

  end

end
