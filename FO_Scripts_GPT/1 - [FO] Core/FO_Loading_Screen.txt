#==============================================================================
# ■ FO_Loading_Screen
#------------------------------------------------------------------------------
# Fields Online MMORPG
# Sistema profissional de Loading Screen
#------------------------------------------------------------------------------
# Autor: ChatGPT + Caio
# Versão: 2.0
# Data: 17/05/2026
# Compatibilidade: RPG Maker XP
#==============================================================================

module FO_Loading_Screen

  #===========================================================================
  # ■ CONFIGURAÇÕES
  #===========================================================================

  ENABLE_BACKGROUND = true

  BACKGROUND_COLOR = Color.new(0, 0, 0)

  BAR_WIDTH  = 400
  BAR_HEIGHT = 24

  BAR_COLOR_BG = Color.new(40, 40, 40)
  BAR_COLOR    = Color.new(0, 180, 255)

  TEXT_COLOR = Color.new(255, 255, 255)

  FONT_NAME = "Arial"
  FONT_SIZE = 22

  SUB_FONT_SIZE = 18

  FADE_SPEED = 15

  #===========================================================================
  # ■ Variáveis internas
  #===========================================================================

  @viewport = nil
  @background = nil
  @bar_bg = nil
  @bar_fill = nil
  @text_sprite = nil
  @sub_text_sprite = nil

  @progress = 0
  @visible = false

  #===========================================================================
  # ■ show
  #===========================================================================

  def self.show(title = "Loading...")

    return if @visible

    @visible = true

    create_viewport
    create_background
    create_bar
    create_text(title)

    Graphics.update

    fade_in

    FO_Core.log("FO_Loading_Screen iniciado.")
  end

  #===========================================================================
  # ■ update_progress
  #===========================================================================

  def self.update_progress(value, message = "")

    return unless @visible

    @progress = [[value, 0].max, 100].min

    update_bar
    update_message(message)

    Graphics.update
  end

  #===========================================================================
  # ■ finish
  #===========================================================================

  def self.finish

    return unless @visible

    update_progress(100, "Concluído")

    fade_out

    dispose_all

    @visible = false

    FO_Core.log("FO_Loading_Screen finalizado.")
  end

  #===========================================================================
  # ■ create_viewport
  #===========================================================================

  def self.create_viewport

    @viewport = Viewport.new(0, 0, 640, 480)
    @viewport.z = 99999
  end

  #===========================================================================
  # ■ create_background
  #===========================================================================

  def self.create_background

    @background = Sprite.new(@viewport)
    @background.bitmap = Bitmap.new(640, 480)

    @background.bitmap.fill_rect(
      0,
      0,
      640,
      480,
      BACKGROUND_COLOR
    )

    @background.opacity = 0
  end

  #===========================================================================
  # ■ create_bar
  #===========================================================================

  def self.create_bar

    center_x = (640 - BAR_WIDTH) / 2
    center_y = 300

    #-----------------------------------------------------------------------
    # Fundo da barra
    #-----------------------------------------------------------------------

    @bar_bg = Sprite.new(@viewport)
    @bar_bg.bitmap = Bitmap.new(BAR_WIDTH, BAR_HEIGHT)

    @bar_bg.bitmap.fill_rect(
      0,
      0,
      BAR_WIDTH,
      BAR_HEIGHT,
      BAR_COLOR_BG
    )

    @bar_bg.x = center_x
    @bar_bg.y = center_y
    @bar_bg.opacity = 0

    #-----------------------------------------------------------------------
    # Barra preenchimento
    #-----------------------------------------------------------------------

    @bar_fill = Sprite.new(@viewport)
    @bar_fill.bitmap = Bitmap.new(BAR_WIDTH, BAR_HEIGHT)

    @bar_fill.x = center_x
    @bar_fill.y = center_y
    @bar_fill.opacity = 0

    refresh_fill_bar
  end

  #===========================================================================
  # ■ create_text
  #===========================================================================

  def self.create_text(title)

    #-----------------------------------------------------------------------
    # Texto principal
    #-----------------------------------------------------------------------

    @text_sprite = Sprite.new(@viewport)
    @text_sprite.bitmap = Bitmap.new(640, 64)

    @text_sprite.bitmap.font.name = FONT_NAME
    @text_sprite.bitmap.font.size = FONT_SIZE
    @text_sprite.bitmap.font.color = TEXT_COLOR
    @text_sprite.bitmap.font.bold = true

    @text_sprite.bitmap.draw_text(
      0,
      0,
      640,
      64,
      title,
      1
    )

    @text_sprite.y = 180
    @text_sprite.opacity = 0

    #-----------------------------------------------------------------------
    # Sub texto
    #-----------------------------------------------------------------------

    @sub_text_sprite = Sprite.new(@viewport)
    @sub_text_sprite.bitmap = Bitmap.new(640, 32)

    @sub_text_sprite.bitmap.font.name = FONT_NAME
    @sub_text_sprite.bitmap.font.size = SUB_FONT_SIZE
    @sub_text_sprite.bitmap.font.color = TEXT_COLOR

    @sub_text_sprite.y = 340
    @sub_text_sprite.opacity = 0
  end

  #===========================================================================
  # ■ update_bar
  #===========================================================================

  def self.update_bar

    refresh_fill_bar
  end

  #===========================================================================
  # ■ refresh_fill_bar
  #===========================================================================

  def self.refresh_fill_bar

    return if @bar_fill.nil?

    @bar_fill.bitmap.clear

    fill_width = (BAR_WIDTH * @progress / 100.0).to_i

    @bar_fill.bitmap.fill_rect(
      0,
      0,
      fill_width,
      BAR_HEIGHT,
      BAR_COLOR
    )
  end

  #===========================================================================
  # ■ update_message
  #===========================================================================

  def self.update_message(message)

    return if @sub_text_sprite.nil?

    @sub_text_sprite.bitmap.clear

    @sub_text_sprite.bitmap.font.name = FONT_NAME
    @sub_text_sprite.bitmap.font.size = SUB_FONT_SIZE
    @sub_text_sprite.bitmap.font.color = TEXT_COLOR

    @sub_text_sprite.bitmap.draw_text(
      0,
      0,
      640,
      32,
      message,
      1
    )
  end

  #===========================================================================
  # ■ fade_in
  #===========================================================================

  def self.fade_in

    while @background.opacity < 255

      @background.opacity += FADE_SPEED

      @bar_bg.opacity += FADE_SPEED
      @bar_fill.opacity += FADE_SPEED

      @text_sprite.opacity += FADE_SPEED
      @sub_text_sprite.opacity += FADE_SPEED

      Graphics.update
    end
  end

  #===========================================================================
  # ■ fade_out
  #===========================================================================

  def self.fade_out

    while @background.opacity > 0

      @background.opacity -= FADE_SPEED

      @bar_bg.opacity -= FADE_SPEED
      @bar_fill.opacity -= FADE_SPEED

      @text_sprite.opacity -= FADE_SPEED
      @sub_text_sprite.opacity -= FADE_SPEED

      Graphics.update
    end
  end

  #===========================================================================
  # ■ dispose_all
  #===========================================================================

  def self.dispose_all

    sprites = [
      @background,
      @bar_bg,
      @bar_fill,
      @text_sprite,
      @sub_text_sprite
    ]

    for sprite in sprites
      next if sprite.nil?

      sprite.bitmap.dispose if sprite.bitmap and !sprite.bitmap.disposed?
      sprite.dispose unless sprite.disposed?
    end

    @viewport.dispose unless @viewport.nil? or @viewport.disposed?

    @background = nil
    @bar_bg = nil
    @bar_fill = nil
    @text_sprite = nil
    @sub_text_sprite = nil
    @viewport = nil
  end

end
