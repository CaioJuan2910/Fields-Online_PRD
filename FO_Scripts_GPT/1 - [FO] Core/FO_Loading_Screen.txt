#==============================================================================
# ■ FO_LOADING_SCREEN
#==============================================================================

module FO_LOADING_SCREEN

  #--------------------------------------------------------------------------
  # ● Resolução
  #--------------------------------------------------------------------------

  WIDTH  = 640
  HEIGHT = 480

  #--------------------------------------------------------------------------
  # ● Barra
  #--------------------------------------------------------------------------

  BAR_WIDTH  = 400
  BAR_HEIGHT = 24

  #--------------------------------------------------------------------------
  # ● Posição
  #--------------------------------------------------------------------------

  BAR_X =
    (WIDTH - BAR_WIDTH) / 2

  BAR_Y =
    320

  #--------------------------------------------------------------------------
  # ● Fonte
  #--------------------------------------------------------------------------

  FONT_NAME = "Arial"

  TITLE_SIZE = 32

  STATUS_SIZE = 18

  #--------------------------------------------------------------------------
  # ● Cores
  #--------------------------------------------------------------------------

  BACKGROUND_COLOR =
    Color.new(0,0,0)

  BAR_BG_COLOR =
    Color.new(40,40,40)

  BAR_FILL_COLOR =
    Color.new(80,180,255)

  TEXT_COLOR =
    Color.new(255,255,255)

end

#==============================================================================
# ■ FO_Loading_Screen
#==============================================================================

module FO_Loading_Screen

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @viewport = nil

  @background = nil

  @bar_sprite = nil

  @text_sprite = nil

  @status_sprite = nil

  @progress = 0

  @status = "Initializing..."

  @finished = false

  #--------------------------------------------------------------------------
  # ● Start
  #--------------------------------------------------------------------------

  def self.start

    create_viewport

    create_background

    create_bar

    create_texts

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Create Viewport
  #--------------------------------------------------------------------------

  def self.create_viewport

    @viewport =
      Viewport.new(
        0,
        0,
        Graphics.width,
        Graphics.height
      )

    @viewport.z = 999999

  end

  #--------------------------------------------------------------------------
  # ● Create Background
  #--------------------------------------------------------------------------

  def self.create_background

    @background =
      Sprite.new(@viewport)

    @background.bitmap =
      Bitmap.new(
        Graphics.width,
        Graphics.height
      )

    @background.bitmap.fill_rect(
      0,
      0,
      Graphics.width,
      Graphics.height,
      FO_LOADING_SCREEN::BACKGROUND_COLOR
    )

  end

  #--------------------------------------------------------------------------
  # ● Create Bar
  #--------------------------------------------------------------------------

  def self.create_bar

    @bar_sprite =
      Sprite.new(@viewport)

    @bar_sprite.bitmap =
      Bitmap.new(
        Graphics.width,
        Graphics.height
      )

  end

  #--------------------------------------------------------------------------
  # ● Create Texts
  #--------------------------------------------------------------------------

  def self.create_texts

    #--------------------------------------------------------------------------
    # ● Title
    #--------------------------------------------------------------------------

    @text_sprite =
      Sprite.new(@viewport)

    @text_sprite.bitmap =
      Bitmap.new(
        Graphics.width,
        Graphics.height
      )

    font =
      @text_sprite.bitmap.font

    font.name =
      FO_LOADING_SCREEN::FONT_NAME

    font.size =
      FO_LOADING_SCREEN::TITLE_SIZE

    font.bold = true

    font.color =
      FO_LOADING_SCREEN::TEXT_COLOR

    @text_sprite.bitmap.draw_text(
      0,
      160,
      Graphics.width,
      48,
      "FIELDS ONLINE",
      1
    )

    #--------------------------------------------------------------------------
    # ● Status
    #--------------------------------------------------------------------------

    @status_sprite =
      Sprite.new(@viewport)

    @status_sprite.bitmap =
      Bitmap.new(
        Graphics.width,
        Graphics.height
      )

  end

  #--------------------------------------------------------------------------
  # ● Set Status
  #--------------------------------------------------------------------------

  def self.set_status(text)

    @status =
      text

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Set Progress
  #--------------------------------------------------------------------------

  def self.set_progress(value)

    @progress =
      value

    if @progress < 0
      @progress = 0
    end

    if @progress > 100
      @progress = 100
    end

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Refresh
  #--------------------------------------------------------------------------

  def self.refresh

    refresh_bar

    refresh_status

    Graphics.update

  end

  #--------------------------------------------------------------------------
  # ● Refresh Bar
  #--------------------------------------------------------------------------

  def self.refresh_bar

    bitmap =
      @bar_sprite.bitmap

    bitmap.clear

    #--------------------------------------------------------------------------
    # ● Fundo
    #--------------------------------------------------------------------------

    bitmap.fill_rect(
      FO_LOADING_SCREEN::BAR_X,
      FO_LOADING_SCREEN::BAR_Y,
      FO_LOADING_SCREEN::BAR_WIDTH,
      FO_LOADING_SCREEN::BAR_HEIGHT,
      FO_LOADING_SCREEN::BAR_BG_COLOR
    )

    #--------------------------------------------------------------------------
    # ● Fill
    #--------------------------------------------------------------------------

    fill_width =
      (
        FO_LOADING_SCREEN::BAR_WIDTH *
        @progress / 100.0
      ).to_i

    bitmap.fill_rect(
      FO_LOADING_SCREEN::BAR_X,
      FO_LOADING_SCREEN::BAR_Y,
      fill_width,
      FO_LOADING_SCREEN::BAR_HEIGHT,
      FO_LOADING_SCREEN::BAR_FILL_COLOR
    )

    #--------------------------------------------------------------------------
    # ● Percent
    #--------------------------------------------------------------------------

    font =
      bitmap.font

    font.name =
      FO_LOADING_SCREEN::FONT_NAME

    font.size = 16

    font.bold = true

    font.color =
      FO_LOADING_SCREEN::TEXT_COLOR

    bitmap.draw_text(
      0,
      FO_LOADING_SCREEN::BAR_Y + 28,
      Graphics.width,
      24,
      "#{@progress.to_i}%",
      1
    )

  end

  #--------------------------------------------------------------------------
  # ● Refresh Status
  #--------------------------------------------------------------------------

  def self.refresh_status

    bitmap =
      @status_sprite.bitmap

    bitmap.clear

    font =
      bitmap.font

    font.name =
      FO_LOADING_SCREEN::FONT_NAME

    font.size =
      FO_LOADING_SCREEN::STATUS_SIZE

    font.color =
      FO_LOADING_SCREEN::TEXT_COLOR

    bitmap.draw_text(
      0,
      270,
      Graphics.width,
      32,
      @status,
      1
    )

  end

  #--------------------------------------------------------------------------
  # ● Finish
  #--------------------------------------------------------------------------

  def self.finish

    return if @finished

    @finished = true

    fade_out

    dispose

  end

  #--------------------------------------------------------------------------
  # ● Fade Out
  #--------------------------------------------------------------------------

  def self.fade_out

    for i in 0..30

      opacity =
        255 - (i * 8)

      set_opacity(opacity)

      Graphics.update

    end

  end

  #--------------------------------------------------------------------------
  # ● Set Opacity
  #--------------------------------------------------------------------------

  def self.set_opacity(value)

    @background.opacity =
      value

    @bar_sprite.opacity =
      value

    @text_sprite.opacity =
      value

    @status_sprite.opacity =
      value

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def self.dispose

    dispose_sprite(@background)
    dispose_sprite(@bar_sprite)
    dispose_sprite(@text_sprite)
    dispose_sprite(@status_sprite)

    if @viewport != nil

      @viewport.dispose

    end

  end

  #--------------------------------------------------------------------------
  # ● Dispose Sprite
  #--------------------------------------------------------------------------

  def self.dispose_sprite(sprite)

    return if sprite.nil?

    if sprite.bitmap != nil

      sprite.bitmap.dispose

    end

    sprite.dispose

  end

end
