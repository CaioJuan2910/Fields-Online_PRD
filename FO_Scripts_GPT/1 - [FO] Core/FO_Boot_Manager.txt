#==============================================================================
# ■ FO_Boot_Manager
#==============================================================================

module FO_Boot_Manager

  #--------------------------------------------------------------------------
  # ■ Estado
  #--------------------------------------------------------------------------

  @started = false

  #--------------------------------------------------------------------------
  # ● Start
  #--------------------------------------------------------------------------

  def self.start

    return if @started

    @started = true

    #--------------------------------------------------------------------------
    # ● Loading Screen
    #--------------------------------------------------------------------------

    FO_Loading_Screen.start

    #--------------------------------------------------------------------------
    # ● Sequência
    #--------------------------------------------------------------------------

    boot_sequence

    #--------------------------------------------------------------------------
    # ● Finaliza
    #--------------------------------------------------------------------------

    FO_Loading_Screen.finish

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      if FO_BOOT_MANAGER::ENABLE_BOOT_DEBUG

        FO_DEBUG.warning(
          "[BOOT COMPLETED]"
        )

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Boot Sequence
  #--------------------------------------------------------------------------

  def self.boot_sequence

    load_engine

    load_database

    load_player

    load_world

    load_ui

    finalize_boot

  end

  #--------------------------------------------------------------------------
  # ● Load Engine
  #--------------------------------------------------------------------------

  def self.load_engine

    update_loading(
      "Loading Engine...",
      10
    )

    Graphics.update

    #--------------------------------------------------------------------------
    # ● Inicializa Engine
    #--------------------------------------------------------------------------

    if defined?(FO_ENGINE)

      FO_ENGINE.start

    end

  end

  #--------------------------------------------------------------------------
  # ● Load Database
  #--------------------------------------------------------------------------

  def self.load_database

    update_loading(
      "Loading Database...",
      30
    )

    Graphics.update

    #--------------------------------------------------------------------------
    # ● Data Manager
    #--------------------------------------------------------------------------

    if defined?(FO_Data_Manager)

      FO_Data_Manager.reload

    end

  end

  #--------------------------------------------------------------------------
  # ● Load Player
  #--------------------------------------------------------------------------

  def self.load_player

    update_loading(
      "Loading Player...",
      50
    )

    Graphics.update

    #--------------------------------------------------------------------------
    # ● Player Data
    #--------------------------------------------------------------------------

    if defined?(FO_Player_Data)

      FO_Player_Data.load

    end

    #--------------------------------------------------------------------------
    # ● Player Stats
    #--------------------------------------------------------------------------

    if defined?(FO_Player_Stats)

      FO_Player_Stats.refresh

    end

  end

  #--------------------------------------------------------------------------
  # ● Load World
  #--------------------------------------------------------------------------

  def self.load_world

    update_loading(
      "Loading World...",
      75
    )

    Graphics.update

    #--------------------------------------------------------------------------
    # ● Entity Manager
    #--------------------------------------------------------------------------

    if defined?(FO_Entity_Manager)

      FO_Entity_Manager.refresh

    end

    #--------------------------------------------------------------------------
    # ● Spawn System
    #--------------------------------------------------------------------------

    if defined?(FO_Spawn_Manager)

      FO_Spawn_Manager.setup_map_spawns

    end

  end

  #--------------------------------------------------------------------------
  # ● Load UI
  #--------------------------------------------------------------------------

  def self.load_ui

    update_loading(
      "Loading Interface...",
      90
    )

    Graphics.update

    #--------------------------------------------------------------------------
    # ● HUD
    #--------------------------------------------------------------------------

    if defined?(FO_Player_HUD)

      FO_Player_HUD.update

    end

  end

  #--------------------------------------------------------------------------
  # ● Finalize Boot
  #--------------------------------------------------------------------------

  def self.finalize_boot

    update_loading(
      "Entering World...",
      100
    )

    #--------------------------------------------------------------------------
    # ● Pequeno Delay
    #--------------------------------------------------------------------------

    for i in 0..20

      Graphics.update

    end

  end

  #--------------------------------------------------------------------------
  # ● Update Loading
  #--------------------------------------------------------------------------

  def self.update_loading(text, progress)

    FO_Loading_Screen.set_status(
      text
    )

    FO_Loading_Screen.set_progress(
      progress
    )

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      if FO_BOOT_MANAGER::ENABLE_BOOT_DEBUG

        FO_DEBUG.log(
          "[BOOT] #{text}"
        )

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Started?
  #--------------------------------------------------------------------------

  def self.started?

    return @started == true

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  #--------------------------------------------------------------------------
  # ● Alias Main
  #--------------------------------------------------------------------------

  alias fo_boot_manager_main_v1 main

  #--------------------------------------------------------------------------
  # ● Main
  #--------------------------------------------------------------------------

  def main

    #--------------------------------------------------------------------------
    # ● Boot Manager
    #--------------------------------------------------------------------------

    unless FO_Boot_Manager.started?

      FO_Boot_Manager.start

    end

    #--------------------------------------------------------------------------
    # ● Original
    #--------------------------------------------------------------------------

    fo_boot_manager_main_v1

  end

end
