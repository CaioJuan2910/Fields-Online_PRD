#==============================================================================
# ■ FO_Entity_Manager
#==============================================================================

module FO_Entity_Manager

  #--------------------------------------------------------------------------
  # ■ Entities
  #--------------------------------------------------------------------------

  @entities = []

  #--------------------------------------------------------------------------
  # ● Register Entity
  #--------------------------------------------------------------------------
  def self.register(entity)

    return if entity.nil?

    #--------------------------------------------------------------------------
    # Combat Entity?
    #--------------------------------------------------------------------------

    return unless entity.respond_to?(
      :fo_combat_entity?
    )

    return unless entity.fo_combat_entity?

    #--------------------------------------------------------------------------
    # Já registrado?
    #--------------------------------------------------------------------------

    return if @entities.include?(entity)

    #--------------------------------------------------------------------------
    # Registra
    #--------------------------------------------------------------------------

    @entities.push(entity)

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[ENTITY REGISTER] #{entity.event.name}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Remove Entity
  #--------------------------------------------------------------------------
  def self.remove(entity)

    return if entity.nil?

    @entities.delete(entity)

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    begin

      FO_CORE.debug(
        "[ENTITY REMOVE] #{entity.event.name}"
      )

    rescue

    end

  end

  #--------------------------------------------------------------------------
  # ● Entities
  #--------------------------------------------------------------------------
  def self.entities

    return @entities

  end

  #--------------------------------------------------------------------------
  # ● Valid?
  #--------------------------------------------------------------------------
  def self.valid_entity?(entity)

    return false if entity.nil?

    #--------------------------------------------------------------------------
    # Morto?
    #--------------------------------------------------------------------------

    return false if entity.fo_dead?

    #--------------------------------------------------------------------------
    # HP
    #--------------------------------------------------------------------------

    return false if entity.fo_hp <= 0

    #--------------------------------------------------------------------------
    # Existe?
    #--------------------------------------------------------------------------

    begin

      entity.event

    rescue

      return false

    end

    return true

  end

  #--------------------------------------------------------------------------
  # ● Find By Event ID
  #--------------------------------------------------------------------------
  def self.find_by_event_id(event_id)

    for entity in @entities

      next if entity.nil?

      next unless valid_entity?(entity)

      if entity.id == event_id

        return entity

      end

    end

    return nil

  end

  #--------------------------------------------------------------------------
  # ● Find Bosses
  #--------------------------------------------------------------------------
  def self.bosses

    results = []

    for entity in @entities

      next if entity.nil?

      next unless valid_entity?(entity)

      #----------------------------------------------------------------------
      # Boss tag
      #----------------------------------------------------------------------

      begin

        if entity.event.name.include?("!")

          results.push(entity)

        end

      rescue

      end

    end

    return results

  end

  #--------------------------------------------------------------------------
  # ● Find Enemies
  #--------------------------------------------------------------------------
  def self.enemies

    results = []

    for entity in @entities

      next if entity.nil?

      next unless valid_entity?(entity)

      results.push(entity)

    end

    return results

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    #--------------------------------------------------------------------------
    # Remove inválidos
    #--------------------------------------------------------------------------

    @entities.delete_if do |entity|

      !valid_entity?(entity)

    end

  end

end

#==============================================================================
# ■ Registro no System Manager
#==============================================================================

FO_System_Manager.register(
  FO_Entity_Manager
)

#==============================================================================
# ■ Game_Event
#==============================================================================

class Game_Event < Game_Character

  #--------------------------------------------------------------------------
  # Alias Setup Enemy
  #--------------------------------------------------------------------------

  alias fo_entity_manager_setup_enemy_v1 setup_fo_enemy

  #--------------------------------------------------------------------------
  # ● Setup Enemy
  #--------------------------------------------------------------------------
  def setup_fo_enemy

    #--------------------------------------------------------------------------
    # Original
    #--------------------------------------------------------------------------

    fo_entity_manager_setup_enemy_v1

    #--------------------------------------------------------------------------
    # Enemy?
    #--------------------------------------------------------------------------

    return unless fo_enemy?

    #--------------------------------------------------------------------------
    # Registra entidade
    #--------------------------------------------------------------------------

    FO_Entity_Manager.register(
      self
    )

  end

end
