module FO_ENTITY_MANAGER

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  AUTO_REFRESH_INTERVAL =
    60

end

#==============================================================================
# ■ FO_Entity_Manager
#==============================================================================

module FO_Entity_Manager

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @entities = []

  @combat_entities = []

  @npc_entities = []

  @quest_entities = []

  @refresh_timer = 0

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_refresh_timer

  end

  #--------------------------------------------------------------------------
  # ● Update Refresh Timer
  #--------------------------------------------------------------------------

  def self.update_refresh_timer

    @refresh_timer += 1

    return if @refresh_timer <
              FO_ENTITY_MANAGER::AUTO_REFRESH_INTERVAL

    @refresh_timer = 0

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Refresh
  #--------------------------------------------------------------------------

  def self.refresh

    clear

    return if $game_map.nil?

    for event in $game_map.events.values

      next if event.nil?

      register_entity(event)

    end

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[ENTITY REFRESH] #{@entities.size}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Register Entity
  #--------------------------------------------------------------------------

  def self.register_entity(entity)

    return if entity.nil?

    @entities.push(entity)

    #--------------------------------------------------------------------------
    # ● Combat
    #--------------------------------------------------------------------------

    if entity.respond_to?(:fo_combat_entity?)

      if entity.fo_combat_entity?

        @combat_entities.push(entity)

      end

    end

    #--------------------------------------------------------------------------
    # ● NPC
    #--------------------------------------------------------------------------

    if entity.respond_to?(:fo_npc?)

      if entity.fo_npc?

        @npc_entities.push(entity)

      end

    end

    #--------------------------------------------------------------------------
    # ● Quest
    #--------------------------------------------------------------------------

    if entity.respond_to?(:fo_quest?)

      if entity.fo_quest?

        @quest_entities.push(entity)

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    @entities.clear

    @combat_entities.clear

    @npc_entities.clear

    @quest_entities.clear

  end

  #--------------------------------------------------------------------------
  # ● All Entities
  #--------------------------------------------------------------------------

  def self.entities

    return @entities

  end

  #--------------------------------------------------------------------------
  # ● Combat Entities
  #--------------------------------------------------------------------------

  def self.combat_entities

    return @combat_entities

  end

  #--------------------------------------------------------------------------
  # ● NPC Entities
  #--------------------------------------------------------------------------

  def self.npc_entities

    return @npc_entities

  end

  #--------------------------------------------------------------------------
  # ● Quest Entities
  #--------------------------------------------------------------------------

  def self.quest_entities

    return @quest_entities

  end

  #--------------------------------------------------------------------------
  # ● Alive Combat Entities
  #--------------------------------------------------------------------------

  def self.alive_combat_entities

    result = []

    for entity in @combat_entities

      next if entity.nil?

      next if entity.fo_dead?

      result.push(entity)

    end

    return result

  end

  #--------------------------------------------------------------------------
  # ● Find By Event ID
  #--------------------------------------------------------------------------

  def self.find_by_event_id(event_id)

    return nil if event_id.nil?

    for entity in @entities

      next if entity.nil?

      return entity if entity.id == event_id

    end

    return nil

  end

  #--------------------------------------------------------------------------
  # ● Nearest Combat Entity
  #--------------------------------------------------------------------------

  def self.nearest_combat_entity(range = 8)

    player =
      $game_player

    return nil if player.nil?

    nearest = nil

    nearest_distance = nil

    for entity in alive_combat_entities

      next if entity.nil?

      distance =
        calculate_distance(
          player,
          entity
        )

      next if distance > range

      if nearest.nil?

        nearest = entity

        nearest_distance = distance

        next

      end

      if distance < nearest_distance

        nearest = entity

        nearest_distance = distance

      end

    end

    return nearest

  end

  #--------------------------------------------------------------------------
  # ● Calculate Distance
  #--------------------------------------------------------------------------

  def self.calculate_distance(a, b)

    return 999 unless a
    return 999 unless b

    dx =
      (a.x - b.x).abs

    dy =
      (a.y - b.y).abs

    return dx + dy

  end

  #--------------------------------------------------------------------------
  # ● Entities In Range
  #--------------------------------------------------------------------------

  def self.entities_in_range(x, y, range)

    result = []

    for entity in @entities

      next if entity.nil?

      dx =
        (entity.x - x).abs

      dy =
        (entity.y - y).abs

      distance =
        dx + dy

      if distance <= range

        result.push(entity)

      end

    end

    return result

  end

  #--------------------------------------------------------------------------
  # ● Remove Entity
  #--------------------------------------------------------------------------

  def self.remove_entity(entity)

    return if entity.nil?

    @entities.delete(entity)

    @combat_entities.delete(entity)

    @npc_entities.delete(entity)

    @quest_entities.delete(entity)

  end

  #--------------------------------------------------------------------------
  # ● Map Changed
  #--------------------------------------------------------------------------

  def self.map_changed

    clear

    refresh

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  #--------------------------------------------------------------------------
  # ● Alias
  #--------------------------------------------------------------------------

  alias fo_entity_manager_transfer_player_v4 transfer_player

  #--------------------------------------------------------------------------
  # ● Transfer Player
  #--------------------------------------------------------------------------

  def transfer_player

    fo_entity_manager_transfer_player_v4

    FO_Entity_Manager.map_changed

  end

end
