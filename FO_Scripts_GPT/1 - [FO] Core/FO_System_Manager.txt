module FO_System_Manager

  #===========================================================================
  # ■ Sistemas registrados
  #===========================================================================

  @systems = []

  #===========================================================================
  # ● Register
  #---------------------------------------------------------------------------
  # Registra sistema
  #===========================================================================

  def self.register(system)

    return if system.nil?

    #--------------------------------------------------------------------------
    # Evita duplicidade
    #--------------------------------------------------------------------------

    return if @systems.include?(system)

    #--------------------------------------------------------------------------
    # Adiciona
    #--------------------------------------------------------------------------

    @systems.push(system)

    #--------------------------------------------------------------------------
    # Debug
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[SYSTEM REGISTER] #{system}"
    )

  end

  #===========================================================================
  # ● Update
  #===========================================================================

  def self.update

    #--------------------------------------------------------------------------
    # Atualiza sistemas
    #--------------------------------------------------------------------------

    for system in @systems

      next if system.nil?

      #----------------------------------------------------------------------
      # Possui update?
      #----------------------------------------------------------------------

      next unless system.respond_to?(:update)

      #----------------------------------------------------------------------
      # Update
      #----------------------------------------------------------------------

      begin

        system.update

      rescue => e

        FO_CORE.debug(
          "[SYSTEM ERROR] #{system}"
        )

        FO_CORE.debug(
          e.to_s
        )

      end

    end

  end

  #===========================================================================
  # ● Systems
  #===========================================================================

  def self.systems

    return @systems

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  #--------------------------------------------------------------------------
  # Alias Update
  #--------------------------------------------------------------------------

  alias fo_system_manager_update update

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def update

    #--------------------------------------------------------------------------
    # Original
    #--------------------------------------------------------------------------

    fo_system_manager_update

    #--------------------------------------------------------------------------
    # Systems Update
    #--------------------------------------------------------------------------

    FO_System_Manager.update

  end

end
