#==============================================================================
# ■ FO_System_Manager
#==============================================================================

module FO_System_Manager

  #--------------------------------------------------------------------------
  # ■ Sistemas
  #--------------------------------------------------------------------------

  @systems = []

  #--------------------------------------------------------------------------
  # ● Register System
  #--------------------------------------------------------------------------

  def self.register_system(system)

    return if system.nil?

    return if @systems.include?(system)

    @systems.push(system)

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      if FO_SYSTEM_MANAGER::ENABLE_DEBUG

        FO_DEBUG.warning(
          "[LEGACY SYSTEM] #{system}"
        )

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Remove System
  #--------------------------------------------------------------------------

  def self.remove_system(system)

    return if system.nil?

    @systems.delete(system)

  end

  #--------------------------------------------------------------------------
  # ● Systems
  #--------------------------------------------------------------------------

  def self.systems

    return @systems

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    for system in @systems

      next if system.nil?

      begin

        if system.respond_to?(:update)

          system.update

        end

      rescue => e

        if defined?(FO_DEBUG)

          FO_DEBUG.error(
            "[SYSTEM ERROR] #{system} => #{e}"
          )

        end

      end

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    @systems.clear

  end

  #--------------------------------------------------------------------------
  # ● Registered?
  #--------------------------------------------------------------------------

  def self.registered?(system)

    return @systems.include?(system)

  end

  #--------------------------------------------------------------------------
  # ● Count
  #--------------------------------------------------------------------------

  def self.count

    return @systems.size

  end

end

#==============================================================================
# ■ FO_ENGINE Bridge
#==============================================================================

if defined?(FO_ENGINE)

  module FO_ENGINE

    #--------------------------------------------------------------------------
    # ● Alias
    #--------------------------------------------------------------------------

    class << self

      alias fo_system_manager_register_system_v4 register_system

    end

    #--------------------------------------------------------------------------
    # ● Register System
    #--------------------------------------------------------------------------

    def self.register_system(system)

      #----------------------------------------------------------------------
      # ● Original
      #----------------------------------------------------------------------

      fo_system_manager_register_system_v4(
        system
      )

      #----------------------------------------------------------------------
      # ● Compatibilidade
      #----------------------------------------------------------------------

      unless FO_System_Manager.registered?(system)

        FO_System_Manager.register_system(
          system
        )

      end

    end

  end

end

  alias fo_system_manager_update_v4 update

  #--------------------------------------------------------------------------

    # ● Original
    #--------------------------------------------------------------------------

    fo_system_manager_update_v4

    #--------------------------------------------------------------------------
    # ● Legacy Update
    #--------------------------------------------------------------------------

    FO_System_Manager.update

  end

end
