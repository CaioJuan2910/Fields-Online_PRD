#==============================================================================
# ■ FO_DEBUG
#==============================================================================

module FO_DEBUG

  #--------------------------------------------------------------------------
  # ● Log
  #--------------------------------------------------------------------------

  def self.log(text)

    return unless FO_CORE::ENABLE_DEBUG

    FO_Core.log
      "[FO LOG] #{text}\n"
    )

  end

  #--------------------------------------------------------------------------
  # ● Warning
  #--------------------------------------------------------------------------

  def self.warning(text)

    return unless FO_CORE::ENABLE_WARNINGS

    FO_Core.log
      "[FO WARNING] #{text}\n"
    )

  end

  #--------------------------------------------------------------------------
  # ● Error
  #--------------------------------------------------------------------------

  def self.error(text)

    FO_Core.log
      "[FO ERROR] #{text}\n"
    )

    if FO_CORE::ENABLE_ERROR_POPUPS

      begin

        FO_Core.log(text)

      rescue

      end

    end

  end

end

#==============================================================================
# ■ FO_Utils
#==============================================================================

module FO_Utils

  #--------------------------------------------------------------------------
  # ● Clamp
  #--------------------------------------------------------------------------

  def self.clamFO_Core.logvalue, min, max)

    return min if value < min

    return max if value > max

    return value

  end

  #--------------------------------------------------------------------------
  # ● Distance
  #--------------------------------------------------------------------------

  def self.distance(x1, y1, x2, y2)

    dx =
      (x2 - x1).abs

    dy =
      (y2 - y1).abs

    return dx + dy

  end

  #--------------------------------------------------------------------------
  # ● Percent
  #--------------------------------------------------------------------------

  def self.percent(current, maximum)

    return 0 if maximum <= 0

    return (
      current * 100 / maximum
    ).to_i

  end

  #--------------------------------------------------------------------------
  # ● Random Percent
  #--------------------------------------------------------------------------

  def self.random_percent(chance)

    roll =
      rand(100) + 1

    return roll <= chance

  end

  #--------------------------------------------------------------------------
  # ● Safe Name
  #--------------------------------------------------------------------------

  def self.safe_name(name)

    return "" if name.nil?

    text =
      name.clone

    text.gsub!("$", "")
    text.gsub!("!", "")
    text.gsub!("@", "")
    text.gsub!("?", "")

    return text

  end

end
