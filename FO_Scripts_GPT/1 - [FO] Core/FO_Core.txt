#==============================================================================
# ■ [FO] Segurança
#==============================================================================
module FO_SECURITY

  #--------------------------------------------------------------------------
  # Sanitiza texto simples
  #--------------------------------------------------------------------------
  def self.sanitize_text(text)
    return "" if text.nil?

    text = text.to_s

    # Remove caracteres perigosos
    text.gsub!(/\r\n/, " ")
    text.gsub!(/\n/, " ")
    text.gsub!(/\t/, " ")

    return text
  end

  #--------------------------------------------------------------------------
  # Validação simples de número
  #--------------------------------------------------------------------------
  def self.valid_number?(value)
    return false if value.nil?

    begin
      Float(value)
      return true
    rescue
      return false
    end
  end

end

#==============================================================================
# ■ [FO] Sistema de Logs
#==============================================================================
module FO_LOG

  @log_cache = []

  #--------------------------------------------------------------------------
  # Cria pasta de logs
  #--------------------------------------------------------------------------
  def self.create_log_folder

    return unless FO_CORE_CONFIG::ENABLE_LOGS

    begin

      unless FileTest.exist?(FO_CORE_CONFIG::LOG_FOLDER)
        Dir.mkdir(FO_CORE_CONFIG::LOG_FOLDER)
      end

    rescue

      p "FO_Core: Não foi possível criar pasta Logs."

    end

  end

  #--------------------------------------------------------------------------
  # Caminho do log
  #--------------------------------------------------------------------------
  def self.log_path

    return "#{FO_CORE_CONFIG::LOG_FOLDER}/#{FO_CORE_CONFIG::MAIN_LOG_FILE}"

  end

  #--------------------------------------------------------------------------
  # Escreve no log
  #--------------------------------------------------------------------------
  def self.write(message)

    return unless FO_CORE_CONFIG::ENABLE_LOGS

    begin

      create_log_folder

      time = Time.now.strftime("%d/%m/%Y %H:%M:%S")

      clean_message = FO_SECURITY.sanitize_text(message)

      final_text = "[#{time}] #{clean_message}"

      @log_cache << final_text

      trim_log_cache

      File.open(log_path, "a") do |file|
        file.puts(final_text)
      end

      if FO_CORE_CONFIG::ENABLE_CONSOLE_OUTPUT
        p final_text
      end

    rescue

      p "FO_Core: Erro ao escrever log."

    end

  end

  #--------------------------------------------------------------------------
  # Limita cache
  #--------------------------------------------------------------------------
  def self.trim_log_cache

    while @log_cache.size > FO_CORE_CONFIG::MAX_LOG_LINES
      @log_cache.shift
    end

  end

  #--------------------------------------------------------------------------
  # Retorna cache
  #--------------------------------------------------------------------------
  def self.cache

    return @log_cache

  end

end

#==============================================================================
# ■ [FO] Utilitários Globais
#==============================================================================
module FO_UTILS

  #--------------------------------------------------------------------------
  # Clampa valor mínimo/máximo
  #--------------------------------------------------------------------------
  def self.clamp(value, min, max)

    return [[value, min].max, max].min

  end

  #--------------------------------------------------------------------------
  # Valor aleatório entre mínimo/máximo
  #--------------------------------------------------------------------------
  def self.random(min, max)

    return rand(max - min + 1) + min

  end

  #--------------------------------------------------------------------------
  # Distância entre dois pontos
  #--------------------------------------------------------------------------
  def self.distance(x1, y1, x2, y2)

    return Math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)

  end

  #--------------------------------------------------------------------------
  # Verifica nil ou vazio
  #--------------------------------------------------------------------------
  def self.blank?(value)

    return true if value.nil?
    return true if value == ""

    return false

  end

  #--------------------------------------------------------------------------
  # Tempo em milissegundos
  #--------------------------------------------------------------------------
  def self.current_time_ms

    return (Time.now.to_f * 1000).to_i

  end

end

#==============================================================================
# ■ [FO] Fail Safe
#==============================================================================
module FO_FAIL_SAFE

  #--------------------------------------------------------------------------
  # Executa código protegido
  #--------------------------------------------------------------------------
  def self.safe_execute(method_name = "Unknown")

    begin

      yield if block_given?

    rescue Exception => e

      FO_DEBUG.error("Fail Safe: #{method_name}")
      FO_DEBUG.error(e.message)

      if e.backtrace != nil
        FO_DEBUG.error(e.backtrace.join("\n"))
      end

    end

  end

end

#==============================================================================
# ■ [FO] Hooks Globais
#==============================================================================
module FO_HOOKS

  @hooks = {}

  #--------------------------------------------------------------------------
  # Registra hook
  #--------------------------------------------------------------------------
  def self.register(name, &block)

    return if name.nil?
    return unless block_given?

    @hooks[name] = [] if @hooks[name].nil?

    @hooks[name] << block

  end

  #--------------------------------------------------------------------------
  # Executa hook
  #--------------------------------------------------------------------------
  def self.call(name, *args)

    return if @hooks[name].nil?

    @hooks[name].each do |hook|

      FO_FAIL_SAFE.safe_execute("Hook #{name}") do
        hook.call(*args)
      end

    end

  end

end

#==============================================================================
# ■ [FO] Monitor de Performance
#==============================================================================
module FO_PERFORMANCE

  @timers = {}

  #--------------------------------------------------------------------------
  # Inicia medição
  #--------------------------------------------------------------------------
  def self.start(name)

    return unless FO_CORE_CONFIG::ENABLE_PERFORMANCE_MONITOR

    @timers[name] = Time.now

  end

  #--------------------------------------------------------------------------
  # Finaliza medição
  #--------------------------------------------------------------------------
  def self.finish(name)

    return unless FO_CORE_CONFIG::ENABLE_PERFORMANCE_MONITOR

    return if @timers[name].nil?

    elapsed = Time.now - @timers[name]

    elapsed_ms = (elapsed * 1000.0).to_i

    FO_DEBUG.log("#{name} levou #{elapsed_ms}ms")

    @timers.delete(name)

  end

end

#==============================================================================
# ■ [FO] Inicialização Principal
#==============================================================================
module FO_CORE

  @initialized = false

  #--------------------------------------------------------------------------
  # Inicialização do sistema
  #--------------------------------------------------------------------------
  def self.initialize

    return if @initialized

    FO_LOG.create_log_folder

    FO_LOG.write("==================================================")
    FO_LOG.write("Inicializando #{FO_CORE_CONFIG::GAME_NAME}")
    FO_LOG.write("Versão: #{FO_CORE_CONFIG::GAME_VERSION}")
    FO_LOG.write("FO_Core v#{FO_VERSION::CORE_VERSION}")
    FO_LOG.write("==================================================")

    FO_COMPATIBILITY.check_rmxp_objects

    validate_environment

    load_global_hooks

    @initialized = true

    FO_LOG.write("FO_Core inicializado com sucesso.")

  end

  #--------------------------------------------------------------------------
  # Valida ambiente
  #--------------------------------------------------------------------------
  def self.validate_environment

    FO_DEBUG.log("Validando ambiente...")

    if RUBY_VERSION.nil?
      FO_DEBUG.warning("Versão Ruby não identificada.")
    else
      FO_DEBUG.log("Ruby detectado.")
    end

  end

  #--------------------------------------------------------------------------
  # Carrega hooks globais
  #--------------------------------------------------------------------------
  def self.load_global_hooks

    return unless FO_CORE_CONFIG::ENABLE_GLOBAL_HOOKS

    FO_DEBUG.log("Carregando hooks globais...")

  end

  #--------------------------------------------------------------------------
  # Verifica se inicializou
  #--------------------------------------------------------------------------
  def self.initialized?

    return @initialized

  end

end

#==============================================================================
# ■ [FO] Auto Inicialização
#------------------------------------------------------------------------------

#==============================================================================
FO_FAIL_SAFE.safe_execute("FO_Core Initialize") do
