module FO_CORE

  #--------------------------------------------------------------------------
  # ● Projeto
  #--------------------------------------------------------------------------

  PROJECT_NAME = "Fields Online"

  VERSION = "5.0"

  #--------------------------------------------------------------------------
  # ● Debug
  #--------------------------------------------------------------------------

  ENABLE_DEBUG = true

  ENABLE_WARNINGS = true

  ENABLE_ERROR_LOG = true

  ENABLE_POPUPS = false

  #--------------------------------------------------------------------------
  # ● Logs
  #--------------------------------------------------------------------------

  LOG_FOLDER = "Logs"

  LOG_FILE = "Fields_Online_Log.txt"

end

#==============================================================================
# ■ FO_Core
#==============================================================================

module FO_Core

  #--------------------------------------------------------------------------
  # ● Cria pasta de logs
  #--------------------------------------------------------------------------

  def self.create_log_folder

    begin

      if !FileTest.exist?(FO_CORE::LOG_FOLDER)
        Dir.mkdir(FO_CORE::LOG_FOLDER)
      end

    rescue

    end

  end

  #--------------------------------------------------------------------------
  # ● Caminho do log
  #--------------------------------------------------------------------------

  def self.log_path

    return FO_CORE::LOG_FOLDER + "/" + FO_CORE::LOG_FILE

  end

  #--------------------------------------------------------------------------
  # ● Log principal
  #--------------------------------------------------------------------------

  def self.log(text)

    return if text == nil

    create_log_folder

    begin

      file = File.open(log_path, "a")

      time = Time.now.strftime("%H:%M:%S")

      file.write("[" + time + "] ")

      file.write(text.to_s)

      file.write("\n")

      file.close

    rescue

    end

  end

  #--------------------------------------------------------------------------
  # ● Debug
  #--------------------------------------------------------------------------

  def self.debug(text)

    return unless FO_CORE::ENABLE_DEBUG

    log("[DEBUG] " + text.to_s)

  end

  #--------------------------------------------------------------------------
  # ● Warning
  #--------------------------------------------------------------------------

  def self.warning(text)

    return unless FO_CORE::ENABLE_WARNINGS

    log("[WARNING] " + text.to_s)

  end

  #--------------------------------------------------------------------------
  # ● Error
  #--------------------------------------------------------------------------

  def self.error(text)

    return unless FO_CORE::ENABLE_ERROR_LOG

    log("[ERROR] " + text.to_s)

  end

  #--------------------------------------------------------------------------
  # ● Popup seguro
  #--------------------------------------------------------------------------

  def self.popup(text)

    return unless FO_CORE::ENABLE_POPUPS

    begin

      msgbox(text.to_s)

    rescue

    end

  end

end

#==============================================================================
# ■ FO_Utils
#==============================================================================

module FO_Utils

  #--------------------------------------------------------------------------
  # ● Clamp
  #--------------------------------------------------------------------------

  def self.clamp(value, min, max)

    return min if value < min

    return max if value > max

    return value

  end

  #--------------------------------------------------------------------------
  # ● Distância
  #--------------------------------------------------------------------------

  def self.distance(x1, y1, x2, y2)

    dx = (x2 - x1).abs
    dy = (y2 - y1).abs

    return dx + dy

  end

  #--------------------------------------------------------------------------
  # ● Percentual
  #--------------------------------------------------------------------------

  def self.percent(current, maximum)

    return 0 if maximum <= 0

    return ((current * 100) / maximum).to_i

  end

  #--------------------------------------------------------------------------
  # ● Chance %
  #--------------------------------------------------------------------------

  def self.random_percent(chance)

    roll = rand(100) + 1

    return roll <= chance

  end

  #--------------------------------------------------------------------------
  # ● Nome seguro
  #--------------------------------------------------------------------------

  def self.safe_name(name)

    return "" if name == nil

    text = name.clone

    text.gsub!("$", "")
    text.gsub!("!", "")
    text.gsub!("@", "")
    text.gsub!("?", "")

    return text

  end

end

  end

  def self.error(text)

  end

end
