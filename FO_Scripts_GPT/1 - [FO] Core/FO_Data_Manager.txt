module FO_DATA_MANAGER

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  ENABLE_CACHE = true

end

#==============================================================================
# ■ FO_Data_Manager
#==============================================================================

module FO_Data_Manager

  #--------------------------------------------------------------------------
  # ■ Cache
  #--------------------------------------------------------------------------

  @items_cache = {}
  @weapons_cache = {}
  @armors_cache = {}
  @skills_cache = {}
  @enemies_cache = {}

  #--------------------------------------------------------------------------
  # ● Setup
  #--------------------------------------------------------------------------

  def self.setup

    build_item_cache

    build_weapon_cache

    build_armor_cache

    build_skill_cache

    build_enemy_cache

  end

  #--------------------------------------------------------------------------
  # ● Build Item Cache
  #--------------------------------------------------------------------------

  def self.build_item_cache

    @items_cache.clear

    return if $data_items.nil?

    for item in $data_items

      next if item.nil?

      @items_cache[item.id] = item

    end

  end

  #--------------------------------------------------------------------------
  # ● Build Weapon Cache
  #--------------------------------------------------------------------------

  def self.build_weapon_cache

    @weapons_cache.clear

    return if $data_weapons.nil?

    for weapon in $data_weapons

      next if weapon.nil?

      @weapons_cache[weapon.id] = weapon

    end

  end

  #--------------------------------------------------------------------------
  # ● Build Armor Cache
  #--------------------------------------------------------------------------

  def self.build_armor_cache

    @armors_cache.clear

    return if $data_armors.nil?

    for armor in $data_armors

      next if armor.nil?

      @armors_cache[armor.id] = armor

    end

  end

  #--------------------------------------------------------------------------
  # ● Build Skill Cache
  #--------------------------------------------------------------------------

  def self.build_skill_cache

    @skills_cache.clear

    return if $data_skills.nil?

    for skill in $data_skills

      next if skill.nil?

      @skills_cache[skill.id] = skill

    end

  end

  #--------------------------------------------------------------------------
  # ● Build Enemy Cache
  #--------------------------------------------------------------------------

  def self.build_enemy_cache

    @enemies_cache.clear

    return if $data_enemies.nil?

    for enemy in $data_enemies

      next if enemy.nil?

      @enemies_cache[enemy.id] = enemy

    end

  end

  #--------------------------------------------------------------------------
  # ● Get Item
  #--------------------------------------------------------------------------

  def self.item(id)

    return nil if id.nil?

    if FO_DATA_MANAGER::ENABLE_CACHE

      return @items_cache[id]

    end

    return $data_items[id]

  end

  #--------------------------------------------------------------------------
  # ● Get Weapon
  #--------------------------------------------------------------------------

  def self.weapon(id)

    return nil if id.nil?

    if FO_DATA_MANAGER::ENABLE_CACHE

      return @weapons_cache[id]

    end

    return $data_weapons[id]

  end

  #--------------------------------------------------------------------------
  # ● Get Armor
  #--------------------------------------------------------------------------

  def self.armor(id)

    return nil if id.nil?

    if FO_DATA_MANAGER::ENABLE_CACHE

      return @armors_cache[id]

    end

    return $data_armors[id]

  end

  #--------------------------------------------------------------------------
  # ● Get Skill
  #--------------------------------------------------------------------------

  def self.skill(id)

    return nil if id.nil?

    if FO_DATA_MANAGER::ENABLE_CACHE

      return @skills_cache[id]

    end

    return $data_skills[id]

  end

  #--------------------------------------------------------------------------
  # ● Get Enemy
  #--------------------------------------------------------------------------

  def self.enemy(id)

    return nil if id.nil?

    if FO_DATA_MANAGER::ENABLE_CACHE

      return @enemies_cache[id]

    end

    return $data_enemies[id]

  end

  #--------------------------------------------------------------------------
  # ● Item Name
  #--------------------------------------------------------------------------

  def self.item_name(id)

    item =
      self.item(id)

    return "Unknown Item" if item.nil?

    return item.name

  end

  #--------------------------------------------------------------------------
  # ● Weapon Name
  #--------------------------------------------------------------------------

  def self.weapon_name(id)

    weapon =
      self.weapon(id)

    return "Unknown Weapon" if weapon.nil?

    return weapon.name

  end

  #--------------------------------------------------------------------------
  # ● Armor Name
  #--------------------------------------------------------------------------

  def self.armor_name(id)

    armor =
      self.armor(id)

    return "Unknown Armor" if armor.nil?

    return armor.name

  end

  #--------------------------------------------------------------------------
  # ● Skill Name
  #--------------------------------------------------------------------------

  def self.skill_name(id)

    skill =
      self.skill(id)

    return "Unknown Skill" if skill.nil?

    return skill.name

  end

  #--------------------------------------------------------------------------
  # ● Item Exists?
  #--------------------------------------------------------------------------

  def self.item_exists?(id)

    return self.item(id) != nil

  end

  #--------------------------------------------------------------------------
  # ● Weapon Exists?
  #--------------------------------------------------------------------------

  def self.weapon_exists?(id)

    return self.weapon(id) != nil

  end

  #--------------------------------------------------------------------------
  # ● Armor Exists?
  #--------------------------------------------------------------------------

  def self.armor_exists?(id)

    return self.armor(id) != nil

  end

  #--------------------------------------------------------------------------
  # ● Skill Exists?
  #--------------------------------------------------------------------------

  def self.skill_exists?(id)

    return self.skill(id) != nil

  end

  #--------------------------------------------------------------------------
  # ● Enemy Exists?
  #--------------------------------------------------------------------------

  def self.enemy_exists?(id)

    return self.enemy(id) != nil

  end

  #--------------------------------------------------------------------------
  # ● Reload
  #--------------------------------------------------------------------------

  def self.reload

    setup

    if defined?(FO_DEBUG)

      FO_DEBUG.warning(
        "[DATA MANAGER] Cache Reloaded"
      )

    end

  end

end

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_Data_Manager.setup

if defined?(FO_DEBUG)

  FO_DEBUG.log(
    "FO_Data_Manager_v4 carregado com sucesso"
  )

end
