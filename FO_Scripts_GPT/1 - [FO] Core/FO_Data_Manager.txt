module FO_Data_Manager

  #--------------------------------------------------------------------------
  # ● Player
  #--------------------------------------------------------------------------
  def self.player

    return nil if $game_party.nil?

    return nil if $game_party.actors.nil?

    return $game_party.actors[0]

  end

  #--------------------------------------------------------------------------
  # ● Actor
  #--------------------------------------------------------------------------
  def self.actor(actor_id)

    return nil if $data_actors.nil?

    return $data_actors[actor_id]

  end

  #--------------------------------------------------------------------------
  # ● Enemy
  #--------------------------------------------------------------------------
  def self.enemy(enemy_id)

    return nil if $data_enemies.nil?

    return $data_enemies[enemy_id]

  end

  #--------------------------------------------------------------------------
  # ● Item
  #--------------------------------------------------------------------------
  def self.item(item_id)

    return nil if $data_items.nil?

    return $data_items[item_id]

  end

  #--------------------------------------------------------------------------
  # ● Weapon
  #--------------------------------------------------------------------------
  def self.weapon(weapon_id)

    return nil if $data_weapons.nil?

    return $data_weapons[weapon_id]

  end

  #--------------------------------------------------------------------------
  # ● Armor
  #--------------------------------------------------------------------------
  def self.armor(armor_id)

    return nil if $data_armors.nil?

    return $data_armors[armor_id]

  end

  #--------------------------------------------------------------------------
  # ● Skill
  #--------------------------------------------------------------------------
  def self.skill(skill_id)

    return nil if $data_skills.nil?

    return $data_skills[skill_id]

  end

  #--------------------------------------------------------------------------
  # ● Gold
  #--------------------------------------------------------------------------
  def self.gold

    return 0 if $game_party.nil?

    return $game_party.gold

  end

  #--------------------------------------------------------------------------
  # ● Map ID
  #--------------------------------------------------------------------------
  def self.map_id

    return 0 if $game_map.nil?

    return $game_map.map_id

  end

  #--------------------------------------------------------------------------
  # ● Party
  #--------------------------------------------------------------------------
  def self.party

    return $game_party

  end

end
