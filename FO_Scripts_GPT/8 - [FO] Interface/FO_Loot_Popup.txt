#==============================================================================
# ■ FO_LOOT_POPUP
#==============================================================================

module FO_LOOT_POPUP

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  START_X = 20

  START_Y = 140

  WIDTH = 320

  HEIGHT = 28

  SPACING = 4

  MAX_POPUPS = 6

  DURATION =
    3 * Graphics.frame_rate

  FADE_SPEED = 5

  FLOAT_SPEED = 1

  Z = 99999

  #--------------------------------------------------------------------------
  # ● Fonte
  #--------------------------------------------------------------------------

  FONT_NAME = "Arial"

  FONT_SIZE = 18

  #--------------------------------------------------------------------------
  # ● Cores
  #--------------------------------------------------------------------------

  COMMON_COLOR =
    Color.new(255,255,255)

  RARE_COLOR =
    Color.new(80,180,255)

  GOLD_COLOR =
    Color.new(255,220,80)

  SUPER_RARE_COLOR =
    Color.new(255,80,255)

end

#==============================================================================
# ■ FO_Loot_Popup_Object
#==============================================================================

class FO_Loot_Popup_Object

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------

  def initialize(index, text, rarity)

    @finished = false

    @timer =
      FO_LOOT_POPUP::DURATION

    create_sprite(
      index,
      text,
      rarity
    )

  end

  #--------------------------------------------------------------------------
  # ● Create Sprite
  #--------------------------------------------------------------------------

  def create_sprite(index, text, rarity)

    @sprite =
      Sprite.new

    @sprite.bitmap =
      Bitmap.new(
        FO_LOOT_POPUP::WIDTH,
        FO_LOOT_POPUP::HEIGHT
      )

    @sprite.x =
      FO_LOOT_POPUP::START_X

    @sprite.y =
      FO_LOOT_POPUP::START_Y +
      (
        index *
        (
          FO_LOOT_POPUP::HEIGHT +
          FO_LOOT_POPUP::SPACING
        )
      )

    @sprite.z =
      FO_LOOT_POPUP::Z

    #--------------------------------------------------------------------------
    # ● Fonte
    #--------------------------------------------------------------------------

    font =
      @sprite.bitmap.font

    font.name =
      FO_LOOT_POPUP::FONT_NAME

    font.size =
      FO_LOOT_POPUP::FONT_SIZE

    font.bold = true

    #--------------------------------------------------------------------------
    # ● Cor
    #--------------------------------------------------------------------------

    case rarity

    when :rare

      font.color =
        FO_LOOT_POPUP::RARE_COLOR

    when :gold

      font.color =
        FO_LOOT_POPUP::GOLD_COLOR

    when :super_rare

      font.color =
        FO_LOOT_POPUP::SUPER_RARE_COLOR

    else

      font.color =
        FO_LOOT_POPUP::COMMON_COLOR

    end

    #--------------------------------------------------------------------------
    # ● Texto
    #--------------------------------------------------------------------------

    @sprite.bitmap.draw_text(
      0,
      0,
      FO_LOOT_POPUP::WIDTH,
      FO_LOOT_POPUP::HEIGHT,
      text
    )

    @sprite.opacity = 255

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def update

    return if @finished

    return if @sprite.nil?

    @timer -= 1

    #--------------------------------------------------------------------------
    # ● Movimento
    #--------------------------------------------------------------------------

    @sprite.y -=
      FO_LOOT_POPUP::FLOAT_SPEED

    #--------------------------------------------------------------------------
    # ● Fade
    #--------------------------------------------------------------------------

    if @timer <= 60

      @sprite.opacity -=
        FO_LOOT_POPUP::FADE_SPEED

    end

    #--------------------------------------------------------------------------
    # ● Finaliza
    #--------------------------------------------------------------------------

    if @timer <= 0

      dispose

      @finished = true

    end

  end

  #--------------------------------------------------------------------------
  # ● Finished?
  #--------------------------------------------------------------------------

  def finished?

    return @finished

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def dispose

    return if @sprite.nil?

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    @sprite.dispose

  end

end

#==============================================================================
# ■ FO_Loot_Popup_Manager
#==============================================================================

module FO_Loot_Popup_Manager

  #--------------------------------------------------------------------------
  # ■ Objetos
  #--------------------------------------------------------------------------

  @popups = []

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    for popup in @popups

      next if popup.nil?

      popup.update

    end

    @popups.delete_if do |popup|

      popup.finished?

    end

  end

  #--------------------------------------------------------------------------
  # ● Show Loot
  #--------------------------------------------------------------------------

  def self.show_loot(name, rarity, amount)

    return if name.nil?

    text =
      "+ #{name}"

    if amount > 1

      text += " x#{amount}"

    end

    #--------------------------------------------------------------------------
    # ● Limite
    #--------------------------------------------------------------------------

    if @popups.size >=
       FO_LOOT_POPUP::MAX_POPUPS

      old_popup =
        @popups.shift

      if old_popup != nil

        old_popup.dispose

      end

    end

    #--------------------------------------------------------------------------
    # ● Cria popup
    #--------------------------------------------------------------------------

    popup =
      FO_Loot_Popup_Object.new(
        @popups.size,
        text,
        rarity
      )

    @popups.push(popup)

    #--------------------------------------------------------------------------
    # ● Debug
    #--------------------------------------------------------------------------

    if defined?(FO_DEBUG)

      FO_DEBUG.log(
        "[LOOT POPUP] #{text}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------

  def self.clear

    for popup in @popups

      next if popup.nil?

      popup.dispose

    end

    @popups.clear

  end

end

#==============================================================================

#==============================================================================

  begin

    FO_ENGINE.register_system(
      FO_Loot_Popup_Manager
    )

  rescue

  end

end
