#==============================================================================
# ■ FO_Loot_Popup
#==============================================================================

class FO_Loot_Popup

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize(text, color, offset_y = 0)

    @finished = false

    create_sprite(
      text,
      color,
      offset_y
    )

  end

  #--------------------------------------------------------------------------
  # ● Criar sprite
  #--------------------------------------------------------------------------
  def create_sprite(text, color, offset_y)

    #--------------------------------------------------------------------------
    # Sprite
    #--------------------------------------------------------------------------

    @sprite =
      Sprite.new

    #--------------------------------------------------------------------------
    # Bitmap
    #--------------------------------------------------------------------------

    @sprite.bitmap =
      Bitmap.new(320, 32)

    #--------------------------------------------------------------------------
    # Fonte
    #--------------------------------------------------------------------------

    font =
      @sprite.bitmap.font

    font.name =
      "Arial"

    font.size =
      20

    font.bold =
      true

    font.color =
      color

    #--------------------------------------------------------------------------
    # Texto
    #--------------------------------------------------------------------------

    @sprite.bitmap.draw_text(
      0,
      0,
      320,
      32,
      text,
      1
    )

    #--------------------------------------------------------------------------
    # Posição
    #--------------------------------------------------------------------------

    @sprite.x =
      160

    @sprite.y =
      260 - offset_y

    #--------------------------------------------------------------------------
    # Z
    #--------------------------------------------------------------------------

    @sprite.z =
      99999

    #--------------------------------------------------------------------------
    # Opacidade
    #--------------------------------------------------------------------------

    @sprite.opacity =
      255

    #--------------------------------------------------------------------------
    # Movimento
    #--------------------------------------------------------------------------

    @move_speed =
      1

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    return if @finished

    #--------------------------------------------------------------------------
    # Movimento
    #--------------------------------------------------------------------------

    @sprite.y -=
      @move_speed

    #--------------------------------------------------------------------------
    # Fade
    #--------------------------------------------------------------------------

    @sprite.opacity -= 4

    #--------------------------------------------------------------------------
    # Finaliza
    #--------------------------------------------------------------------------

    if @sprite.opacity <= 0

      dispose

      @finished = true

    end

  end

  #--------------------------------------------------------------------------
  # ● Finished?
  #--------------------------------------------------------------------------
  def finished?

    return @finished

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------
  def dispose

    return if @sprite.nil?

    #--------------------------------------------------------------------------
    # Bitmap
    #--------------------------------------------------------------------------

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    #--------------------------------------------------------------------------
    # Sprite
    #--------------------------------------------------------------------------

    @sprite.dispose

  end

end

#==============================================================================
# ■ FO_Loot_Popup_Manager
#==============================================================================

module FO_Loot_Popup_Manager

  #--------------------------------------------------------------------------
  # ■ Popups
  #--------------------------------------------------------------------------

  @popups = []

  #--------------------------------------------------------------------------
  # ● Show Loot
  #--------------------------------------------------------------------------
  def self.show_loot(name, rarity, amount = 1)

    #--------------------------------------------------------------------------
    # Texto
    #--------------------------------------------------------------------------

    text = ""

    if amount > 1

      text =
        "#{name} x#{amount}"

    else

      text =
        name

    end

    #--------------------------------------------------------------------------
    # Cor padrão
    #--------------------------------------------------------------------------

    color =
      Color.new(255,255,255)

    #--------------------------------------------------------------------------
    # Prefixo
    #--------------------------------------------------------------------------

    prefix =
      "+ "

    #--------------------------------------------------------------------------
    # Raridades
    #--------------------------------------------------------------------------

    case rarity

    when :common

      color =
        Color.new(255,255,255)

    when :rare

      color =
        Color.new(80,180,255)

      prefix =
        "RARE + "

    when :gold

      color =
        Color.new(255,220,60)

      prefix =
        "GOLD + "

    when :legendary

      color =
        Color.new(255,80,255)

      prefix =
        "LEGENDARY + "

    end

    #--------------------------------------------------------------------------
    # Offset
    #--------------------------------------------------------------------------

    offset_y =
      @popups.size * 28

    #--------------------------------------------------------------------------
    # Popup
    #--------------------------------------------------------------------------

    popup =
      FO_Loot_Popup.new(
        prefix + text,
        color,
        offset_y
      )

    #--------------------------------------------------------------------------
    # Adiciona
    #--------------------------------------------------------------------------

    @popups.push(popup)

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    #--------------------------------------------------------------------------
    # Atualiza
    #--------------------------------------------------------------------------

    for popup in @popups

      next if popup.nil?

      popup.update

    end

    #--------------------------------------------------------------------------
    # Remove finalizados
    #--------------------------------------------------------------------------

    @popups.delete_if do |popup|

      popup.finished?

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------
  def self.clear

    for popup in @popups

      next if popup.nil?

      popup.dispose

    end

    @popups.clear

  end

end

# ■ Registro no System Manager
#==============================================================================

FO_System_Manager.register(
  FO_Loot_Popup_Manager
)
