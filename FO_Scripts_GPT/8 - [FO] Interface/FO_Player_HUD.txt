#==============================================================================
# ■ FO_HUD
#==============================================================================

module FO_HUD

  #--------------------------------------------------------------------------
  # ■ HUD
  #--------------------------------------------------------------------------

  @sprite = nil
  @bitmap = nil

  #--------------------------------------------------------------------------
  # ■ Cache
  #--------------------------------------------------------------------------

  @last_hp = nil
  @last_mp = nil
  @last_gold = nil
  @last_level = nil

  #--------------------------------------------------------------------------
  # ● Start
  #--------------------------------------------------------------------------
  def self.start

    dispose

    #--------------------------------------------------------------------------
    # Sprite
    #--------------------------------------------------------------------------

    @sprite =
      Sprite.new

    @sprite.z =
      9999

    @sprite.x =
      10

    @sprite.y =
      10

    #--------------------------------------------------------------------------
    # Bitmap
    #--------------------------------------------------------------------------

    @bitmap =
      Bitmap.new(260, 90)

    @sprite.bitmap =
      @bitmap

    #--------------------------------------------------------------------------
    # Primeiro draw
    #--------------------------------------------------------------------------

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Player
  #--------------------------------------------------------------------------
  def self.player

    return FO_Data_Manager.player

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    return if player.nil?

    #--------------------------------------------------------------------------
    # Precisa atualizar?
    #--------------------------------------------------------------------------

    if need_refresh?

      refresh

    end

  end

  #--------------------------------------------------------------------------
  # ● Need Refresh?
  #--------------------------------------------------------------------------
  def self.need_refresh?

    return true if @last_hp != player.hp

    return true if @last_mp != player.sp

    return true if @last_gold != FO_Data_Manager.gold

    return true if @last_level != player.level

    return false

  end

  #--------------------------------------------------------------------------
  # ● Refresh
  #--------------------------------------------------------------------------
  def self.refresh

    return if player.nil?

    #--------------------------------------------------------------------------
    # Cache
    #--------------------------------------------------------------------------

    @last_hp =
      player.hp

    @last_mp =
      player.sp

    @last_gold =
      FO_Data_Manager.gold

    @last_level =
      player.level

    #--------------------------------------------------------------------------
    # Limpa bitmap
    #--------------------------------------------------------------------------

    @bitmap.clear

    #--------------------------------------------------------------------------
    # Fundo
    #--------------------------------------------------------------------------

    @bitmap.fill_rect(
      0,
      0,
      260,
      90,
      Color.new(0,0,0,160)
    )

    #--------------------------------------------------------------------------
    # Fonte
    #--------------------------------------------------------------------------

    font =
      @bitmap.font

    font.name =
      "Arial"

    font.size =
      18

    font.bold =
      true

    font.color =
      Color.new(255,255,255)

    #--------------------------------------------------------------------------
    # Nome
    #--------------------------------------------------------------------------

    @bitmap.draw_text(
      10,
      5,
      240,
      24,
      player.name
    )

    #--------------------------------------------------------------------------
    # Level
    #--------------------------------------------------------------------------

    @bitmap.draw_text(
      180,
      5,
      70,
      24,
      "Lv #{player.level}",
      2
    )

    #--------------------------------------------------------------------------
    # HP Fundo
    #--------------------------------------------------------------------------

    @bitmap.fill_rect(
      10,
      35,
      240,
      14,
      Color.new(60,60,60)
    )

    #--------------------------------------------------------------------------
    # HP Rate
    #--------------------------------------------------------------------------

    hp_rate =
      player.hp.to_f /
      player.maxhp.to_f

    hp_width =
      (240 * hp_rate).to_i

    #--------------------------------------------------------------------------
    # HP Bar
    #--------------------------------------------------------------------------

    @bitmap.fill_rect(
      10,
      35,
      hp_width,
      14,
      Color.new(220,60,60)
    )

    #--------------------------------------------------------------------------
    # HP Texto
    #--------------------------------------------------------------------------

    @bitmap.draw_text(
      10,
      30,
      240,
      24,
      "#{player.hp}/#{player.maxhp}",
      1
    )

    #--------------------------------------------------------------------------
    # MP Fundo
    #--------------------------------------------------------------------------

    @bitmap.fill_rect(
      10,
      60,
      240,
      10,
      Color.new(60,60,60)
    )

    #--------------------------------------------------------------------------
    # MP Rate
    #--------------------------------------------------------------------------

    mp_rate =
      player.sp.to_f /
      player.maxsp.to_f

    mp_width =
      (240 * mp_rate).to_i

    #--------------------------------------------------------------------------
    # MP Bar
    #--------------------------------------------------------------------------

    @bitmap.fill_rect(
      10,
      60,
      mp_width,
      10,
      Color.new(60,120,255)
    )

    #--------------------------------------------------------------------------
    # Gold
    #--------------------------------------------------------------------------

    @bitmap.draw_text(
      10,
      68,
      240,
      20,
      "Gold: #{FO_Data_Manager.gold}"
    )

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------
  def self.dispose

    #--------------------------------------------------------------------------
    # Bitmap
    #--------------------------------------------------------------------------

    if @bitmap != nil

      unless @bitmap.disposed?

        @bitmap.dispose

      end

    end

    #--------------------------------------------------------------------------
    # Sprite
    #--------------------------------------------------------------------------

    if @sprite != nil

      unless @sprite.disposed?

        @sprite.dispose

      end

    end

  end

end

#==============================================================================
# ■ Registro no System Manager
#==============================================================================

FO_System_Manager.register(
  FO_HUD
)

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_HUD.start

FO_CORE.debug(
  "FO_HUD_v3 carregado com sucesso"
)
