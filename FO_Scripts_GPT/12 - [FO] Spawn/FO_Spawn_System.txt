  #--------------------------------------------------------------------------

  #--------------------------------------------------------------------------

  #   {
  #     :event_id => TEMPLATE,
  #     :amount => QUANTIDADE

  SPAWNS = {

    1 => [

      {
        :event_id => 5,
        :amount => 3
      },
      {
        :event_id => 2,
        :amount => 3
      }


    ]

  }

end

#==============================================================================
# ■ FO_Spawn_Manager
#==============================================================================

class FO_Spawn_Manager

  #--------------------------------------------------------------------------
  # ● Setup mapa
  #--------------------------------------------------------------------------
  def self.setup_map_spawns

    map_id = $game_map.map_id

    return unless FO_Spawn_System::SPAWNS.has_key?(map_id)

    spawns = FO_Spawn_System::SPAWNS[map_id]

    for spawn_data in spawns

      template_event_id = spawn_data[:event_id]

      amount = spawn_data[:amount]

      create_spawn_group(
        template_event_id,
        amount
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Criar grupo
  #--------------------------------------------------------------------------
  def self.create_spawn_group(template_event_id, amount)

    template = $game_map.events[template_event_id]

    return if template.nil?

    #--------------------------------------------------------------------------
    # ● Esconde template original
    #--------------------------------------------------------------------------

    template.transparent = true

    #--------------------------------------------------------------------------
    # ● Cria clones
    #--------------------------------------------------------------------------

    for i in 1..amount

      create_clone(template)

    end

  end

  #--------------------------------------------------------------------------
  # ● Criar clone
  #--------------------------------------------------------------------------
  def self.create_clone(template)

    #--------------------------------------------------------------------------
    # ● Novo ID
    #--------------------------------------------------------------------------

    new_id = ($game_map.events.keys.max || 0) + 1

#--------------------------------------------------------------------------
# ● Clona RPG::Event original
#--------------------------------------------------------------------------

original_event = template.instance_variable_get(:@event)

event_data = Marshal.load(
  Marshal.dump(original_event)
)

    event_data.id = new_id

    #--------------------------------------------------------------------------
    # ● Posição aleatória
    #--------------------------------------------------------------------------

    x = rand($game_map.width)
    y = rand($game_map.height)

    tries = 30

    while tries > 0

      if $game_map.passable?(x, y, 2)

        break

      end

      x = rand($game_map.width)
      y = rand($game_map.height)

      tries -= 1

    end

    #--------------------------------------------------------------------------
    # ● Cria evento
    #--------------------------------------------------------------------------

    cloned_event = Game_Event.new(
      $game_map.map_id,
      event_data
    )

    cloned_event.moveto(x, y)

    #--------------------------------------------------------------------------
    # ● Adiciona no mapa
    #--------------------------------------------------------------------------

    $game_map.events[new_id] = cloned_event

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_spawn_system_main_v10 main

  #--------------------------------------------------------------------------
  # ● Main
  #--------------------------------------------------------------------------
  def main

    # Cria spawns
    FO_Spawn_Manager.setup_map_spawns

    # Original
    fo_spawn_system_main_v10

  end

end
