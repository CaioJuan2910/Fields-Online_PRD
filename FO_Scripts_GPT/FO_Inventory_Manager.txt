# ■ FO_Inventory_Manager
#==============================================================================

module FO_Inventory_Manager

  #===========================================================================
  # ● Gain Loot
  #---------------------------------------------------------------------------
  # Recebe:
  # type   => :item / :weapon / :armor / :gold
  # id     => ID database
  # amount => quantidade
  #===========================================================================

  def self.gain_loot(type, id, amount = 1)

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if amount <= 0

    #--------------------------------------------------------------------------
    # ITEM
    #--------------------------------------------------------------------------

    if type == :item

      gain_item(id, amount)

    #--------------------------------------------------------------------------
    # WEAPON
    #--------------------------------------------------------------------------

    elsif type == :weapon

      gain_weapon(id, amount)

    #--------------------------------------------------------------------------
    # ARMOR
    #--------------------------------------------------------------------------

    elsif type == :armor

      gain_armor(id, amount)

    #--------------------------------------------------------------------------
    # GOLD
    #--------------------------------------------------------------------------

    elsif type == :gold

      gain_gold(amount)

    end

  end

  #===========================================================================
  # ● Gain Item
  #===========================================================================

  def self.gain_item(item_id, amount)

    #--------------------------------------------------------------------------
    # Database
    #--------------------------------------------------------------------------

    item =
      $data_items[item_id]

    return if item.nil?

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    $game_party.gain_item(
      item_id,
      amount
    )

    #--------------------------------------------------------------------------
    # Log
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[LOOT][ITEM] #{item.name} x#{amount}"
    )

  end

  #===========================================================================
  # ● Gain Weapon
  #===========================================================================

  def self.gain_weapon(weapon_id, amount)

    #--------------------------------------------------------------------------
    # Database
    #--------------------------------------------------------------------------

    weapon =
      $data_weapons[weapon_id]

    return if weapon.nil?

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    $game_party.gain_weapon(
      weapon_id,
      amount
    )

    #--------------------------------------------------------------------------
    # Log
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[LOOT][WEAPON] #{weapon.name} x#{amount}"
    )

  end

  #===========================================================================
  # ● Gain Armor
  #===========================================================================

  def self.gain_armor(armor_id, amount)

    #--------------------------------------------------------------------------
    # Database
    #--------------------------------------------------------------------------

    armor =
      $data_armors[armor_id]

    return if armor.nil?

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    $game_party.gain_armor(
      armor_id,
      amount
    )

    #--------------------------------------------------------------------------
    # Log
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[LOOT][ARMOR] #{armor.name} x#{amount}"
    )

  end

  #===========================================================================
  # ● Gain Gold
  #===========================================================================

  def self.gain_gold(amount)

    #--------------------------------------------------------------------------
    # Entrega
    #--------------------------------------------------------------------------

    $game_party.gain_gold(amount)

    #--------------------------------------------------------------------------
    # Log
    #--------------------------------------------------------------------------

    FO_CORE.debug(
      "[LOOT][GOLD] #{amount}"
    )

  end

  #===========================================================================
  # ● Get Database Object
  #---------------------------------------------------------------------------
  # Retorna o objeto real do database
  #===========================================================================

  def self.get_object(type, id)

    #--------------------------------------------------------------------------
    # ITEM
    #--------------------------------------------------------------------------

    if type == :item

      return $data_items[id]

    #--------------------------------------------------------------------------
    # WEAPON
    #--------------------------------------------------------------------------

    elsif type == :weapon

      return $data_weapons[id]

    #--------------------------------------------------------------------------
    # ARMOR
    #--------------------------------------------------------------------------

    elsif type == :armor

      return $data_armors[id]

    end

    return nil

  end

  #===========================================================================
  # ● Get Name
  #===========================================================================

  def self.get_name(type, id)

    obj =
      get_object(type, id)

    return "Unknown" if obj.nil?

    return obj.name

  end

end
