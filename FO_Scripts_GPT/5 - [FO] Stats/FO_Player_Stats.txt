module FO_PLAYER_STATS

  #--------------------------------------------------------------------------
  # ● Pontos por level
  #--------------------------------------------------------------------------

  POINTS_PER_LEVEL = 5

  #--------------------------------------------------------------------------
  # ● Base
  #--------------------------------------------------------------------------

  BASE_STR = 10
  BASE_AGI = 10
  BASE_VIT = 10
  BASE_INT = 10
  BASE_DEX = 10

  #--------------------------------------------------------------------------
  # ● Fórmulas
  #--------------------------------------------------------------------------

  ATTACK_STR_MULTIPLIER = 2

  DEFENSE_VIT_MULTIPLIER = 2

  ACCURACY_DEX_MULTIPLIER = 2

  CRITICAL_AGI_DIVISOR = 5

  EVADE_AGI_MULTIPLIER = 1

end

#==============================================================================
# ■ FO_Player_Stats
#==============================================================================

module FO_Player_Stats

  #--------------------------------------------------------------------------
  # ■ Accessors
  #--------------------------------------------------------------------------

  class << self

    attr_accessor :str
    attr_accessor :agi
    attr_accessor :vit
    attr_accessor :int
    attr_accessor :dex

    attr_accessor :points

  end

  #--------------------------------------------------------------------------
  # ● Setup
  #--------------------------------------------------------------------------

  def self.setup

    @str =
      FO_PLAYER_STATS::BASE_STR

    @agi =
      FO_PLAYER_STATS::BASE_AGI

    @vit =
      FO_PLAYER_STATS::BASE_VIT

    @int =
      FO_PLAYER_STATS::BASE_INT

    @dex =
      FO_PLAYER_STATS::BASE_DEX

    @points = 0

  end

  #--------------------------------------------------------------------------
  # ● Refresh
  #--------------------------------------------------------------------------

  def self.refresh

    validate_stats

  end

  #--------------------------------------------------------------------------
  # ● Validate Stats
  #--------------------------------------------------------------------------

  def self.validate_stats

    @str = 1 if @str < 1
    @agi = 1 if @agi < 1
    @vit = 1 if @vit < 1
    @int = 1 if @int < 1
    @dex = 1 if @dex < 1

    @points = 0 if @points < 0

  end

  #--------------------------------------------------------------------------
  # ● Add Points
  #--------------------------------------------------------------------------

  def self.add_points(amount)

    return if amount <= 0

    @points += amount

  end

  #--------------------------------------------------------------------------
  # ● Spend Point
  #--------------------------------------------------------------------------

  def self.spend_point(stat)

    return false if @points <= 0

    case stat

    when :str

      @str += 1

    when :agi

      @agi += 1

    when :vit

      @vit += 1

    when :int

      @int += 1

    when :dex

      @dex += 1

    else

      return false

    end

    @points -= 1

    refresh

    return true

  end

  #--------------------------------------------------------------------------
  # ● Attack
  #--------------------------------------------------------------------------

  def self.attack

    value =
      (
        @str *
        FO_PLAYER_STATS::ATTACK_STR_MULTIPLIER
      )

    value += weapon_attack

    return value

  end

  #--------------------------------------------------------------------------
  # ● Defense
  #--------------------------------------------------------------------------

  def self.defense

    value =
      (
        @vit *
        FO_PLAYER_STATS::DEFENSE_VIT_MULTIPLIER
      )

    value += armor_defense

    return value

  end

  #--------------------------------------------------------------------------
  # ● Magic Attack
  #--------------------------------------------------------------------------

  def self.magic_attack

    return (
      @int * 3
    )

  end

  #--------------------------------------------------------------------------
  # ● Magic Defense
  #--------------------------------------------------------------------------

  def self.magic_defense

    return (
      @int * 2
    )

  end

  #--------------------------------------------------------------------------
  # ● Accuracy
  #--------------------------------------------------------------------------

  def self.accuracy

    value =
      75 +
      (
        @dex *
        FO_PLAYER_STATS::ACCURACY_DEX_MULTIPLIER
      )

    if value > 100

      value = 100

    end

    return value

  end

  #--------------------------------------------------------------------------
  # ● Critical
  #--------------------------------------------------------------------------

  def self.critical

    value =
      (
        @agi /
        FO_PLAYER_STATS::CRITICAL_AGI_DIVISOR
      )

    if value > 50

      value = 50

    end

    return value

  end

  #--------------------------------------------------------------------------
  # ● Evade
  #--------------------------------------------------------------------------

  def self.evade

    return (
      @agi *
      FO_PLAYER_STATS::EVADE_AGI_MULTIPLIER
    )

  end

  #--------------------------------------------------------------------------
  # ● Attack Speed
  #--------------------------------------------------------------------------

  def self.attack_speed

    speed =
      30 -
      (
        @agi / 10
      )

    if speed < 8

      speed = 8

    end

    return speed

  end

  #--------------------------------------------------------------------------
  # ● Weapon Attack
  #--------------------------------------------------------------------------

  def self.weapon_attack

    #--------------------------------------------------------------------------
    # ● Futuro:
    # ● Equipment System
    #--------------------------------------------------------------------------

    return 0

  end

  #--------------------------------------------------------------------------
  # ● Armor Defense
  #--------------------------------------------------------------------------

  def self.armor_defense

    #--------------------------------------------------------------------------
    # ● Futuro:
    # ● Equipment System
    #--------------------------------------------------------------------------

    return 0

  end

  #--------------------------------------------------------------------------
  # ● Apply Level Up
  #--------------------------------------------------------------------------

  def self.apply_level_up

    add_points(
      FO_PLAYER_STATS::POINTS_PER_LEVEL
    )

    #--------------------------------------------------------------------------
    # ● Crescimento automático
    #--------------------------------------------------------------------------

    FO_Player_Data.maxhp +=
      (@vit * 2)

    FO_Player_Data.maxmp +=
      (@int)

  end

  #--------------------------------------------------------------------------
  # ● Save Data
  #--------------------------------------------------------------------------

  def self.save_data

    return {

      :str    => @str,
      :agi    => @agi,
      :vit    => @vit,
      :int    => @int,
      :dex    => @dex,

      :points => @points

    }

  end

  #--------------------------------------------------------------------------
  # ● Load Data
  #--------------------------------------------------------------------------

  def self.load_data(data)

    return if data.nil?

    @str =
      data[:str] ||
      FO_PLAYER_STATS::BASE_STR

    @agi =
      data[:agi] ||
      FO_PLAYER_STATS::BASE_AGI

    @vit =
      data[:vit] ||
      FO_PLAYER_STATS::BASE_VIT

    @int =
      data[:int] ||
      FO_PLAYER_STATS::BASE_INT

    @dex =
      data[:dex] ||
      FO_PLAYER_STATS::BASE_DEX

    @points =
      data[:points] || 0

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Debug Stats
  #--------------------------------------------------------------------------

  def self.debug_stats

    return {

      :attack   => attack,
      :defense  => defense,

      :matk     => magic_attack,
      :mdef     => magic_defense,

      :accuracy => accuracy,
      :critical => critical,
      :evade    => evade

    }

  end

end

#==============================================================================
# ■ Inicialização
#==============================================================================

FO_Player_Stats.setup

if defined?(FO_DEBUG)

  FO_DEBUG.log(
    "FO_Player_Stats_v4 carregado com sucesso"
  )

end
