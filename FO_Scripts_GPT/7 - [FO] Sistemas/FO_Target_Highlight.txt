#==============================================================================
# ■ FO_TARGET_HIGHLIGHT
#==============================================================================

module FO_TARGET_HIGHLIGHT

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  OFFSET_X = -16

  OFFSET_Y = -16

  Z = 99998

  #--------------------------------------------------------------------------
  # ● Animação
  #--------------------------------------------------------------------------

  PULSE_SPEED = 8

  MIN_OPACITY = 120

  MAX_OPACITY = 255

  #--------------------------------------------------------------------------
  # ● Cor
  #--------------------------------------------------------------------------

  NORMAL_COLOR =
    Color.new(255,255,255)

  BOSS_COLOR =
    Color.new(255,80,80)

end

#==============================================================================
# ■ FO_Target_Highlight
#==============================================================================

module FO_Target_Highlight

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @sprite = nil

  @opacity_direction = -1

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_visibility

    update_position

    update_animation

  end

  #--------------------------------------------------------------------------
  # ● Update Visibility
  #--------------------------------------------------------------------------

  def self.update_visibility

    target =
      FO_Target_System.target

    if target.nil?

      dispose

      return

    end

    if @sprite.nil?

      create_sprite

    end

  end

  #--------------------------------------------------------------------------
  # ● Create Sprite
  #--------------------------------------------------------------------------

  def self.create_sprite

    dispose

    @sprite =
      Sprite.new

    @sprite.bitmap =
      Bitmap.new(64, 64)

    @sprite.z =
      FO_TARGET_HIGHLIGHT::Z

    draw_highlight

  end

  #--------------------------------------------------------------------------
  # ● Draw Highlight
  #--------------------------------------------------------------------------

  def self.draw_highlight

    return if @sprite.nil?

    bitmap =
      @sprite.bitmap

    bitmap.clear

    #--------------------------------------------------------------------------
    # ● Cor
    #--------------------------------------------------------------------------

    color =
      FO_TARGET_HIGHLIGHT::NORMAL_COLOR

    if FO_Target_System.target_boss?

      color =
        FO_TARGET_HIGHLIGHT::BOSS_COLOR

    end

    #--------------------------------------------------------------------------
    # ● Moldura
    #--------------------------------------------------------------------------

    # Top
    bitmap.fill_rect(
      0,
      0,
      64,
      2,
      color
    )

    # Bottom
    bitmap.fill_rect(
      0,
      62,
      64,
      2,
      color
    )

    # Left
    bitmap.fill_rect(
      0,
      0,
      2,
      64,
      color
    )

    # Right
    bitmap.fill_rect(
      62,
      0,
      2,
      64,
      color
    )

  end

  #--------------------------------------------------------------------------
  # ● Update Position
  #--------------------------------------------------------------------------

  def self.update_position

    return if @sprite.nil?

    target =
      FO_Target_System.target

    return if target.nil?

    @sprite.x =
      target.screen_x +
      FO_TARGET_HIGHLIGHT::OFFSET_X

    @sprite.y =
      target.screen_y +
      FO_TARGET_HIGHLIGHT::OFFSET_Y

  end

  #--------------------------------------------------------------------------
  # ● Update Animation
  #--------------------------------------------------------------------------

  def self.update_animation

    return if @sprite.nil?

    @sprite.opacity +=
      (
        FO_TARGET_HIGHLIGHT::PULSE_SPEED *
        @opacity_direction
      )

    #--------------------------------------------------------------------------
    # ● Limites
    #--------------------------------------------------------------------------

    if @sprite.opacity <=
       FO_TARGET_HIGHLIGHT::MIN_OPACITY

      @sprite.opacity =
        FO_TARGET_HIGHLIGHT::MIN_OPACITY

      @opacity_direction = 1

    end

    if @sprite.opacity >=
       FO_TARGET_HIGHLIGHT::MAX_OPACITY

      @sprite.opacity =
        FO_TARGET_HIGHLIGHT::MAX_OPACITY

      @opacity_direction = -1

    end

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------

  def self.dispose

    return if @sprite.nil?

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    @sprite.dispose

    @sprite = nil

  end

end

#==============================================================================
# ■ Registro no FO_ENGINE
#==============================================================================

if defined?(FO_ENGINE)

  begin

    FO_ENGINE.register_system(
      FO_Target_Highlight
    )

  rescue

  end

end
