#==============================================================================

module FO_Target_Status

  #--------------------------------------------------------------------------
  # ● Entidades
  #--------------------------------------------------------------------------

  MOB_HP = 100
  BOSS_HP = 1000

  MOB_EXP = 25
  BOSS_EXP = 250

  MOB_DAMAGE = 8
  BOSS_DAMAGE = 25

  MOB_ATTACK_SPEED = 90
  BOSS_ATTACK_SPEED = 45

  #--------------------------------------------------------------------------
  # ● Respawn
  #--------------------------------------------------------------------------

  MOB_RESPAWN = 5 * Graphics.frame_rate
  BOSS_RESPAWN = 60 * Graphics.frame_rate

  RANDOM_RESPAWN_RANGE = 6

  #--------------------------------------------------------------------------
  # ● IA
  #--------------------------------------------------------------------------

  MAX_WANDER_DISTANCE = 5

  MOB_MOVE_CHANCE = 25
  BOSS_MOVE_CHANCE = 12

  AGGRO_RANGE = 5

  MAX_FAILED_MOVES = 20

  COMBAT_TIME = 5 * Graphics.frame_rate

end

#==============================================================================
# ■ Player Status
#==============================================================================

$fo_player_level = 1

$fo_player_exp = 0
$fo_player_next_exp = 100

$fo_player_maxhp = 810
$fo_player_hp = 810

$fo_player_maxmp = 585
$fo_player_mp = 585

#==============================================================================
# ■ FO_Player_Level
#==============================================================================

module FO_Player_Level

  #--------------------------------------------------------------------------
  # ● Gain EXP
  #--------------------------------------------------------------------------
  def self.gain_exp(amount)

    $fo_player_exp += amount

    check_level_up

  end

  #--------------------------------------------------------------------------
  # ● Check Level Up
  #--------------------------------------------------------------------------
  def self.check_level_up

    while $fo_player_exp >= $fo_player_next_exp

      level_up

    end

  end

  #--------------------------------------------------------------------------
  # ● Level Up
  #--------------------------------------------------------------------------
  def self.level_up

    $fo_player_exp -= $fo_player_next_exp

    $fo_player_level += 1

    #--------------------------------------------------------------------------
    # ● Próximo level
    #--------------------------------------------------------------------------

    $fo_player_next_exp =
      ($fo_player_next_exp * 1.5).to_i

    #--------------------------------------------------------------------------
    # ● HP
    #--------------------------------------------------------------------------

    $fo_player_maxhp += 50

    $fo_player_hp = $fo_player_maxhp

    #--------------------------------------------------------------------------
    # ● MP
    #--------------------------------------------------------------------------

    $fo_player_maxmp += 25

    $fo_player_mp = $fo_player_maxmp

  end

end

#==============================================================================
# ■ Game_Event
#==============================================================================

class Game_Event < Game_Character

  attr_accessor :fo_hp
  attr_accessor :fo_maxhp
  attr_accessor :fo_entity_type
  attr_accessor :fo_dead
  attr_accessor :fo_respawn_time
  attr_accessor :fo_in_combat

  alias fo_target_status_initialize initialize
  alias fo_target_status_update update

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize(map_id, event)

    fo_target_status_initialize(map_id, event)

    @fo_spawn_x = @x
    @fo_spawn_y = @y

    setup_fo_entity

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    fo_target_status_update

    update_fo_respawn

    update_fo_combat

    update_fo_ai

    update_fo_attack

  end

  #--------------------------------------------------------------------------
  # ● Setup entidade
  #--------------------------------------------------------------------------
  def setup_fo_entity

    @fo_dead = false

    @fo_respawn_time = 0

    @fo_in_combat = false

    @fo_failed_moves = 0

    @fo_attack_cooldown = 0

    @fo_combat_time = 0

    @fo_entity_type = nil

    event_name = @event.name

    #--------------------------------------------------------------------------
    # ● Mob
    #--------------------------------------------------------------------------

    if event_name.include?("$")

      @fo_entity_type = :mob

      @fo_maxhp = FO_Target_Status::MOB_HP

      @fo_hp = @fo_maxhp

    end

    #--------------------------------------------------------------------------
    # ● Boss
    #--------------------------------------------------------------------------

    if event_name.include?("!")

      @fo_entity_type = :boss

      @fo_maxhp = FO_Target_Status::BOSS_HP

      @fo_hp = @fo_maxhp

    end

  end

  #--------------------------------------------------------------------------
  # ● Combat entity?
  #--------------------------------------------------------------------------
  def fo_combat_entity?

    return @fo_entity_type == :mob ||
           @fo_entity_type == :boss

  end

  #--------------------------------------------------------------------------
  # ● Dead?
  #--------------------------------------------------------------------------
  def fo_dead?

    return @fo_dead == true

  end

  #--------------------------------------------------------------------------
  # ● Gain EXP
  #--------------------------------------------------------------------------
  def fo_gain_exp

    return unless fo_combat_entity?

    if @fo_entity_type == :boss

      FO_Player_Level.gain_exp(
        FO_Target_Status::BOSS_EXP
      )

    else

      FO_Player_Level.gain_exp(
        FO_Target_Status::MOB_EXP
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Kill
  #--------------------------------------------------------------------------
  def fo_kill

    return if fo_dead?

    @fo_dead = true

    @fo_hp = 0

    @fo_in_combat = false

    @fo_failed_moves = 0

    @fo_combat_time = 0

    #--------------------------------------------------------------------------
    # ● EXP
    #--------------------------------------------------------------------------

    fo_gain_exp

    #--------------------------------------------------------------------------
    # ● Respawn
    #--------------------------------------------------------------------------

    if @fo_entity_type == :boss

      @fo_respawn_time =
        FO_Target_Status::BOSS_RESPAWN

    else

      @fo_respawn_time =
        FO_Target_Status::MOB_RESPAWN

    end

    #--------------------------------------------------------------------------
    # ● Limpa target
    #--------------------------------------------------------------------------

    if $fo_target == self

      $fo_target = nil
      $fo_target_name = ""
      $fo_target_type = nil
      $fo_boss_target = nil

    end

    #--------------------------------------------------------------------------
    # ● Invisível
    #--------------------------------------------------------------------------

    @transparent = true

    #--------------------------------------------------------------------------
    # ● Sem colisão
    #--------------------------------------------------------------------------

    @through = true

  end

  #--------------------------------------------------------------------------
  # ● Update respawn
  #--------------------------------------------------------------------------
  def update_fo_respawn

    return unless fo_dead?

    return if @fo_respawn_time <= 0

    @fo_respawn_time -= 1

    if @fo_respawn_time <= 0

      fo_respawn

    end

  end

  #--------------------------------------------------------------------------
  # ● Respawn
  #--------------------------------------------------------------------------
  def fo_respawn

    @fo_dead = false

    @fo_hp = @fo_maxhp

    @fo_in_combat = false

    @fo_failed_moves = 0

    @fo_combat_time = 0

    random_respawn_position

    @transparent = false

    @through = false

  end

  #--------------------------------------------------------------------------
  # ● Update combat
  #--------------------------------------------------------------------------
  def update_fo_combat

    return unless @fo_in_combat

    if $fo_target == self

      @fo_combat_time =
        FO_Target_Status::COMBAT_TIME

      return

    end

    if @fo_combat_time > 0

      @fo_combat_time -= 1

    end

    if @fo_combat_time <= 0

      @fo_in_combat = false

      @fo_failed_moves = 0

    end

  end

  #--------------------------------------------------------------------------
  # ● Update attack
  #--------------------------------------------------------------------------
  def update_fo_attack

    return if fo_dead?

    return unless fo_combat_entity?

    return unless @fo_in_combat

    if @fo_attack_cooldown > 0

      @fo_attack_cooldown -= 1

      return

    end

    distance_x = (@x - $game_player.x).abs
    distance_y = (@y - $game_player.y).abs

    return unless distance_x <= 1
    return unless distance_y <= 1

    execute_fo_attack

  end

  #--------------------------------------------------------------------------
  # ● Execute attack
  #--------------------------------------------------------------------------
  def execute_fo_attack

    damage = 0

    damage_type = :normal

    text = ""

    roll = rand(100)

    base_damage =
      @fo_entity_type == :boss ?
      FO_Target_Status::BOSS_DAMAGE :
      FO_Target_Status::MOB_DAMAGE

    #--------------------------------------------------------------------------
    # ● MISS
    #--------------------------------------------------------------------------

    if roll < 10

      text = "MISS"

      damage_type = :miss

    #--------------------------------------------------------------------------
    # ● 10%
    #--------------------------------------------------------------------------

    elsif roll < 25

      damage = (base_damage * 0.10).to_i

    #--------------------------------------------------------------------------
    # ● 50%
    #--------------------------------------------------------------------------

    elsif roll < 60

      damage = (base_damage * 0.50).to_i

    #--------------------------------------------------------------------------
    # ● 80%
    #--------------------------------------------------------------------------

    elsif roll < 85

      damage = (base_damage * 0.80).to_i

    #--------------------------------------------------------------------------
    # ● 100%
    #--------------------------------------------------------------------------

    elsif roll < 95

      damage = base_damage

    #--------------------------------------------------------------------------
    # ● Critical
    #--------------------------------------------------------------------------

    else

      critical_rate = rand(81) + 120

      damage = (
        base_damage *
        critical_rate / 100.0
      ).to_i

      damage_type = :critical

      text = "CRITICAL"

    end

    #--------------------------------------------------------------------------
    # ● Texto normal
    #--------------------------------------------------------------------------

    if damage > 0 && text == ""

      text = damage.to_s

    end

    #--------------------------------------------------------------------------
    # ● Aplica dano
    #--------------------------------------------------------------------------

    if damage > 0

      $fo_player_hp -= damage

      if $fo_player_hp < 0

        $fo_player_hp = 0

      end

    end

    #--------------------------------------------------------------------------
    # ● Floating damage
    #--------------------------------------------------------------------------

    FO_Floating_Damage.create_player(
      text,
      damage_type
    )

    #--------------------------------------------------------------------------
    # ● Cooldown
    #--------------------------------------------------------------------------

    if @fo_entity_type == :boss

      @fo_attack_cooldown =
        FO_Target_Status::BOSS_ATTACK_SPEED

    else

      @fo_attack_cooldown =
        FO_Target_Status::MOB_ATTACK_SPEED

    end

  end

  #--------------------------------------------------------------------------
  # ● Update AI
  #--------------------------------------------------------------------------
  def update_fo_ai

    return if fo_dead?

    return unless fo_combat_entity?

    return if moving?

    #--------------------------------------------------------------------------
    # ● Apenas combate
    #--------------------------------------------------------------------------

    if @fo_in_combat

      follow_player_diagonal

      return

    end

    #--------------------------------------------------------------------------
    # ● Movimento aleatório
    #--------------------------------------------------------------------------

    chance =
      @fo_entity_type == :boss ?
      FO_Target_Status::BOSS_MOVE_CHANCE :
      FO_Target_Status::MOB_MOVE_CHANCE

    return unless rand(100) < chance

    move_random

  end

  #--------------------------------------------------------------------------
  # ● Follow diagonal
  #--------------------------------------------------------------------------
  def follow_player_diagonal

    distance_x = (@x - $game_player.x).abs
    distance_y = (@y - $game_player.y).abs

    #--------------------------------------------------------------------------
    # ● Já perto
    #--------------------------------------------------------------------------

    if distance_x <= 1 && distance_y <= 1

      return

    end

    old_x = @x
    old_y = @y

    sx = @x - $game_player.x
    sy = @y - $game_player.y

    #--------------------------------------------------------------------------
    # ● Diagonal
    #--------------------------------------------------------------------------

    if sx != 0 && sy != 0

      horizontal = sx > 0 ? 4 : 6
      vertical   = sy > 0 ? 8 : 2

      move_diagonal(horizontal, vertical)

    else

      move_toward_player

    end

    #--------------------------------------------------------------------------
    # ● Falha movimento
    #--------------------------------------------------------------------------

    if @x == old_x && @y == old_y

      @fo_failed_moves += 1

    else

      @fo_failed_moves = 0

    end

    #--------------------------------------------------------------------------
    # ● Muitas falhas
    #--------------------------------------------------------------------------

    if @fo_failed_moves >=
       FO_Target_Status::MAX_FAILED_MOVES

      @fo_in_combat = false

      @fo_failed_moves = 0

    end

  end

  #--------------------------------------------------------------------------
  # ● Move diagonal
  #--------------------------------------------------------------------------
  def move_diagonal(horizontal, vertical)

    case horizontal
    when 4
      move_left
    when 6
      move_right
    end

    case vertical
    when 8
      move_up
    when 2
      move_down
    end

  end

  #--------------------------------------------------------------------------
  # ● Respawn aleatório
  #--------------------------------------------------------------------------
  def random_respawn_position

    max_range =
      FO_Target_Status::RANDOM_RESPAWN_RANGE

    tries = 20

    while tries > 0

      new_x =
        @fo_spawn_x +
        rand(max_range * 2 + 1) -
        max_range

      new_y =
        @fo_spawn_y +
        rand(max_range * 2 + 1) -
        max_range

      if new_x < 0 || new_y < 0

        tries -= 1
        next

      end

      if new_x >= $game_map.width ||
         new_y >= $game_map.height

        tries -= 1
        next

      end

      if $game_map.passable?(new_x, new_y, 2)

        moveto(new_x, new_y)

        return

      end

      tries -= 1

    end

    moveto(@fo_spawn_x, @fo_spawn_y)

  end

end
