#==============================================================================
# ■ FO_Floating_Damage_Object
#==============================================================================

class FO_Floating_Damage_Object

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize(target, text, type)

    @finished = false

    create_sprite(
      target,
      text,
      type
    )

  end

  #--------------------------------------------------------------------------
  # ● Criar sprite
  #--------------------------------------------------------------------------
  def create_sprite(target, text, type)

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if target.nil?

    #--------------------------------------------------------------------------
    # Sprite
    #--------------------------------------------------------------------------

    @sprite =
      Sprite.new

    #--------------------------------------------------------------------------
    # Bitmap
    #--------------------------------------------------------------------------

    @sprite.bitmap =
      Bitmap.new(120, 32)

    #--------------------------------------------------------------------------
    # Fonte
    #--------------------------------------------------------------------------

    font =
      @sprite.bitmap.font

    font.name =
      "Arial"

    font.size =
      22

    font.bold =
      true

    #--------------------------------------------------------------------------
    # Cor
    #--------------------------------------------------------------------------

    font.color =
      Color.new(255,255,255)

    case type

    when :critical

      font.color =
        Color.new(255,80,80)

    when :miss

      font.color =
        Color.new(180,180,180)

    end

    #--------------------------------------------------------------------------
    # Texto
    #--------------------------------------------------------------------------

    @sprite.bitmap.draw_text(
      0,
      0,
      120,
      32,
      text,
      1
    )

    #--------------------------------------------------------------------------
    # Posição
    #--------------------------------------------------------------------------

    screen_x =
      target.screen_x

    screen_y =
      target.screen_y

    @sprite.x =
      screen_x - 60

    @sprite.y =
      screen_y - 48

    #--------------------------------------------------------------------------
    # Z
    #--------------------------------------------------------------------------

    @sprite.z =
      99999

    #--------------------------------------------------------------------------
    # Opacidade
    #--------------------------------------------------------------------------

    @sprite.opacity =
      255

    #--------------------------------------------------------------------------
    # Movimento
    #--------------------------------------------------------------------------

    @move_speed =
      1

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    return if @finished

    return if @sprite.nil?

    #--------------------------------------------------------------------------
    # Movimento
    #--------------------------------------------------------------------------

    @sprite.y -=
      @move_speed

    #--------------------------------------------------------------------------
    # Fade
    #--------------------------------------------------------------------------

    @sprite.opacity -= 5

    #--------------------------------------------------------------------------
    # Finaliza
    #--------------------------------------------------------------------------

    if @sprite.opacity <= 0

      dispose

      @finished = true

    end

  end

  #--------------------------------------------------------------------------
  # ● Finished?
  #--------------------------------------------------------------------------
  def finished?

    return @finished

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------
  def dispose

    return if @sprite.nil?

    #--------------------------------------------------------------------------
    # Bitmap
    #--------------------------------------------------------------------------

    if @sprite.bitmap != nil

      @sprite.bitmap.dispose

    end

    #--------------------------------------------------------------------------
    # Sprite
    #--------------------------------------------------------------------------

    @sprite.dispose

  end

end

#==============================================================================
# ■ FO_Floating_Damage
#==============================================================================

module FO_Floating_Damage

  #--------------------------------------------------------------------------
  # ■ Floating Objects
  #--------------------------------------------------------------------------

  @objects = []

  #--------------------------------------------------------------------------
  # ● Create
  #--------------------------------------------------------------------------
  def self.create(target, text, type = :normal)

    object =
      FO_Floating_Damage_Object.new(
        target,
        text,
        type
      )

    @objects.push(object)

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    #--------------------------------------------------------------------------
    # Atualiza
    #--------------------------------------------------------------------------

    for object in @objects

      next if object.nil?

      object.update

    end

    #--------------------------------------------------------------------------
    # Remove finalizados
    #--------------------------------------------------------------------------

    @objects.delete_if do |object|

      object.finished?

    end

  end

  #--------------------------------------------------------------------------
  # ● Clear
  #--------------------------------------------------------------------------
  def self.clear

    for object in @objects

      next if object.nil?

      object.dispose

    end

    @objects.clear

  end

end

)
