module FO_PLAYER_COMBAT

  #--------------------------------------------------------------------------
  # ● Configurações
  #--------------------------------------------------------------------------

  ATTACK_KEY =
    Input::C

  ATTACK_RANGE = 1

  DEFAULT_ATTACK_SPEED = 30

end

#==============================================================================
# ■ FO_Player_Combat
#==============================================================================

module FO_Player_Combat

  #--------------------------------------------------------------------------
  # ■ Variáveis
  #--------------------------------------------------------------------------

  @attack_cooldown = 0

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------

  def self.update

    update_cooldown

    update_attack_input

  end

  #--------------------------------------------------------------------------
  # ● Update Cooldown
  #--------------------------------------------------------------------------

  def self.update_cooldown

    if @attack_cooldown > 0

      @attack_cooldown -= 1

    end

  end

  #--------------------------------------------------------------------------
  # ● Update Input
  #--------------------------------------------------------------------------

  def self.update_attack_input

    return unless Input.trigger?(
      FO_PLAYER_COMBAT::ATTACK_KEY
    )

    try_attack

  end

  #--------------------------------------------------------------------------
  # ● Try Attack
  #--------------------------------------------------------------------------

  def self.try_attack

    #--------------------------------------------------------------------------
    # ● Cooldown
    #--------------------------------------------------------------------------

    return if @attack_cooldown > 0

    #--------------------------------------------------------------------------
    # ● Target
    #--------------------------------------------------------------------------

    target =
      FO_Target_System.target

    return if target.nil?

    #--------------------------------------------------------------------------
    # ● Validação
    #--------------------------------------------------------------------------

    unless valid_target?(target)

      FO_Target_System.clear_target

      return

    end

    #--------------------------------------------------------------------------
    # ● Distância
    #--------------------------------------------------------------------------

    unless target_in_range?(target)

      return

    end

    #--------------------------------------------------------------------------
    # ● Executa combate
    #--------------------------------------------------------------------------

    FO_Combat_Manager.attack_target(
      $game_player,
      target
    )

    #--------------------------------------------------------------------------
    # ● Cooldown
    #--------------------------------------------------------------------------

    @attack_cooldown =
      attack_speed

  end

  #--------------------------------------------------------------------------
  # ● Valid Target?
  #--------------------------------------------------------------------------

  def self.valid_target?(target)

    return false if target.nil?

    return false unless target.respond_to?(
      :fo_combat_entity?
    )

    return false unless target.fo_combat_entity?

    return false if target.fo_dead?

    return false if target.transparent

    return true

  end

  #--------------------------------------------------------------------------
  # ● Target In Range?
  #--------------------------------------------------------------------------

  def self.target_in_range?(target)

    player =
      $game_player

    return false if player.nil?

    dx =
      (target.x - player.x).abs

    dy =
      (target.y - player.y).abs

    return false if dx >
                    FO_PLAYER_COMBAT::ATTACK_RANGE

    return false if dy >
                    FO_PLAYER_COMBAT::ATTACK_RANGE

    return true

  end

  #--------------------------------------------------------------------------
  # ● Attack Speed
  #--------------------------------------------------------------------------

  def self.attack_speed

    #--------------------------------------------------------------------------
    # ● Futuro:
    # ● Weapon Speed
    # ● Attack Speed Bonus
    # ● Agility Scaling
    #--------------------------------------------------------------------------

    speed =
      FO_PLAYER_COMBAT::DEFAULT_ATTACK_SPEED

    if defined?(FO_Player_Stats)

      agility =
        FO_Player_Stats.agi

      bonus =
        agility / 10

      speed -= bonus

    end

    if speed < 8

      speed = 8

    end

    return speed

  end

  #--------------------------------------------------------------------------
  # ● Cooldown Active?
  #--------------------------------------------------------------------------

  def self.cooldown_active?

    return @attack_cooldown > 0

  end

  #--------------------------------------------------------------------------
  # ● Remaining Cooldown
  #--------------------------------------------------------------------------

  def self.remaining_cooldown

    return @attack_cooldown

  end

end

#==============================================================================

#==============================================================================

if defined?(FO_ENGINE)

  begin

    FO_ENGINE.register_system(
      FO_Player_Combat
    )

  rescue

  end

end
