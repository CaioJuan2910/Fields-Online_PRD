#==============================================================================
# ■ FO_Player_Combat
#==============================================================================

module FO_Player_Combat

  #--------------------------------------------------------------------------
  # ■ Configuração
  #--------------------------------------------------------------------------

  ATTACK_KEY =
    Input::CTRL

end

#==============================================================================
# ■ FO_Player_Combat_System
#==============================================================================

module FO_Player_Combat_System

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    update_attack_input

  end

  #--------------------------------------------------------------------------
  # ● Attack Input
  #--------------------------------------------------------------------------
  def self.update_attack_input

    #--------------------------------------------------------------------------
    # Input
    #--------------------------------------------------------------------------

    return unless Input.trigger?(
      FO_Player_Combat::ATTACK_KEY
    )

    #--------------------------------------------------------------------------
    # Target
    #--------------------------------------------------------------------------

    target =
      FO_Target_System.target

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if target.nil?

    #--------------------------------------------------------------------------
    # Target válido?
    #--------------------------------------------------------------------------

    return unless FO_Target_System.valid_target?

    #--------------------------------------------------------------------------
    # Player
    #--------------------------------------------------------------------------

    attacker =
      $game_party.actors[0]

    return if attacker.nil?

    #--------------------------------------------------------------------------
    # Combat Manager
    #--------------------------------------------------------------------------

    FO_Combat_Manager.attack_target(
      attacker,
      target
    )

  end

end

#==============================================================================
# ■ Registro no System Manager
#==============================================================================

FO_System_Manager.register(
  FO_Player_Combat_System
)

#==============================================================================

FO_CORE.debug(
