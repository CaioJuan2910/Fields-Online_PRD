module FO_Combat_Manager

  #--------------------------------------------------------------------------
  # ● Attack Target
  #--------------------------------------------------------------------------
  def self.attack_target(attacker, target)

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    return if attacker.nil?

    return if target.nil?

    #--------------------------------------------------------------------------
    # Combat Entity?
    #--------------------------------------------------------------------------

    return unless target.respond_to?(
      :fo_combat_entity?
    )

    return unless target.fo_combat_entity?

    #--------------------------------------------------------------------------
    # Morto?
    #--------------------------------------------------------------------------

    return if target.fo_dead?

    return if target.fo_hp <= 0

    #--------------------------------------------------------------------------
    # Calcula dano
    #--------------------------------------------------------------------------

    result =
      calculate_damage(
        attacker,
        target
      )

    #--------------------------------------------------------------------------
    # Resultado
    #--------------------------------------------------------------------------

    damage =
      result[:damage]

    text =
      result[:text]

    damage_type =
      result[:type]

    #--------------------------------------------------------------------------
    # MISS
    #--------------------------------------------------------------------------

    if damage <= 0

      FO_Floating_Damage.create(
        target,
        text,
        damage_type
      )

      return

    end

    #--------------------------------------------------------------------------
    # Combat State
    #--------------------------------------------------------------------------

    target.fo_in_combat =
      true

    #--------------------------------------------------------------------------
    # Reseta movimento IA
    #--------------------------------------------------------------------------

    target.instance_variable_set(
      :@fo_failed_moves,
      0
    )

    #--------------------------------------------------------------------------
    # Aplica dano
    #--------------------------------------------------------------------------

    apply_damage(
      target,
      damage
    )

    #--------------------------------------------------------------------------
    # Floating
    #--------------------------------------------------------------------------

    FO_Floating_Damage.create(
      target,
      text,
      damage_type
    )

    #--------------------------------------------------------------------------
    # Morte
    #--------------------------------------------------------------------------

    if target.fo_hp <= 0

      process_death(
        attacker,
        target
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Calculate Damage
  #--------------------------------------------------------------------------
  def self.calculate_damage(attacker, target)

    #--------------------------------------------------------------------------
    # Resultado
    #--------------------------------------------------------------------------

    result = {

      :damage => 0,
      :text => "",
      :type => :normal

    }

    #--------------------------------------------------------------------------
    # Accuracy
    #--------------------------------------------------------------------------

    accuracy =
      FO_Player_Stats.accuracy

    accuracy_roll =
      rand(100)

    #--------------------------------------------------------------------------
    # MISS
    #--------------------------------------------------------------------------

    if accuracy_roll > accuracy

      result[:type] =
        :miss

      result[:text] =
        "MISS"

      return result

    end

    #--------------------------------------------------------------------------
    # Attack
    #--------------------------------------------------------------------------

    attack =
      FO_Player_Stats.attack

    #--------------------------------------------------------------------------
    # Defense
    #--------------------------------------------------------------------------

    defense =
      target.fo_pdef

    #--------------------------------------------------------------------------
    # Fórmula base
    #--------------------------------------------------------------------------

    damage =
      attack - (defense / 2)

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    if damage < 1

      damage = 1

    end

    #--------------------------------------------------------------------------
    # Variance
    #--------------------------------------------------------------------------

    variance =
      rand(6)

    damage += variance

    #--------------------------------------------------------------------------
    # Critical
    #--------------------------------------------------------------------------

    critical_rate =
      FO_Player_Stats.critical

    critical_roll =
      rand(100)

    if critical_roll < critical_rate

      damage =
        (damage * 1.5).to_i

      result[:type] =
        :critical

      result[:text] =
        "CRITICAL #{damage}"

    else

      result[:text] =
        damage.to_s

    end

    #--------------------------------------------------------------------------
    # Resultado final
    #--------------------------------------------------------------------------

    result[:damage] =
      damage

    return result

  end

  #--------------------------------------------------------------------------
  # ● Apply Damage
  #--------------------------------------------------------------------------
  def self.apply_damage(target, damage)

    return if target.nil?

    #--------------------------------------------------------------------------
    # Reduz HP
    #--------------------------------------------------------------------------

    target.fo_hp -= damage

    #--------------------------------------------------------------------------
    # Segurança
    #--------------------------------------------------------------------------

    if target.fo_hp < 0

      target.fo_hp = 0

    end

  end

  #--------------------------------------------------------------------------
  # ● Process Death
  #--------------------------------------------------------------------------
  def self.process_death(attacker, target)

    return if target.nil?

    #--------------------------------------------------------------------------
    # Drop Flow
    #--------------------------------------------------------------------------

    process_drops(
      target
    )

    #--------------------------------------------------------------------------
    # Kill Flow
    #--------------------------------------------------------------------------

    process_kill(
      target
    )

  end

  #--------------------------------------------------------------------------
  # ● Process Drops
  #--------------------------------------------------------------------------
  def self.process_drops(target)

    begin

      #----------------------------------------------------------------------
      # Drops
      #----------------------------------------------------------------------

      drops =
        FO_DropManager.generate_drops(
          FO_Data_Manager.map_id,
          target.id
        )

      #----------------------------------------------------------------------
      # Entrega
      #----------------------------------------------------------------------

      FO_DropManager.deliver_drops(
        drops
      )

    rescue => e

      FO_CORE.debug(
        "[DROP ERROR] #{e}"
      )

    end

  end

  #--------------------------------------------------------------------------
  # ● Process Kill
  #--------------------------------------------------------------------------
  def self.process_kill(target)

    begin

      #----------------------------------------------------------------------
      # Estado
      #----------------------------------------------------------------------

      target.fo_kill

      #----------------------------------------------------------------------
      # Visual
      #----------------------------------------------------------------------

      target.transparent =
        true

      target.through =
        true

      #----------------------------------------------------------------------
      # Remove mapa
      #----------------------------------------------------------------------

      target.erase

    rescue => e

      FO_CORE.debug(
        "[KILL ERROR] #{e}"
      )

    end

  end

end
