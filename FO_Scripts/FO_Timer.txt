#==============================================================================
# ■ FO_Timer
#------------------------------------------------------------------------------
# Classe principal do Timer
#==============================================================================

class FO_Timer

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize

    # Hash de timers
    @timers = {}

    # Debug
    FO_CORE.debug("FO_Timer iniciado")

  end

  #--------------------------------------------------------------------------
  # ● Criar timer
  #--------------------------------------------------------------------------
  def create_timer(name, seconds)

    # Cria timer
    @timers[name] = seconds

    # Debug
    FO_CORE.debug("Timer criado: #{name} (#{seconds}s)")

  end

  #--------------------------------------------------------------------------
  # ● Verificar timer
  #--------------------------------------------------------------------------
  def timer_exists?(name)

    return @timers.has_key?(name)

  end

  #--------------------------------------------------------------------------
  # ● Pegar tempo restante
  #--------------------------------------------------------------------------
  def get_timer(name)

    return 0 unless timer_exists?(name)

    return @timers[name]

  end

  #--------------------------------------------------------------------------
  # ● Remover timer
  #--------------------------------------------------------------------------
  def remove_timer(name)

    return unless timer_exists?(name)

    @timers.delete(name)

    FO_CORE.debug("Timer removido: #{name}")

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    # Percorre todos os timers
    @timers.keys.each do |name|

      # Reduz 1 segundo
      @timers[name] -= 1

      # Debug
      FO_CORE.debug("Timer #{name}: #{@timers[name]}s")

      # Remove se terminar
      if @timers[name] <= 0

        FO_CORE.debug("Timer finalizado: #{name}")

        remove_timer(name)

      end

    end

  end

end

#==============================================================================
# ■ FO_Engine
#------------------------------------------------------------------------------
# Integra Timer à Engine
#==============================================================================

class FO_Engine

  #--------------------------------------------------------------------------
  # ● Alias Initialize
  #--------------------------------------------------------------------------
  alias fo_timer_initialize initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize

    # Executa original
    fo_timer_initialize

    # Cria sistema de timer
    @fo_timer = FO_Timer.new

    # Timer de teste
    @fo_timer.create_timer("teste_skill", 10)

  end

  #--------------------------------------------------------------------------
  # ● Alias Update Second
  #--------------------------------------------------------------------------
  alias fo_timer_update_second update_second

  #--------------------------------------------------------------------------
  # ● Update Second
  #--------------------------------------------------------------------------
  def update_second

    # Executa original
    fo_timer_update_second

    # Atualiza timers
    @fo_timer.update if @fo_timer

  end

end
