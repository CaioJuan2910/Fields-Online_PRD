module FO_DROP

  #===========================================================================
  # ■ CONFIGURAÇÕES
  #===========================================================================

  RARITY_CHANCE = {

    :comum       => 70,
    :raro        => 20,
    :gold        => 9,
    :super_raro  => 1

  }

  #===========================================================================
  # ■ TABELA DE DROP
  #===========================================================================

  DROP_TABLE = {

    #=========================================================================
    # MAPA 1
    #=========================================================================
    1 => {

      #-----------------------------------------------------------------------
      # EVENTO 5 = $Slime
      #-----------------------------------------------------------------------
      5 => {

        #---------------------------------------------------------------------
        # COMUM
        #---------------------------------------------------------------------

        :comum => [

          [:item, 1]

        ],

        #---------------------------------------------------------------------
        # RARO
        #---------------------------------------------------------------------

        :raro => [

          [:weapon, 1]

        ],

        #---------------------------------------------------------------------
        # GOLD
        #---------------------------------------------------------------------

        :gold => [

          [:armor, 1]

        ],

        #---------------------------------------------------------------------
        # SUPER RARO
        #---------------------------------------------------------------------

        :super_raro => [

          [:weapon, 2]

        ]

      }

    }

  }

end

#==============================================================================
# ■ FO_DropManager
#==============================================================================

class FO_DropManager

  #===========================================================================
  # ■ generate_drop
  #===========================================================================

  def self.generate_drop(map_id, event_id)

    #--------------------------------------------------------------------------
    # MAPA
    #--------------------------------------------------------------------------

    map_data =
      FO_DROP::DROP_TABLE[map_id]

    return nil if map_data.nil?

    #--------------------------------------------------------------------------
    # EVENTO
    #--------------------------------------------------------------------------

    drop_data =
      map_data[event_id]

    return nil if drop_data.nil?

    #--------------------------------------------------------------------------
    # RARIDADE
    #--------------------------------------------------------------------------

    rarity =
      roll_rarity

    #--------------------------------------------------------------------------
    # LISTA
    #--------------------------------------------------------------------------

    item_list =
      drop_data[rarity]

    return nil if item_list.nil?

    return nil if item_list.empty?

    #--------------------------------------------------------------------------
    # SORTEIA ITEM
    #--------------------------------------------------------------------------

    loot =
      item_list[rand(item_list.size)]

    return [loot, rarity]

  end

  #===========================================================================
  # ■ roll_rarity
  #===========================================================================

  def self.roll_rarity

    roll =
      rand(100) + 1

    current = 0

    FO_DROP::RARITY_CHANCE.each do |rarity, chance|

      current += chance

      if roll <= current

        return rarity

      end

    end

    return :comum

  end

end
