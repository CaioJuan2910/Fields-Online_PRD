#==============================================================================
# ■ FO_Player_Data
#------------------------------------------------------------------------------
# Classe principal de dados do jogador
#==============================================================================

class FO_Player_Data

  #===========================================================================
  # ■ ATTR ACCESSOR
  #===========================================================================

  attr_accessor :name
  attr_accessor :level
  attr_accessor :hp
  attr_accessor :mp
  attr_accessor :exp
  attr_accessor :gold
  attr_accessor :player_class

  #===========================================================================
  # ● Initialize
  #===========================================================================
  def initialize

    # Nome do jogador
    @name = "Caio"

    # Nível
    @level = 1

    # HP
    @hp = 100

    # MP
    @mp = 30

    # Experiência
    @exp = 0

    # Ouro
    @gold = 500

    # Classe
    @player_class = "Aventureiro"

    # Debug
    FO_CORE.debug("FO_Player_Data iniciado")

    # Mostrar dados
    show_player_data

  end

  #===========================================================================
  # ● Mostrar dados
  #===========================================================================
  def show_player_data

    FO_CORE.debug("==============================")
    FO_CORE.debug("PLAYER DATA")
    FO_CORE.debug("Nome...: #{@name}")
    FO_CORE.debug("Level..: #{@level}")
    FO_CORE.debug("HP.....: #{@hp}")
    FO_CORE.debug("MP.....: #{@mp}")
    FO_CORE.debug("EXP....: #{@exp}")
    FO_CORE.debug("Gold...: #{@gold}")
    FO_CORE.debug("Classe.: #{@player_class}")
    FO_CORE.debug("==============================")

  end

  #===========================================================================
  # ● Ganhar experiência
  #===========================================================================
  def gain_exp(amount)

    # Adiciona experiência
    @exp += amount

    FO_CORE.debug("#{@name} ganhou #{amount} EXP")

    # Verifica level up
    check_level_up

  end

  #===========================================================================
  # ● Verificar level up
  #===========================================================================
  def check_level_up

    # EXP necessária
    required_exp = @level * 100

    # Verifica level up
    if @exp >= required_exp

      # Aumenta nível
      @level += 1

      # Reseta EXP
      @exp = 0

      # Aumenta atributos
      @hp += 20
      @mp += 10

      FO_CORE.debug("LEVEL UP!")
      FO_CORE.debug("#{@name} agora está no nível #{@level}")

    end

  end

end

#==============================================================================
# ■ FO_Engine
#------------------------------------------------------------------------------
# Integra Player Data à Engine
#==============================================================================

class FO_Engine

  #--------------------------------------------------------------------------
  # ● Alias Initialize
  #--------------------------------------------------------------------------
  alias fo_player_initialize initialize

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize

    # Executa original
    fo_player_initialize

    # Cria jogador
    @player_data = FO_Player_Data.new

    # Teste de EXP
    @player_data.gain_exp(100)

  end

end
