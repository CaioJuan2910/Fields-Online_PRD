#==============================================================================
# ■ Sprite_Character
#==============================================================================

class Sprite_Character < RPG::Sprite

  #--------------------------------------------------------------------------
  # ● Alias
  #--------------------------------------------------------------------------

  alias fo_aggro_icon_initialize_v20 initialize
  alias fo_aggro_icon_update_v20 update
  alias fo_aggro_icon_dispose_v20 dispose

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize(viewport, character = nil)

    fo_aggro_icon_initialize_v20(viewport, character)

    @fo_aggro_icon = nil

    create_fo_aggro_icon

  end

  #--------------------------------------------------------------------------
  # ● Criar ícone
  #--------------------------------------------------------------------------
  def create_fo_aggro_icon

    return if self.viewport.nil?

    @fo_aggro_icon = Sprite.new(self.viewport)

    @fo_aggro_icon.bitmap =
      RPG::Cache.picture("IconFuria")

    @fo_aggro_icon.visible = false

    @fo_aggro_icon.z = self.z + 50

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    fo_aggro_icon_update_v20

    update_fo_aggro_icon

  end

  #--------------------------------------------------------------------------
  # ● Update ícone
  #--------------------------------------------------------------------------
  def update_fo_aggro_icon

    #--------------------------------------------------------------------------
    # ● Sprite existe?
    #--------------------------------------------------------------------------

    return if @fo_aggro_icon.nil?

    #--------------------------------------------------------------------------
    # ● Character existe?
    #--------------------------------------------------------------------------

    if @character.nil?

      @fo_aggro_icon.visible = false

      return

    end

    #--------------------------------------------------------------------------
    # ● Apenas eventos
    #--------------------------------------------------------------------------

    unless @character.is_a?(Game_Event)

      @fo_aggro_icon.visible = false

      return

    end

    #--------------------------------------------------------------------------
    # ● Morto?
    #--------------------------------------------------------------------------

    if @character.respond_to?(:fo_dead?)

      if @character.fo_dead?

        @fo_aggro_icon.visible = false

        return

      end

    end

    #--------------------------------------------------------------------------
    # ● Evento invisível?
    #--------------------------------------------------------------------------

    if @character.transparent

      @fo_aggro_icon.visible = false

      return

    end

    #--------------------------------------------------------------------------
    # ● Sprite invisível?
    #--------------------------------------------------------------------------

    if self.opacity <= 0

      @fo_aggro_icon.visible = false

      return

    end

#--------------------------------------------------------------------------
# ● Possui agressividade?
#--------------------------------------------------------------------------

aggro = false

if @character.instance_variables.include?("@fo_aggressive")

  aggro =
    @character.instance_variable_get(:@fo_aggressive)

end

    #--------------------------------------------------------------------------
    # ● Sem aggro
    #--------------------------------------------------------------------------

    if !aggro

      @fo_aggro_icon.visible = false

      return

    end

    #--------------------------------------------------------------------------
    # ● Mostra ícone
    #--------------------------------------------------------------------------

    @fo_aggro_icon.visible = true

    #--------------------------------------------------------------------------
    # ● Bitmap existe?
    #--------------------------------------------------------------------------

    return if @fo_aggro_icon.bitmap.nil?

    #--------------------------------------------------------------------------
    # ● Posição X
    #--------------------------------------------------------------------------

    @fo_aggro_icon.x =
      self.x -
      (@fo_aggro_icon.bitmap.width / 2)

    #--------------------------------------------------------------------------
    # ● Posição Y
    #--------------------------------------------------------------------------

    @fo_aggro_icon.y =
      self.y -
      self.oy -
      42

    #--------------------------------------------------------------------------
    # ● Z
    #--------------------------------------------------------------------------

    @fo_aggro_icon.z = self.z + 50

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------
  def dispose

    fo_aggro_icon_dispose_v20

    if @fo_aggro_icon != nil

      @fo_aggro_icon.dispose

      @fo_aggro_icon = nil

    end

  end

end
