module FO_Player_Combat

  ATTACK_KEY = Input::CTRL

  BASE_DAMAGE = 25

end

#==============================================================================
# ■ FO_Player_Combat_System
#==============================================================================

class FO_Player_Combat_System

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def self.update

    update_attack_input

  end

  #--------------------------------------------------------------------------
  # ● Input
  #--------------------------------------------------------------------------
  def self.update_attack_input

    if Input.trigger?(FO_Player_Combat::ATTACK_KEY)

      attack_target

    end

  end

  #--------------------------------------------------------------------------
  # ● Atacar alvo
  #--------------------------------------------------------------------------
  def self.attack_target

    return if $fo_target.nil?

    return unless $fo_target.respond_to?(:fo_combat_entity?)

    return unless $fo_target.fo_combat_entity?

    return if $fo_target.fo_dead?

    return if $fo_target.fo_hp <= 0

    #--------------------------------------------------------------------------
    # Sistema MMORPG
    #--------------------------------------------------------------------------

    roll = rand(100)

    damage = 0

    damage_type = :normal

    text = ""

    #--------------------------------------------------------------------------
    # MISS
    #--------------------------------------------------------------------------

    if roll < 10

      text = "MISS"

      damage_type = :miss

    #--------------------------------------------------------------------------
    # DANO 10%
    #--------------------------------------------------------------------------

    elsif roll < 25

      damage =
        (FO_Player_Combat::BASE_DAMAGE * 0.10).to_i

    #--------------------------------------------------------------------------
    # DANO 50%
    #--------------------------------------------------------------------------

    elsif roll < 60

      damage =
        (FO_Player_Combat::BASE_DAMAGE * 0.50).to_i

    #--------------------------------------------------------------------------
    # DANO 80%
    #--------------------------------------------------------------------------

    elsif roll < 85

      damage =
        (FO_Player_Combat::BASE_DAMAGE * 0.80).to_i

    #--------------------------------------------------------------------------
    # DANO 100%
    #--------------------------------------------------------------------------

    elsif roll < 95

      damage =
        FO_Player_Combat::BASE_DAMAGE

    #--------------------------------------------------------------------------
    # CRITICAL
    #--------------------------------------------------------------------------

    else

      critical_rate =
        rand(81) + 120

      damage = (
        FO_Player_Combat::BASE_DAMAGE *
        critical_rate / 100.0
      ).to_i

      damage_type = :critical

      text = "CRITICAL"

    end

    #--------------------------------------------------------------------------
    # TEXTO
    #--------------------------------------------------------------------------

    if damage > 0 && text == ""

      text = damage.to_s

    end

    #--------------------------------------------------------------------------
    # APLICA DANO
    #--------------------------------------------------------------------------

    if damage > 0

      $fo_target.fo_in_combat = true

      $fo_target.instance_variable_set(
        :@fo_failed_moves,
        0
      )

      $fo_target.fo_hp -= damage

      if $fo_target.fo_hp < 0

        $fo_target.fo_hp = 0

      end

      #--------------------------------------------------------------------------
      # MORTE
      #--------------------------------------------------------------------------

      if $fo_target.fo_hp <= 0

        dead_target = $fo_target

#----------------------------------------------------------------------
# DROP SYSTEM
#----------------------------------------------------------------------

begin

  #--------------------------------------------------------------------
  # MAP ID
  #--------------------------------------------------------------------

  map_id =
    $game_map.map_id

  #--------------------------------------------------------------------
  # EVENT ID
  #--------------------------------------------------------------------

  event_id =
    dead_target.id

  #--------------------------------------------------------------------
  # DROP
  #--------------------------------------------------------------------

  drop_result =
    FO_DropManager.generate_drop(
      map_id,
      event_id
    )

  #--------------------------------------------------------------------
  # TEM DROP?
  #--------------------------------------------------------------------

  if drop_result != nil

    loot =
      drop_result[0]

    rarity =
      drop_result[1]

    #------------------------------------------------------------------
    # DADOS
    #------------------------------------------------------------------

    loot_type =
      loot[0]

    loot_id =
      loot[1]

    database = nil

    #------------------------------------------------------------------
    # ITEM
    #------------------------------------------------------------------

    if loot_type == :item

      database =
        $data_items

    #------------------------------------------------------------------
    # WEAPON
    #------------------------------------------------------------------

    elsif loot_type == :weapon

      database =
        $data_weapons

    #------------------------------------------------------------------
    # ARMOR
    #------------------------------------------------------------------

    elsif loot_type == :armor

      database =
        $data_armors

    end

    #------------------------------------------------------------------
    # DATABASE
    #------------------------------------------------------------------

    if database != nil

      item =
        database[loot_id]

      if item != nil

        #--------------------------------------------------------------
        # ENTREGA ITEM
        #--------------------------------------------------------------

        $game_party.gain_item(
          loot_id,
          1
        )

        #--------------------------------------------------------------
        # POPUP
        #--------------------------------------------------------------

        popup_text = ""

        case rarity

        when :comum
          popup_text =
            "+ #{item.name}"

        when :raro
          popup_text =
            "RARO! #{item.name}"

        when :gold
          popup_text =
            "GOLD! #{item.name}"

        when :super_raro
          popup_text =
            "SUPER RARO! #{item.name}"

        end

        FO_Loot_Popup.show(
          popup_text
        )

      end

    end

  end

rescue => e

  print("ERRO DROP")
  print(e)

end

        #----------------------------------------------------------------------
        # Floating Damage Final
        #----------------------------------------------------------------------

        FO_Floating_Damage.create(
          dead_target,
          text,
          damage_type
        )

        #----------------------------------------------------------------------
        # MORTE
        #----------------------------------------------------------------------

        dead_target.fo_kill

        return

      end

    end

    #--------------------------------------------------------------------------
    # Floating Damage
    #--------------------------------------------------------------------------

    FO_Floating_Damage.create(
      $fo_target,
      text,
      damage_type
    )

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_player_combat_update_v51 update

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    fo_player_combat_update_v51

    FO_Player_Combat_System.update

  end

end
