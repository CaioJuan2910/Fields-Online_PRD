#==============================================================================
# ■ FO_Player_HUD
#------------------------------------------------------------------------------
# Fields Online MMORPG
#------------------------------------------------------------------------------
# HUD MMORPG dinâmica
# Compatível com RPG Maker XP ORIGINAL
#==============================================================================

class FO_Player_HUD

  #--------------------------------------------------------------------------
  # ● Initialize
  #--------------------------------------------------------------------------
  def initialize

    create_hud

  end

  #--------------------------------------------------------------------------
  # ● Create HUD
  #--------------------------------------------------------------------------
  def create_hud

    @sprite = Sprite.new

    @sprite.bitmap = Bitmap.new(320, 120)

    @sprite.x = 10
    @sprite.y = 10
    @sprite.z = 9999

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Refresh
  #--------------------------------------------------------------------------
  def refresh

    bitmap = @sprite.bitmap

    bitmap.clear

    #--------------------------------------------------------------------------
    # ● Fundo
    #--------------------------------------------------------------------------

    bitmap.fill_rect(
      0,
      0,
      320,
      120,
      Color.new(0, 0, 0, 160)
    )

    #--------------------------------------------------------------------------
    # ● Fonte
    #--------------------------------------------------------------------------

    bitmap.font.name = "Arial"

    bitmap.font.size = 18

    bitmap.font.bold = true

    bitmap.font.color = Color.new(255,255,255)

    #--------------------------------------------------------------------------
    # ● Nome
    #--------------------------------------------------------------------------

    bitmap.draw_text(
      10,
      5,
      300,
      24,
      "Fields Online"
    )

    #--------------------------------------------------------------------------
    # ● Level
    #--------------------------------------------------------------------------

    bitmap.draw_text(
      10,
      30,
      300,
      24,
      "Level: #{$fo_player_level}"
    )

    #--------------------------------------------------------------------------
    # ● HP
    #--------------------------------------------------------------------------

    hp_rate =
      $fo_player_hp.to_f /
      $fo_player_maxhp.to_f

    hp_width = (200 * hp_rate).to_i

    bitmap.fill_rect(
      10,
      58,
      200,
      16,
      Color.new(60, 60, 60)
    )

    bitmap.fill_rect(
      10,
      58,
      hp_width,
      16,
      Color.new(220, 40, 40)
    )

    bitmap.draw_text(
      220,
      54,
      90,
      24,
      "#{$fo_player_hp}/#{$fo_player_maxhp}"
    )

    #--------------------------------------------------------------------------
    # ● MP
    #--------------------------------------------------------------------------

    mp_rate =
      $fo_player_mp.to_f /
      $fo_player_maxmp.to_f

    mp_width = (200 * mp_rate).to_i

    bitmap.fill_rect(
      10,
      82,
      200,
      16,
      Color.new(60, 60, 60)
    )

    bitmap.fill_rect(
      10,
      82,
      mp_width,
      16,
      Color.new(40, 120, 255)
    )

    bitmap.draw_text(
      220,
      78,
      90,
      24,
      "#{$fo_player_mp}/#{$fo_player_maxmp}"
    )

    #--------------------------------------------------------------------------
    # ● EXP
    #--------------------------------------------------------------------------

    exp_rate =
      $fo_player_exp.to_f /
      $fo_player_next_exp.to_f

    exp_width = (300 * exp_rate).to_i

    bitmap.fill_rect(
      10,
      105,
      300,
      8,
      Color.new(60, 60, 60)
    )

    bitmap.fill_rect(
      10,
      105,
      exp_width,
      8,
      Color.new(255, 220, 40)
    )

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    refresh

  end

  #--------------------------------------------------------------------------
  # ● Dispose
  #--------------------------------------------------------------------------
  def dispose

    @sprite.bitmap.dispose

    @sprite.dispose

  end

end

#==============================================================================
# ■ Scene_Map
#==============================================================================

class Scene_Map

  alias fo_player_hud_main_v30 main
  alias fo_player_hud_update_v30 update

  #--------------------------------------------------------------------------
  # ● Main
  #--------------------------------------------------------------------------
  def main

    @fo_player_hud = FO_Player_HUD.new

    fo_player_hud_main_v30

    @fo_player_hud.dispose

  end

  #--------------------------------------------------------------------------
  # ● Update
  #--------------------------------------------------------------------------
  def update

    fo_player_hud_update_v30

    @fo_player_hud.update

  end

end
